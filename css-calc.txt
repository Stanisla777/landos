# 1. Удалите несовместимые пакеты
npm uninstall @maxflex/v-number vue-slider-component vue2-dropzone vue-property-decorator

# 2. Установите альтернативы (если нужны)
npm install @vueuse/core  # Универсальные утилиты для Vue 3

# 3. Переустановите зависимости с флагом
npm install --legacy-peer-deps

# 4. Очистите кэш Rspack
rmdir /s /q node_modules\.rspack 2>nul

# 5. Запустите сборку
npm run start
