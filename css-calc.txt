preloadImages({ state }) {
    // Собираем все URL изображений из всех историй
    const imageUrls = [];
    
    state.STORIES.DETAIL.forEach(story => {
      story.slides.forEach(slide => {
        if (slide.image.url) {
          imageUrls.push(slide.image.url);
        }
        if (slide.video.preimg) {
          imageUrls.push(slide.video.preimg);
        }
      });
    });
    
    // Предзагружаем изображения
    imageUrls.forEach(url => {
      const img = new Image();
      img.src = url;
    });
    
  
  },



 store.dispatch('axiosGetParametr').then(() => {
      // После загрузки данных предзагружаем изображения
      store.dispatch('preloadImages');
    });
