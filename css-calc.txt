SimpleRatingUniversal.vue
<template>
  <div class="form-rating__wr-form js--form-rating-universal js--validate-required js--validate-radio-group"
       :data-count-stars="countStars"
       data-required-hint="Пожалуйста, поделитесь своими достижениями">
    <div class="form-rating__wr-grade" ref="formRating">
      <div
        v-for="(starId, index) in arrayStar"
        :key="starId"
        class="form-rating__grade-item-universal"
      >
        <input
          class="js--grade-item"
          type="radio"
          :for-raiting="arrayStar.length - index"
          :id="starId"
          @click="changeGrade($event, index)"
        />
        <label :for="starId"></label>
      </div>
    </div>
    <input
      class="js--form-rating-hidden-input"
      type="hidden"
      ref="ratingInputHidden"
      :value="rate"
    />
  </div>
</template>

<script>
export default {
  name: 'SimpleRatingUniversal',
  props: {
    instanceId: {
      type: String,
      required: true
    },
    countStars: {
      type: Number,
      default: 5
    },
    maxStars: {
      type: Number,
      default: 10
    }
  },
  data() {
    return {
      arrayStar: [],
      rate: 0
    };
  },
  methods: {
    changeGrade(event, index) {
      const element = event.currentTarget;
      const parent = this.$refs.formRating;
      this.rate = this.arrayStar.length - index;

      const arrayStar = parent.querySelectorAll('.js--grade-item');
      for (let i = 0; i < arrayStar.length - this.rate; i++) {
        arrayStar[i].checked = false;
      }
      for (let i = 0; i < this.rate; i++) {
        arrayStar[arrayStar.length - 1 - i].checked = true;
      }

      this.$refs.ratingInputHidden.value = element.getAttribute('for-raiting');

      // Удаляем ошибку если была
      const container = element.closest('.js--form-rating-universal');
      if (container && container.querySelector('.js--input__error_radio-group')) {
        container.querySelector('.js--input__error_radio-group').remove();
        container.classList.remove('input_error');
      }
    },

    handleExternalClick() {
      const arrayRadio = this.$refs.formRating.querySelectorAll('.js--grade-item');
      let countCheck = 0;
      for (let radio of arrayRadio) {
        if (radio.checked) countCheck++;
      }
      // Можно добавить логику проверки, если нужно
    }
  },
  mounted() {
    // Генерируем массив звёзд с учётом instanceId
    this.arrayStar = Array.from(
      { length: this.countStars },
      (_, i) => `simple-rating-${this.instanceId}-${i + 1}`
    );

    // Подключаем обработчик внешней кнопки, если она есть
    const externalButton = document.querySelector('.js--another-thing-new-foc .js--btn-universal-form');
    if (externalButton) {
      this.externalClickHandler = this.handleExternalClick.bind(this);
      externalButton.addEventListener('click', this.externalClickHandler);
    }
  },
  beforeUnmount() {
    const externalButton = document.querySelector('.js--another-thing-new-foc .js--btn-universal-form');
    if (externalButton) {
      externalButton.removeEventListener('click', this.externalClickHandler);
    }
  }
};
</script>
