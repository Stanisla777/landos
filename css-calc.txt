if (data?.url) {
  state.btn_load = false;
  commit('setDownloadUrl', data.url);
  commit('clearPolling');

  // Программное скачивание
  const link = document.createElement('a');
  link.href = data.url;
  link.download = ''; // имя файла (опционально)
  link.style.display = 'none';
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}
