<script>
export default {
  name: 'TelegramBlock',
  props: {
    instanceId: { type: String, required: true },
    qr: { type: String, default: '' },
    mobilicon: { type: String, default: '' },
    link: { type: String, default: '' },
    description: { type: String, default: '' }
  },
  data() {
    return {
      active: false,
      isHidden: true,
      screenWidth: window.innerWidth,
      _resizeHandler: null // Для удаления слушателя
    };
  },
  computed: {
    cookieKey() { return `telegram_${this.instanceId}`; }
  },
  methods: {
    // Обновляем ширину экрана
    updateScreenWidth() {
      this.screenWidth = window.innerWidth;
    },

    // ✅ Показываем блок (убираем mobile-hidden)
    showBlock() {
      this.isHidden = false; // Было true — это была ошибка!
    },

    // Скрываем блок (добавляем mobile-hidden) — для ресайза
    hideOnMobile() {
      if (this.screenWidth <= 480) {
        this.isHidden = true;
      }
    },

    hideBlock() {
      this.setCookie(this.cookieKey, true, 1);
      this.active = false;
    },

    isMobile() {
      return this.screenWidth <= 480;
    },

    getCookieCookies(name) {
      const matches = document.cookie.match(new RegExp(
        `(?:^|; )${name.replace(/([\.$?*|{}\(\)\[\]\\\/\+^])/g, '\\$1')}=([^;]*)`
      ));
      return matches ? decodeURIComponent(matches[1]) : undefined;
    },

    setCookie(key, value, days) {
      const expires = new Date();
      expires.setTime(expires.getTime() + days * 24 * 60 * 60 * 1000);
      document.cookie = `${key}=${value};path=/;expires=${expires.toUTCString()}`;
    },

    getCookie(key) {
      const keyValue = document.cookie.match(`(^|;) ?${key}=([^;]*)(;|$)`);
      return keyValue ? keyValue[2] : null;
    }
  },

  watch: {
    // Реагируем на изменение ширины
    screenWidth(newWidth) {
      if (newWidth <= 480) {
        // Перешли на мобильный → скрываем блок (он появится позже по таймеру/событию)
        this.isHidden = true;
      } else {
        // Перешли на десктоп → показываем сразу
        this.isHidden = false;
      }
    }
  },

  mounted() {
    // Если пользователь уже закрыл Telegram вручную — ничего не делаем
    if (this.getCookie(this.cookieKey)) {
      return;
    }

    this.active = true;

    // === Логика отображения ===
    const cookiesClosed = this.getCookieCookies('cookiewindowclosed');
    const isMobileNow = window.innerWidth <= 480;

    if (isMobileNow) {
      if (cookiesClosed === 'yes') {
        // Куки есть → показываем сразу
        this.isHidden = false;
      } else {
        // Куки нет → ждём событие от Cookies-приложения
        window.addEventListener('cookies-closed-mobile', this.showBlock);
        
        // ✅ Добавляем resize-слушатель ТОЛЬКО если он нужен
        this._resizeHandler = this.updateScreenWidth;
        window.addEventListener('resize', this._resizeHandler);
      }
    } else {
      // Десктоп → показываем сразу
      this.isHidden = false;
    }
  },

  beforeUnmount() {
    // ✅ Чистим ВСЕ слушатели
    window.removeEventListener('cookies-closed-mobile', this.showBlock);
    if (this._resizeHandler) {
      window.removeEventListener('resize', this._resizeHandler);
    }
  }
};
</script>
