 isCalendarInitialized: false

showCalendar(el){
  const element = el.currentTarget;
  const calendarContainer = element.closest('.js--credit-calendar-input').querySelector('.js__vanilla-calendar-calc');
  
  // Показываем контейнер
  calendarContainer.classList.add('active');
  
  // ✅ Инициализируем календарь ПОСЛЕ того, как DOM обновился
  this.$nextTick(() => {
    if (!this.isCalendarInitialized) {
      // Первая инициализация
      this.initPluginCalendarVanilla();
      this.isCalendarInitialized = true;
    } else if (this.calendar && typeof this.calendar.update === 'function') {
      // Если календарь уже инициализирован — обновляем (если библиотека поддерживает)
      this.calendar.update();
    } else {
      // Если update нет — переинициализируем
      this.calendar = null;
      this.initPluginCalendarVanilla();
    }
  });
},
