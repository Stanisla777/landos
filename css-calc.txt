<template>
  <div class="test-cotainer-answer" :class="{ mandatory: req }">
    <div v-if="circle_number" class="test-window__number" :data-color="[`${item_key}`]">
      <p><span>{{ item_key }}</span>/<span>{{ count_answers }}</span></p>
    </div>
    <p class="test-window__question" v-html="question"></p>
    <div class="test-item">
      <div class="test-window__col-variant test-window__height-limitation">
        <div
          v-for="(answer, key) in variants"
          :key="key"
          class="test-window__item test-window__radio-item test-window__col-variant-item"
        >
          <div v-for="subAnswer in answer.DROPDOWN" :key="subAnswer.MESSAGE">
            <div
              class="test-window__item test-window__radio-item test-window__col-variant-item"
              @click="pass_clickVariant($event, subAnswer.FIELD_WIDTH, (subAnswer['~FIELD_PARAM'] || subAnswer.FIELD_PARAM), subAnswer.COLOR, subAnswer.NAME_INPUT, subAnswer.VALUE_INPUT)"
              :class="[subAnswer.FIELD_WIDTH == 1 ? 'test-wintdow__item' : 'test-winfdow__item']"
            >
              <input type="radio" :name="`radio-${item_key}`" value="">
              <p class="fs-1rem test-window__possible-answer" v-html="subAnswer.MESSAGE"></p>
            </div>
          </div>
        </div>
      </div>
    </div>
    <button
      v-if="btn_answer"
      class="test-window__btn btn btn-slide-sub"
      :class="{ unact: req }"
      @click="pass_clickButtonAnswer"
    >
      Ответить
    </button>
  </div>
</template>

<script>
export default {
  name: 'QuestionWindowBlock',
  props: [
    'question', 'variants', 'item_key', 'btn_answer', 'circle_number',
    'count_answers', 'required', 'current_answer'
  ],
  data() {
    return {
      req: null,
    };
  },
  methods: {
    pass_clickVariant(elem, param_1, param_2, param_3, param_4, param_5) {
      const element = elem.currentTarget;
      this.$emit('event_clickVariant', [element, elem, param_1, param_2, param_3, param_4, param_5]);
    },
    pass_clickButtonAnswer(elem) {
      const element = elem.currentTarget;
      this.$emit('event_clickButtonAnswer', [element]);
    },
    requiredButton() {
      this.req = this.required === "Y";
    },
  },
  mounted() {
    this.requiredButton();
  },
};
</script>
