function positionDropdown(trigger, dropdown) {
  const rect = trigger.getBoundingClientRect();
  const viewportWidth = window.innerWidth;

  const isMobileOrTablet = isMobileDevice();
  const isNarrow = window.matchMedia('(max-width: 480px)').matches;
  const isMobileShare = dropdown.classList.contains('mobile-share');

  if (isMobileOrTablet || isNarrow || isMobileShare) {
    return;
  }

  // ✅ Динамическое измерение ширины popup (без мигания)
  const originalDisplay = dropdown.style.display;
  const originalOpacity = dropdown.style.opacity;
  const originalTransform = dropdown.style.transform;

  dropdown.style.display = 'block';
  dropdown.style.opacity = '0';
  dropdown.style.transform = 'none'; // сброс transform для точного измерения

  const dropdownWidth = dropdown.getBoundingClientRect().width;

  // Восстанавливаем
  dropdown.style.display = originalDisplay;
  dropdown.style.opacity = originalOpacity;
  dropdown.style.transform = originalTransform;

  const top = rect.top + rect.height + 10;
  let left = rect.left;

  // ✅ Если не помещается — прижимаем popup к ПРАВОМУ КРАЮ экрана
  if (left + dropdownWidth > viewportWidth) {
    left = viewportWidth - dropdownWidth;
  }

  // Защита от ухода за левый край
  if (left < 0) left = 10;

  dropdown.style.transform = `translate(${left}px, ${top}px)`; // ← применяется здесь
}
