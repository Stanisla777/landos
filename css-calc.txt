// switcher-choice-story.js
export default function initSwitcherChoice() {
  const parentContainer = document.querySelector('.js--for-swither-container');
  
  if (!parentContainer) return;

  // Вешаем ОДИН слушатель на родительский контейнер
  parentContainer.addEventListener('click', (event) => {
    // Ищем ближайшего родителя с классом таба, на который кликнули
    // closest работает даже если кликнули на текст или иконку внутри таба
    const clickedTab = event.target.closest('.ds-switcher__item');
    
    if (clickedTab) {
      // 🎯 МЫ ПОЙМАЛИ КЛИК!
      console.log('Клик по табу!', clickedTab);
      
      // Пример: как получить индекс или ID таба
      // (зависит от того, как DsSwitcher рендерит HTML, возможно там будет data-атрибут)
      const tabIndex = Array.from(clickedTab.parentNode.children).indexOf(clickedTab);
      console.log('Индекс нажатого таба:', tabIndex);
      
      // ЗДЕСЬ ВАША ЛОГИКА:
      // 1. Убираем активный класс у всех табов
      // 2. Добавляем активный класс нажатому табу
      // 3. Скрываем все блоки с контентом из PHP
      // 4. Показываем блок с контентом, соответствующий tabIndex
    }
  });
}
