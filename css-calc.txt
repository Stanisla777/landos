export default function testNbuild() {
  const testElements = document.querySelectorAll('test-n');
  
  testElements.forEach((item) => {
    // Проверяем :id (Vue binding) ИЛИ id (обычный атрибут)
    const id = item.getAttribute(':id') || item.getAttribute('id');
    
    // То же самое для param, если используется :param
    const dataParam = item.getAttribute(':param') || item.getAttribute('param');

    // Защита: если нет ни id, ни :id — пропускаем с предупреждением
    if (!id) {
      console.warn('Пропущен <test-n> без id/:id:', item);
      return;
    }

    // Создаём новый контейнер
    const vueContainer = document.createElement('div');
    vueContainer.setAttribute('data-vue-app', 'test-n');
    vueContainer.setAttribute('data-id', id);
    
    if (dataParam) {
      vueContainer.setAttribute('data-param', dataParam);
    }

    // Переносим контент (если внутри были дети)
    vueContainer.innerHTML = item.innerHTML;

    // Заменяем элемент
    item.parentNode?.replaceChild(vueContainer, item);
  });
}
