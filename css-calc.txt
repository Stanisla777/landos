<script>
import universalForm from '../modules/universal-form/universal-form';
import { catalogModal, modals } from '../modules/modals';

export default {
  name: 'HousingCooperativeRequest',
  props: {
    instanceId: { type: String, required: true },
    webformId: { type: String, default: '' },
    uid: { type: String, default: '' },
    dmpkitEventUid: { type: String, default: '' },
    inputZhskName: { type: String, default: '' },
    inputZhskValue: { type: String, default: '' },
    inputZhskCodeName: { type: String, default: '' },
    inputZhskCodeValue: { type: String, default: '' },
    inputFioName: { type: String, default: '' },
    inputFioValue: { type: String, default: '' },
    inputFioCaption: { type: String, default: '' },
    inputFioPlaceholder: { type: String, default: '' },
    inputFioMaskHint: { type: String, default: '' },
    inputFioRequiredHint: { type: String, default: '' },
    inputEmailName: { type: String, default: '' },
    inputEmailValue: { type: String, default: '' },
    inputEmailCaption: { type: String, default: '' },
    inputEmailPlaceholder: { type: String, default: '' },
    inputEmailMaskHint: { type: String, default: '' },
    inputEmailRequiredHint: { type: String, default: '' },
    inputPhoneName: { type: String, default: '' },
    inputPhoneValue: { type: String, default: '' },
    inputPhoneCaption: { type: String, default: '' },
    inputPhonePlaceholder: { type: String, default: '' },
    inputPhoneMaskHint: { type: String, default: '' },
    inputPhoneRequiredHint: { type: String, default: '' },
    inputRegionName: { type: String, default: '' },
    inputRegionOptions: { type: [String, Array], default: () => [] },
    inputCategoryName: { type: String, default: '' },
    inputCategoryOptions: { type: [String, Array], default: () => [] },
    inputWorkExperienceName: { type: String, default: '' },
    inputWorkExperienceOptions: { type: [String, Array], default: () => [] },
    inputUsedMaternityCapitalName: { type: String, default: '' },
    inputUsedMaternityCapitalOptions: { type: [String, Array], default: () => [] },
    inputHousingTypeName: { type: String, default: '' },
    inputHousingTypeOptions: { type: [String, Array], default: () => [] },
    inputMortgageName: { type: String, default: '' },
    inputMortgageOptions: { type: [String, Array], default: () => [] },
    inputBuildingTypeName: { type: String, default: '' },
    inputBuildingTypeOptions: { type: [String, Array], default: () => [] },
    inputBuildingAreaName: { type: String, default: '' },
    inputBuildingAreaOptions: { type: [String, Array], default: () => [] },
    inputAgreeToReceivePromoName: { type: String, default: '' },
    inputAgreeToReceivePromoValue: { type: String, default: '' },
    inputAgreeToReceivePromoCaption: { type: String, default: '' },
    inputButtonSubmitValue: { type: String, default: '' },
  },
  data() {
    return {
      // Навигация и ошибки
      show_warn: true,
      step_1: true,
      error_step_1: true,
      error_step_2: true,
      // Счётчики валидации — Шаг 1
      fullness_input: 0,
      fullness_input_name: 0,
      fullness_input_mail: 0,
      fullness_input_tel: 0,
      fullness_select: 0,
      fullness_input_radio: 0,
      // Счётчики валидации — Шаг 2
      fullness_radio: 0,
      all_checked: true,
      // Массивы из JSON-пропсов
      inputRegionOptions_array: [],
      inputCategoryOptions_array: [],
      inputWorkExperienceOptions_array: [],
      inputUsedMaternityCapitalOptions_array: [],
      inputHousingTypeOptions_array: [],
      inputMortgageOptions_array: [],
      inputBuildingTypeOptions_array: [],
      inputBuildingAreaOptions_array: [],
    };
  },
  methods: {
    // ========== LIVE SEARCH ==========
    liveSearch(event) {
      const input_search = event.currentTarget;
      this.sampleliveSearch(input_search);
    },
    sampleliveSearch(input_search) {
      const parent = input_search.closest('.js--search-container');
      if (!parent) return;
      const val = input_search.value;
      const array_label = parent.querySelectorAll('.js--searh-item');
      array_label.forEach(item => {
        const text = item.textContent;
        const reg = new RegExp(val, 'gi');
        if (!item.classList.contains('js--select-item-checkbox')) {
          let text_box = item.innerHTML;
          text_box = text_box.replace(/(<span class="highlight">|<\/span>)/gim, '');
          item.innerHTML = text_box.replace(reg, '<span class="highlight">$&</span>');
        }
        if (text.match(reg)) {
          item.style.display = 'block';
          item.classList.add('active');
        } else {
          item.style.display = 'none';
          item.classList.remove('active');
        }
      });
      const array_label_active = parent.querySelectorAll('.js--searh-item.active');
      const array_error = parent.querySelectorAll('.js--search-error');
      if (array_label_active.length === 0) {
        const wrapper_list = parent.querySelector('.js--select-list');
        if (array_error.length === 0) {
          const element_error = document.createElement('li');
          element_error.classList.add('select__list-item', 'js--search-error', 'error');
          element_error.innerHTML = 'Ничего не найдено';
          wrapper_list.append(element_error);
        }
      } else if (array_label_active.length !== 0 && array_error.length !== 0) {
        parent.querySelector('.js--search-error')?.remove();
      }
    },

    // ========== SELECT ==========
    openList(event) {
      const element = event.currentTarget;
      const container = element.closest('.js--parent-filter');
      const array__select = container?.querySelectorAll('.js--container-select.open') || [];
      if (event.target.classList.contains('open') &&
        event.target.tagName !== 'INPUT' &&
        event.target.tagName !== 'LABEL' &&
        event.target.tagName !== 'BUTTON') {
        element.querySelector('.js--select-input').blur();
        event.target.classList.remove('open');
        event.target.setAttribute('style', 'z-index:11;');
      } else if (!element.classList.contains('open') &&
        event.target.tagName !== 'LABEL' &&
        event.target.tagName !== 'BUTTON') {
        array__select.forEach(item => item.classList.remove('open'));
        element.querySelector('.js--select-input').value = '';
        this.sampleliveSearch(element.querySelector('.js--select-input'));
        setTimeout(() => element.classList.add('open'), 100);
      }
      if (element.querySelector('.js--input__error_required')) {
        element.querySelector('.js--input__error_required').remove();
      }
      element.classList.remove('input_error');
    },
    selectItem(event) {
      const element = event.currentTarget;
      const wrapper = element.closest('.js--select-list');
      const wrapperItem = wrapper.querySelectorAll('.js--select-item');
      wrapperItem.forEach(item => item.classList.remove('active-item'));
      element.classList.add('active-item');
      let text = element.textContent.trim();
      const value = element.dataset.value;
      const parent = element.closest('.js--container-select');
      if (parent) {
        if (!parent.classList.contains('js--selected')) {
          this.fullness_select += 1;
          this.fullness_input = this.fullness_input_name + this.fullness_input_tel +
            this.fullness_input_mail + this.fullness_input_radio + this.fullness_select;
        }
        parent.querySelector('.js--select-input').classList.add('active');
        parent.querySelector('.js--select-input').setAttribute('data-search', text);
        parent.classList.add('js--selected');
        parent.classList.remove('input_error');
        this.sampleliveSearch(parent.querySelector('.js--select-input'));
        parent.querySelector('.js--select-input-value').value = value;
      }
    },

    // ========== RADIO ==========
    changeRightParticipate(event) {
      const element = event?.currentTarget || event?.target;
      if (!element) return;
      const val_attr = element.value;
      if (val_attr === 'yes') {
        this.show_warn = false;
        element.closest('.js--izhs-radio-group')?.classList.add('js--selected');
        this.fullness_input_radio = 1;
      } else {
        this.show_warn = true;
        element.closest('.js--izhs-radio-group')?.classList.remove('js--selected');
        this.fullness_input_radio = 0;
      }
      this.fullness_input = this.fullness_input_name + this.fullness_input_tel +
        this.fullness_input_mail + this.fullness_input_radio + this.fullness_select;
    },
    changePolls(event) {
      const element = event.currentTarget;
      const parent = element.closest('.js--step-2');
      const array = parent.querySelectorAll('.js--izhs-radio-required');
      const array_selected = parent.querySelectorAll('.js--izhs-radio-required.js--selected');
      if (element.checked) {
        element.closest('.js--izhs-radio-group-step-2').classList.add('js--selected');
        element.closest('.js--izhs-radio-group-step-2').classList.remove('input_error');
        if (element.classList.contains('js--input_personal') && element.closest('.checkbox-stylized')) {
          element.closest('.checkbox-stylized').classList.remove('input_error');
        }
      } else {
        element.closest('.js--izhs-radio-group-step-2').classList.remove('js--selected');
      }
      if (array_selected.length + 1 === array.length) this.all_checked = true;
      this.buttonActivationLastStep();
    },
    buttonActivationLastStep() {
      const array = this.$el.querySelectorAll('.js--izhs-radio-group-step-2');
      if (array.length === 0) { this.error_step_2 = true; return; }
      this.fullness_radio = 0;
      array.forEach(item => { if (item.classList.contains('js--selected')) this.fullness_radio += 1; });
      this.error_step_2 = array.length !== this.fullness_radio;
    },

    // ========== INPUT VALIDATION ==========
    buttonActivationName(element) {
      this.fullness_input_name = element.value.length > 0 ? 1 : 0;
      this.updateFullnessInput();
    },
    buttonActivationEmail(element) {
      const emailRegex = /^[a-z0-9!#$%&'*+/=?^_`{|}~-]+(?:\.[a-z0-9!#$%&'*+/=?^_`{|}~-]+)*@(?:[a-z0-9](?:[a-z0-9-]*[a-z0-9])?\.)+[a-z0-9](?:[a-z0-9-]*[a-z0-9])?$/i;
      this.fullness_input_mail = (emailRegex.test(element.value) && element.value.length > 0) ? 1 : 0;
      this.updateFullnessInput();
    },
    buttonActivationTel(element) {
      this.fullness_input_tel = element.value.length >= 16 ? 1 : 0;
      this.updateFullnessInput();
    },
    updateFullnessInput() {
      this.fullness_input = this.fullness_input_name + this.fullness_input_tel +
        this.fullness_input_mail + this.fullness_input_radio + this.fullness_select;
      const parent = this.$el.querySelector('.js--step-1');
      if (parent) {
        const array = parent.querySelectorAll('.js--validate-mask');
        this.error_step_1 = array.length !== this.fullness_input;
      }
    },

    // ========== NAVIGATION ==========
    nextSlide() { this.step_1 = false; },
    prevSlide() { this.step_1 = true; },

    // ========== FORM SUBMISSION ==========
    sendingRequestFirstStep(event) {
      const element = event.currentTarget;
      const parent = element.closest('.js--step-1');
      const array_input = parent.querySelectorAll('.js--validate-mask');
      array_input.forEach(item => {
        if (item.classList.contains('js--validate-mask-select')) {
          if (item.value === '') item.closest('.js--izhs-select').classList.add('input_error');
        } else if (item.classList.contains('js--validate-mask')) {
          if (item.value === '') {
            item.closest('label').classList.add('input_error');
            const array_error = item.closest('label').querySelectorAll('.js--input__error_required');
            if (array_error.length === 0) {
              const element_error = document.createElement('p');
              element_error.classList.add('input__error', 'js--input__error_required');
              element_error.innerHTML = 'Поле обязательно для заполнения';
              item.closest('label').append(element_error);
            }
          }
        }
      });
    },
    sendingRequestSecondStep(event) {
      const element = event.currentTarget;
      const array = element.closest('.js--step-2').querySelectorAll('.js--izhs-radio-group-step-2');
      const array_selected_required = element.closest('.js--step-2').querySelectorAll('.js--izhs-radio-group-step-2.js--izhs-radio-required');
      const array_selected = element.closest('.js--step-2').querySelectorAll('.js--izhs-radio-group-step-2.js--selected');
      array.forEach(item => {
        if (!item.classList.contains('js--selected')) {
          item.classList.add('input_error');
          if (item.querySelector('.checkbox-stylized')?.classList.contains('required')) {
            item.querySelector('.checkbox-stylized').classList.add('input_error');
          }
        } else item.classList.remove('input_error');
      });
      this.all_checked = array_selected_required.length === array_selected.length;
    },

    // ========== UTILS ==========
    closeItemList(event) {
      if (!event.target.classList.contains('js--select-input')) {
        const array_search = event.currentTarget.querySelectorAll('.js--container-select.open');
        array_search.forEach(item => {
          if (item.querySelector('.js--select-input')?.hasAttribute('data-search') &&
            item.querySelector('.js--select-input').getAttribute('data-search').length > 0) {
            item.querySelector('.js--select-input').value = item.querySelector('.js--select-input').getAttribute('data-search');
          } else item.querySelector('.js--select-input').value = '';
          item.classList.remove('open');
        });
      }
    },
    getRadioId(name, index) { return `${name}-${index}`; },
    raidoId() {
      const array_radio = document.querySelectorAll('.js--universal-form input[type="radio"]');
      array_radio.forEach((radio, i) => {
        const name = radio.getAttribute('name');
        radio.setAttribute('id', `${name}_${i}`);
        radio.nextSibling?.setAttribute('for', `${name}_${i}`);
      });
    },
    parseJsonToArray(st) {
      if (typeof st === 'string') {
        try { return JSON.parse(st); }
        catch (e) { console.log('Ошибка парсинга JSON:', e); return st; }
      }
      return st;
    },

    // ========== ✨ НОВЫЙ МЕТОД: СБРОС К НАЧАЛЬНОМУ СОСТОЯНИЮ ✨ ==========
    resetToInitial() {
      // 1. Сброс реактивных данных
      Object.assign(this, {
        show_warn: true, step_1: true, error_step_1: true, error_step_2: true,
        fullness_input: 0, fullness_input_name: 0, fullness_input_mail: 0,
        fullness_input_tel: 0, fullness_select: 0, fullness_input_radio: 0,
        fullness_radio: 0, all_checked: true
      });

      // 2. Сброс DOM-элементов (не реактивные)
      this.$nextTick(() => {
        // Селекты
        this.$el.querySelectorAll('.js--izhs-select').forEach(select => {
          select.classList.remove('js--selected', 'input_error');
          const inp = select.querySelector('input.js--select-input');
          const hid = select.querySelector('input.js--select-input-value');
          if (inp) { inp.value = ''; inp.removeAttribute('data-search'); inp.classList.remove('active'); }
          if (hid) hid.value = '';
          select.querySelectorAll('.js--select-item.active-item').forEach(i => i.classList.remove('active-item'));
        });

        // Радио-группы Шага 2
        this.$el.querySelectorAll('.js--izhs-radio-group-step-2').forEach(g => g.classList.remove('js--selected'));

        // Радио "Нет" — Шаг 1 + триггер события для Vue
        const radioNo = this.$el.querySelector('#right_1');
        if (radioNo) {
          radioNo.checked = true;
          radioNo.dispatchEvent(new Event('change', { bubbles: true }));
        }

        // 3. Пересчёт валидации после применения изменений браузером
        setTimeout(() => {
          this.updateFullnessInput();
          this.buttonActivationLastStep();
        }, 0);
      });
    }
  },
  watch: {
    fullness_input() {
      const parent = this.$el.querySelector('.js--step-1');
      if (parent) {
        const array = parent.querySelectorAll('.js--validate-mask');
        this.error_step_1 = array.length !== this.fullness_input;
      }
    }
  },
  beforeUnmount() {
    // ✅ Очистка слушателя события
    this.$refs.formParent?.removeEventListener('vue:form-success', this.resetToInitial);
  },
  mounted() {
    // Парсинг JSON-пропсов
    this.inputRegionOptions_array = this.parseJsonToArray(this.inputRegionOptions);
    this.inputCategoryOptions_array = this.parseJsonToArray(this.inputCategoryOptions);
    this.inputWorkExperienceOptions_array = this.parseJsonToArray(this.inputWorkExperienceOptions);
    this.inputUsedMaternityCapitalOptions_array = this.parseJsonToArray(this.inputUsedMaternityCapitalOptions);
    this.inputHousingTypeOptions_array = this.parseJsonToArray(this.inputHousingTypeOptions);
    this.inputMortgageOptions_array = this.parseJsonToArray(this.inputMortgageOptions);
    this.inputBuildingTypeOptions_array = this.parseJsonToArray(this.inputBuildingTypeOptions);
    this.inputBuildingAreaOptions_array = this.parseJsonToArray(this.inputBuildingAreaOptions);

    // Установка начальных значений в инпуты
    this.$nextTick(() => {
      if (this.inputFioValue && this.$refs.fioInput) this.$refs.fioInput.value = this.inputFioValue;
      if (this.inputEmailValue && this.$refs.emailInput) this.$refs.emailInput.value = this.inputEmailValue;
      if (this.inputPhoneValue && this.$refs.phoneInput) this.$refs.phoneInput.value = this.inputPhoneValue;
    });

    // Инициализация глобальных модулей
    modals();
    catalogModal();
    universalForm();

    // ✅ Подписка на кастомное событие успешной отправки
    this.$refs.formParent?.addEventListener('vue:form-success', this.resetToInitial);
  }
};
</script>
