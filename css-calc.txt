initAgeMask() {
  this.ageMaskRef = IMask(this.$refs.ageInput, {
    mask: Number,
    min: 0,
    max: 30,
    thousandsSeparator: '', // отключаем разделители
    radix: '.',             // не используется, но лучше указать
    mapToRadix: [],         // не принимаем никакие символы как разделитель
    scale: 0,               // только целые числа
    signed: false,
    normalizeZeros: true,
    autofix: 'clamp',       // 🔑 ключевой параметр: автоматически "зажимает" значение в [min, max]
  });

  // Синхронизация с data
  this.ageMaskRef.on('accept', () => {
    this.dataFieldAge = this.ageMaskRef.value;
  });
},
