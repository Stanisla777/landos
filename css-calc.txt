export default function modalNewCourses() {
  const arrayBtn = document.querySelectorAll('.js--modal-training-courses');
  arrayBtn.forEach(btn => {
    btn.addEventListener('click', () => {
      if (btn.closest('.js--training-materials-item') && btn.closest('.js--training-materials-container')) {
        window.dispatchEvent(new CustomEvent('open-universal-modal', {
          detail: {
            htmlTooltip: btn.closest('.js--training-materials-item').dataset.htmlTooltip || '',
            htmlTooltipContainer: btn.closest('.js--training-materials-container').dataset.htmlTooltipContainer || '',
          }
        }));
      }

    })
  })

}

data() {
    return {
      showModal:false,
      modalData:{},
      parseData: {}
    }
  },

methods: {
    handleOpen(event) {
      this.modalData = event.detail || {};
      this.parseDataMethod();
      this.$nextTick(() => {
        this.setupEventListeners();
      })
      this.showModal=true
      document.body.classList.add(
        'modal-opened'
      );
    
    },
    parseDataMethod(){
      // if (!data) return;
      // Если уже объект — используем как есть
      if (typeof this.modalData.htmlTooltip === 'object' && this.modalData.htmlTooltip !== null && !Array.isArray(this.modalData.htmlTooltip)) {
        this.parseData = this.modalData.htmlTooltip
        return;
      }
      // Если строка — парсим
      if (typeof this.modalData.htmlTooltip === 'string') {
        try {
          this.parseData = JSON.parse(this.modalData.htmlTooltip);

        } catch (e) {
          console.warn('Ошибка парсинга', e);
        }
      }

      // Если уже объект — используем как есть
      if (typeof this.modalData.htmlTooltipContainer === 'object' && this.modalData.htmlTooltipContainer !== null && !Array.isArray(this.modalData.htmlTooltipContainer)) {
        this.parseData = this.modalData.htmlTooltipContainer
        return;
      }
      // Если строка — парсим
      if (typeof this.modalData.htmlTooltipContainer === 'string') {
        try {
          this.parseData = JSON.parse(this.modalData.htmlTooltipContainer);

        } catch (e) {
          console.warn('Ошибка парсинга', e);
        }
      }
    },
}

 mounted(){
    window.addEventListener('open-universal-modal', this.handleOpen);
    universalForm();
  },
  beforeUnmount() {
    window.removeEventListener('open-universal-modal', this.handleOpen);
  }
