beforeDestroy() {
  // 1. Удаляем слушатель blur
  if (this.$refs.incomeInput && typeof this.fixLeadingZeros === 'function') {
    this.$refs.incomeInput.removeEventListener('blur', this.fixLeadingZeros);
  }

  // 2. Уничтожаем маску
  if (this.maskRef) {
    this.maskRef.destroy();
  }
}
