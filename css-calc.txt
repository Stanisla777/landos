props: {
  answers: {
    type: [Object, String],  // ✅ Принимаем и объект, и строку
    default: () => ({}),
    validator: (value) => {
      // Если уже объект — ок
      if (typeof value === 'object' && value !== null) return true;
      // Если строка — пробуем распарсить
      if (typeof value === 'string') {
        try {
          const parsed = JSON.parse(value);
          return typeof parsed === 'object' && parsed !== null; // ✅ Объект или массив
        } catch (e) {
          return false;
        }
      }
      return false;
    },
  },
},

methods: {
  parseAnswerData(data) {
    if (!data) return;
    
    // Если уже объект — используем как есть
    if (typeof data === 'object' && data !== null && !Array.isArray(data)) {
      store.dispatch('ActionAnswers', data);
      return;
    }
    
    // Если строка — парсим
    if (typeof data === 'string') {
      try {
        const parsed = JSON.parse(data);
        store.dispatch('ActionAnswers', parsed);
      } catch (e) {
        console.warn('Ошибка парсинга answers:', e);
      }
    }
  },
},

mounted() {
  this.parseAnswerData(this.answers);

}
