props: {
  param: {
    type: [Object, String],
    default: undefined, // ← ключевое изменение
    validator: (value) => {
      if (value === undefined) return true; // разрешаем undefined
      if (typeof value === 'object' && value !== null) return true;
      if (typeof value === 'string') {
        try {
          const parsed = JSON.parse(value);
          return typeof parsed === 'object' && parsed !== null;
        } catch (e) { return false; }
      }
      return false;
    },
  },
}

parseParamData(data) {
  // 1. Если проп не передан вообще → грузим с сервера
  if (data === undefined || data === null) {
    this.getAxios();
    return;
  }

  // 2. Если строка → парсим
  if (typeof data === 'string') {
    if (!data.trim()) { // пустая строка ""
      this.getAxios();
      return;
    }
    try {
      this.param_test = JSON.parse(data);
      this.forDevelopment();
    } catch (e) {
      console.warn('Ошибка парсинга param:', e);
      this.getAxios(); // фоллбэк при ошибке
    }
    return;
  }

  // 3. Если объект → используем (он точно не пустой, т.к. undefined уже отфильтровали)
  if (typeof data === 'object' && !Array.isArray(data)) {
    this.param_test = data;
    this.forDevelopment();
    return;
  }

  // 4. На всякий случай (неожиданный тип)
  this.getAxios();
}
