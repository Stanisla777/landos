console.log('Parsed:', JSON.parse(Cookies.get('array_viewed_stories') || '[]'));

viewedHistory() {
  const array_slider_item = document.querySelectorAll('.stories__slider-item');
  let str_local_storage = Cookies.get('array_viewed_stories');
  
  // Защита от undefined/null/пустой строки
  if (!str_local_storage) {
    str_local_storage = '[]';
  }
  
  try {
    str_local_storage = JSON.parse(str_local_storage);
  } catch (e) {
    console.error('Ошибка парсинга cookie array_viewed_stories:', e);
    str_local_storage = [];
  }
  
  str_local_storage = str_local_storage.map(x => parseInt(x, 10));
  
  for (let i = 0; i < array_slider_item.length; i++) {
    let data = parseInt(array_slider_item[i].getAttribute('data-id'), 10);
    if (str_local_storage.includes(data) && !array_slider_item[i].classList.contains('unactive')) {
      array_slider_item[i].classList.add('unactive');
      // ... ваш код для метрики
    }
  }
}
