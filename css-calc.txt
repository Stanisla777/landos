div.js--overlay-for-logging.overlay-for-logging

.overlay-for-logging {
  position: fixed;
  top: 0;
  left: 0;
  width: 100vw;
  height: 100vh;
  background: rgba(0, 0, 0, 0.3); // полупрозрачный фон
  z-index: 999; // должен быть выше всего, кроме самого попапа
  display: none; // по умолчанию скрыт
}

// Когда активен — показываем
.overlay-for-logging.active {
  display: block;
}

// Убедитесь, что само окно выше оверлея
.header__block-for-logging {
  z-index: 1000;
}

function openHeaderBlockLogging() {
  const array_btn = document.querySelectorAll('.js--open-block-for-logging');
  const overlay = document.querySelector('.js--overlay-for-logging');

  for (let item of array_btn) {
    item.onclick = () => {
      const parent = item.closest('.js--parent-block-for-logging');
      const bodyBlock = parent.querySelector('.js--body-block-for-logging');

      bodyBlock.classList.add('active');
      if (overlay) overlay.classList.add('active');
    };
  }
}

function closeHeaderBlockLogging() {
  const array_close = document.querySelectorAll('.js--header-block-logging-close');
  const overlay = document.querySelector('.js--overlay-for-logging');

  const closePopup = () => {
    document.querySelectorAll('.js--body-block-for-logging').forEach(block => {
      block.classList.remove('active');
    });
    if (overlay) overlay.classList.remove('active');
  };

  // Закрытие по крестику
  for (let item of array_close) {
    item.onclick = closePopup;
  }

  // Закрытие по клику на оверлей
  if (overlay) {
    overlay.onclick = closePopup;
  }
}

export default function personalAccount() {
  openHeaderBlockLogging();
  closeHeaderBlockLogging();
}
