export default function disappearAdditionalBarScroll(el) {
  const headerContainer = document.querySelector('.js--another-thing-header');
  const infoBar = document.querySelector('.js--information-bar-top-page');
  const firstBlock = document.querySelector('.js--another-thing-first-block');

  // Если плашки в будущем не будет, скрипт просто ничего не делает,
  // и хедер останется в своем дефолтном состоянии из CSS.
  if (!headerContainer || !infoBar) return;

  let ticking = false;

  // Функция проверки положения плашки
  function updateHeaderPosition() {
    // Получаю координаты плашки относительно окна браузера (viewport)
    const rect = infoBar.getBoundingClientRect();

    // rect.bottom — это координата нижней границы плашки по оси Y.
    // Если она <= 0, значит нижняя граница ушла за верхний край экрана.
    if (rect.bottom <= 0) {
      headerContainer.classList.add('header-position-fixed');
      if (firstBlock) {
        firstBlock.classList.add('style-first-block');
      }

    } else {
      headerContainer.classList.remove('header-position-fixed');
      if (firstBlock) {
        firstBlock.classList.remove('style-first-block');
      }
    }

    ticking = false; // Сбрасываем флаг для следующего кадра
  }

  // Слушатель скролла с оптимизацией (throttling через requestAnimationFrame)
  window.addEventListener('scroll', () => {
    if (!ticking) {
      window.requestAnimationFrame(updateHeaderPosition);
      ticking = true;
    }
  });

  // Проверяем положение плашки сразу при загрузке страницы
  // (на случай, если пользователь обновил страницу, уже проскроллив её вниз)
  updateHeaderPosition();


}
