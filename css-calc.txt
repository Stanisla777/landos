handleBlurFix() {
  const rawValue = this.maskRef.unmaskedValue;

  if (!rawValue || rawValue === '') return;

  const num = parseFloat(rawValue);
  if (isNaN(num)) return;

  let finalNum = num;
  if (num > this.stgMax) {
    finalNum = this.stgMax;
  }

  // 🔑 Явно обрабатываем ноль: даже если было "0000", должно стать "0"
  if (finalNum === 0) {
    this.$refs.incomeInput.value = '0';
    this.dataFieldIncome = '0';
    this.$refs.incomeInput.dispatchEvent(new Event('input', { bubbles: true }));
    return;
  }

  const formatted = finalNum.toLocaleString('ru-RU', {
    minimumFractionDigits: 0,
    maximumFractionDigits: 2,
    useGrouping: true
  }).replace(/\./g, ',');

  this.$refs.incomeInput.value = formatted;
  this.$refs.incomeInput.dispatchEvent(new Event('input', { bubbles: true }));
  this.dataFieldIncome = formatted;
}
