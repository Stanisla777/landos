<script>
import { DsSwitcher } from '@ringsites/ui-components/DsSwitcher';
export default {
  name: "SwitcherComponent",
  props: {
    instanceId: {
      type: String,
      default: '',
    },
    switcherContentElements: {
      type: [Array, String],
      default: () => [],
      validator: (value) => {
        if (Array.isArray(value)) return true;
        if (typeof value === 'string') {
          try {
            const parsed = JSON.parse(value);
            return Array.isArray(parsed);
          } catch (e) {
            return false;
          }
        }
        return false;
      },
    },
  },
  data() {
    return {
      switcher:[
        // { id: '0', text: 'Банковский бизнес' },
        // { id: '1', text: 'Небанковский бизнес' },
        // { id: '2', text: 'Функции развития' },
        // { id: '3', text: 'Цифровые платформы' },
      ],
      contentModelValue:''
    }
  },
  methods:{
    parseSwitcherData(data) {
      if (!data) return;
      // Если уже объект — используем как есть
      if (typeof data === 'object' && data !== null && !Array.isArray(data)) {
        this.switcher = data
        return;
      }
      // Если строка — парсим
      if (typeof data === 'string') {
        try {
          const parsed = JSON.parse(data);
          this.switcher = parsed
        } catch (e) {
          console.warn('Ошибка парсинга answers:', e);
        }
      }
    },
  },
  mounted() {
    this.parseSwitcherData(this.switcherContentElements)
  },
  components: {DsSwitcher}

}

</script>
