showCalendar(el){
  const element = el.currentTarget;
  // ✅ Ищем контейнер календаря ВНУТРИ ближайшего .js--credit-calendar-input
  const calendarContainer = element.closest('.js--credit-calendar-input')
    .querySelector('.js__vanilla-calendar-calc');
  
  calendarContainer.classList.add('active');
  
  this.$nextTick(() => {
    if (!this.isCalendarInitialized) {
      this.initPluginCalendarVanilla();
      this.isCalendarInitialized = true;
    } else if (this.calendar && typeof this.calendar.update === 'function') {
      this.calendar.update();
    } else {
      this.calendar = null;
      this.initPluginCalendarVanilla();
    }
  });
},

initPluginCalendarVanilla(dates, month, year, contextElement = null){
  let text_footnote = this.pop_up_text
  
  // ✅ Если передан контекст (элемент, рядом с которым клик), ищем календарь относительно него
  const selector = contextElement 
    ? contextElement.closest('.js--credit-calendar-input').querySelector('.js__vanilla-calendar-calc')
    : '.js__vanilla-calendar-calc';
  
  this.calendar = new VanillaCalendar(selector, {
    settings: {
      lang: 'ru',
      range: {
        max: new Date().toISOString().slice(0, 10)
      },
      selected: {
        dates: dates,
        month: month,
        year: year
      },
      selection: {}
    },
    actions: {
      clickDay(e, dates) {
        // ... ваш существующий код ...
      }
    }
  })
  this.calendar.init();
},
