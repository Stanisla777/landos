handleBlurFix() {
  setTimeout(() => {
    if (!this.maskRef || !this.$refs.incomeInput) return;

    const unmasked = this.maskRef.unmaskedValue;
    if (!unmasked || unmasked === '0' || unmasked === '') return;

    const num = parseFloat(unmasked);
    if (isNaN(num)) return;

    let finalNum = num;
    if (num > this.stgMax) {
      finalNum = this.stgMax;
    }

    // Форматируем число как строку БЕЗ ведущих нулей
    const cleanUnmasked = String(finalNum); // например: "98"

    // Устанавливаем значение через value (а не unmaskedValue)
    this.maskRef.value = cleanUnmasked; // imask сам применит форматирование
    this.maskRef.updateValue(); // ← 🔑 КЛЮЧЕВОЙ МЕТОД

    this.dataFieldIncome = this.maskRef.value;
  }, 10);
}
