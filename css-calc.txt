export default function disappearAdditionalBarScroll(el) {
  const headerContainer = document.querySelector('.js--another-thing-header');
  const infoBar = document.querySelector('.js--information-bar-top-page');
  const firstBlock = document.querySelector('.js--another-thing-first-block');

  if (!headerContainer || !infoBar) return;

  let ticking = false;
  let isFixed = false;

  // Создаем элемент-заглушку, который будет сохранять место хедера в потоке
  const placeholder = document.createElement('div');
  placeholder.style.display = 'none';
  placeholder.style.width = '100%';
  
  // Вставляем заглушку после headerContainer
  headerContainer.parentNode.insertBefore(placeholder, headerContainer.nextSibling);

  function updateHeaderPosition() {
    const rect = infoBar.getBoundingClientRect();
    const shouldBeFixed = rect.bottom <= 0;

    // Если состояние не меняется - ничего не делаем
    if (shouldBeFixed === isFixed) {
      ticking = false;
      return;
    }

    if (shouldBeFixed) {
      // Запоминаем текущую высоту хедера перед фиксацией
      const headerHeight = headerContainer.offsetHeight;
      
      // Настраиваем заглушку
      placeholder.style.display = 'block';
      placeholder.style.height = headerHeight + 'px';
      
      // Добавляем класс фиксации
      headerContainer.classList.add('header-position-fixed');
      
      if (firstBlock) {
        firstBlock.classList.add('style-first-block');
      }
      
      isFixed = true;
    } else {
      // Убираем класс фиксации
      headerContainer.classList.remove('header-position-fixed');
      
      if (firstBlock) {
        firstBlock.classList.remove('style-first-block');
      }
      
      // Скрываем заглушку
      placeholder.style.display = 'none';
      placeholder.style.height = '0';
      
      isFixed = false;
    }

    ticking = false;
  }

  // Добавляем обработчик изменения размера окна
  // чтобы обновлять высоту заглушки при ресайзе
  function handleResize() {
    if (isFixed) {
      const headerHeight = headerContainer.offsetHeight;
      placeholder.style.height = headerHeight + 'px';
    }
  }

  window.addEventListener('scroll', () => {
    if (!ticking) {
      window.requestAnimationFrame(updateHeaderPosition);
      ticking = true;
    }
  });

  window.addEventListener('resize', handleResize);

  // Проверяем положение при загрузке
  updateHeaderPosition();
}
