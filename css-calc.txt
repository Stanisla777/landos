const CONFIG = {
  thresholds: [25, 50, 75, 100],
  eventPrefix: 'scroll_',
  throttleMs: 200,
  contentSelector: '.js--calculated-scroll-area',
  offsetBottomPx: 0
};

const state = {
  sent: new Set(),
  maxReached: 0,
  lastCheck: 0,
  geometry: null,
  observer: null
};

/**
 * Рассчитывает границы полезного контента.
 * Работает корректно на iOS/Safari благодаря scrollY + getBoundingClientRect
 */
function calculateGeometry() {
  const elements = document.querySelectorAll(CONFIG.contentSelector);
  if (!elements || elements.length === 0) return null;

  let minTop = Infinity;
  let maxBottom = -Infinity;

  elements.forEach(el => {
    const rect = el.getBoundingClientRect();
    // Пропускаем скрытые или нулевые блоки (ещё не отрендерились)
    if (rect.height <= 0 || rect.width <= 0) return;

    const absTop = window.scrollY + rect.top;
    const absBottom = absTop + rect.height;

    if (absTop < minTop) minTop = absTop;
    if (absBottom > maxBottom) maxBottom = absBottom;
  });

  if (minTop === Infinity) return null;

  return {
    start: minTop,
    end: maxBottom,
    height: maxBottom - minTop // Учитывает разрывы между блоками (в вашем случае ~40px)
  };
}

function setupResizeObserver() {
  if (!('ResizeObserver' in window)) return;
  if (state.observer) state.observer.disconnect();

  state.observer = new ResizeObserver(() => {
    // Контент изменился (картинки загрузились, Swiper инициализировался, Vue подгрузил блок)
    state.geometry = null; 
    // Следующий скролл пересчитает высоту заново
  });

  const elements = document.querySelectorAll(CONFIG.contentSelector);
  elements.forEach(el => state.observer.observe(el));
}

function sendToCleverData(eventName) {
  if (typeof window.dmpkitdl !== 'undefined' && window.dmpkitdl !== null) {
    try {
      window.dmpkitdl.push({ event: eventName });
      // console.log('[CleverData] Sent:', eventName);
    } catch (e) {
      console.warn('[ScrollDepth] Error sending to dmpkitdl:', e);
    }
  }
}

function getScrollPercent() {
  if (!state.geometry) {
    state.geometry = calculateGeometry();
  }
  
  if (!state.geometry || state.geometry.height <= 0) return 0;

  const scrollTop = window.scrollY || document.documentElement.scrollTop;
  const scrolled = scrollTop - state.geometry.start;
  
  if (scrolled < 0) return 0; // Ещё не дошли до начала инструкции

  const targetHeight = state.geometry.height - CONFIG.offsetBottomPx;
  if (targetHeight <= 0) return scrolled >= 0 ? 100 : 0;

  const percent = (scrolled / targetHeight) * 100;
  return Math.min(100, Math.max(0, Math.round(percent)));
}

function checkScrollDepth() {
  const now = Date.now();
  if (now - state.lastCheck < CONFIG.throttleMs) return;
  state.lastCheck = now;

  const percent = getScrollPercent();

  for (const threshold of CONFIG.thresholds) {
    // Отправляем только если перешли порог И этот порог ещё не отправляли
    if (threshold > state.maxReached && percent >= threshold) {
      sendToCleverData(CONFIG.eventPrefix + threshold);
      state.sent.add(threshold);
      state.maxReached = threshold;
      break;
    }
  }
}

export default function scrollDepthTrackerCleverData() {
  // Запускаем трекинг сразу, но с микро-задержкой для первоначального layout
  // Это не блокирует скролл, но даёт браузеру 100мс на отрисовку DOM
  setTimeout(() => {
    state.geometry = calculateGeometry();
    setupResizeObserver();
    checkScrollDepth(); // Проверка якорей или раннего скролла
  }, 100);

  let ticking = false;

  const onScroll = () => {
    if (!ticking) {
      window.requestAnimationFrame(() => {
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  const onResize = () => {
    state.geometry = null; // При ресайзе пересчитаем геометрию на лету
    if (!ticking) {
      window.requestAnimationFrame(() => {
        state.geometry = calculateGeometry();
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  window.addEventListener('scroll', onScroll, { passive: true });
  window.addEventListener('resize', onResize, { passive: true });

  return () => {
    window.removeEventListener('scroll', onScroll);
    window.removeEventListener('resize', onResize);
    if (state.observer) state.observer.disconnect();
  };
}
