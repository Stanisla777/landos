// дата начала выплат
mutationStartPayment(state, received_perem) {
  state.start_payment = [];
  state.start_payment.push(received_perem[0]);
  state.start_payment.push(received_perem[1]);

  state.arr_data_small_debt = [];
  state.annuity_monthly_payment_old_credid = 0;
  state.differentiated_monthly_payment = [];
  state.annuity_debt_plus_interest_old_credit = 0;
  state.annuity_overpayment_old_credit = 0;
  state.payment_annuity_list = [];
  state.payment_differentiated_list = [];
  state.dateArray = [];
  state.dateArrayPsk = [];

  // рассчитываю платежи
  const old_rate = (state.old_rate / 12) / 100;

  // формирую даты
  const month_array = [];
  const month_year = [];
  const amountOfDatesFromPast = state.credit_old_time;

  for (let i = 0; i < amountOfDatesFromPast + 1; i++) {
    const dateObj = new Date(state.start_payment[0]);
    dateObj.setMonth(dateObj.getMonth() + i);
    const month = dateObj.getUTCMonth() + 1;
    month_array.push(month);
    const year = dateObj.getUTCFullYear();
    month_year.push(year);
  }

  for (let i = 0; i < month_array.length; i++) {
    if (month_array[i] === month_array[i + 1]) {
      month_array[i] = month_array[i] - 1;
    }
  }
  for (let i = 0; i < month_array.length; i++) {
    month_array[i] === 1 ? month_array[i] = '01' : month_array[i] === 2 ? month_array[i] = '02' : month_array[i] === 3 ? month_array[i] = '03'
      : month_array[i] === 4 ? month_array[i] = '04' : month_array[i] === 5 ? month_array[i] = '05' : month_array[i] === 6 ? month_array[i] = '06'
        : month_array[i] === 7 ? month_array[i] = '07' : month_array[i] === 8 ? month_array[i] = '08' : month_array[i] === 9 ? month_array[i] = '09'
          : month_array[i] === 10 ? month_array[i] = '10' : month_array[i] === 11 ? month_array[i] = '11' : month_array[i] = '12';
  }
  for (let i = 0; i < month_array.length; i++) {
    state.dateArray.push(state.start_payment[1] + '.' + month_array[i] + '.' + month_year[i]);
  }

  const annuity_ratio = old_rate * ((1 + old_rate) ** state.credit_old_time) / (((1 + old_rate) ** state.credit_old_time) - 1);
  const annuity_monthly_payment_old_credid = (state.old_debt * annuity_ratio).toFixed(2);
  // ежемесячный платёж
  state.annuity_monthly_payment_old_credid = parseFloat(annuity_monthly_payment_old_credid);
  
  // Формирую график платежей
  let main_debt = state.old_debt;
  let monthly_payment = state.annuity_monthly_payment_old_credid;

  for (let i = 1; i <= state.credit_old_time; i++) {
    // ежемесячный платёж по процентам
    let monthly_interest_payment = main_debt * old_rate;
    let principal_payment = monthly_payment - monthly_interest_payment;
    let actual_payment = monthly_payment;
    
    // Проверка: если платёж не покрывает проценты
    if (principal_payment <= 0) {
      principal_payment = 0;
      actual_payment = monthly_interest_payment;
      monthly_interest_payment = main_debt * old_rate;
      actual_payment = monthly_interest_payment;
    }
    
    // Корректируем последний платёж
    if (i === state.credit_old_time || principal_payment >= main_debt) {
      principal_payment = main_debt;
      actual_payment = principal_payment + monthly_interest_payment;
    }
    
    // остаток долга
    main_debt = main_debt - principal_payment;
    
    // Защита от отрицательного остатка
    if (main_debt < 0) {
      actual_payment = actual_payment + main_debt;
      main_debt = 0;
    }
    
    const obj_payment = {
      payment: Math.round(Math.max(0, actual_payment) * 100) / 100,
      principal_payment: Math.round(Math.max(0, principal_payment) * 100) / 100,
      monthly_interest_payment: Math.round(Math.max(0, monthly_interest_payment) * 100) / 100,
      remainder: Math.round(main_debt * 100) / 100,
      month: state.dateArray[i]
    };
    
    state.payment_annuity_list.push(obj_payment);
    
    // Если долг выплачен, выходим
    if (main_debt <= 0) {
      state.payment_annuity_list = state.payment_annuity_list.slice(0, i);
      break;
    }
  }
  
  // Пересчитываем итоговые суммы
  let total_payment = 0;
  let total_interest = 0;
  for (let i = 0; i < state.payment_annuity_list.length; i++) {
    total_payment += state.payment_annuity_list[i].payment;
    total_interest += state.payment_annuity_list[i].monthly_interest_payment;
  }
  state.annuity_debt_plus_interest_old_credit = total_payment;
  state.annuity_overpayment_old_credit = total_interest;
},
