initMask() {
      this.maskRef = IMask(this.$refs.statementInput, {
        mask: [
          {
            mask: '$00000.00.0000',
            definitions: {
              // Разрешаем ввод и строчных, и заглавных букв
              '$': /[eеEЕ]/ 
            },
            // Хук prepare срабатывает ДО применения маски
            prepare: function (str) {
              // Заменяем заглавную латинскую E на строчную e
              // Заменяем заглавную кириллическую Е на строчную е
              // (используем replace с флагом 'g', чтобы это работало даже при вставке из буфера обмена)
              return str.replace(/[EЕ]/g, function(match) {
                return match === 'E' ? 'e' : 'е';
              });
            }
          },
          {
            mask: '00000.00.0000'
          }
            ],
        lazy: false 
      });

      this.maskRef.on('accept', () => {
        this.statementValue = this.maskRef.value;
        const unmasked = this.maskRef.unmaskedValue;
        
        // Здесь ничего менять не нужно! 
        // Так как prepare уже тихо заменил заглавные буквы на строчные, 
        // в unmaskedValue придёт уже правильный вариант, и эта проверка сработает идеально.
        this.canForm = /^([eе]\d{11}|\d{11})$/.test(unmasked);
      });
    }
