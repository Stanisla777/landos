// pdf-fonts.js
const GILROY_MEDIUM = 'AAEAAAASAQAABAAgR...'; // base64
const GILROY_SEMIBOLD = 'AAEAAAASAQAABAAgR...';

export function registerGilroy(pdfMake) {
  pdfMake.vfs = pdfMake.vfs || {};
  pdfMake.vfs['Gilroy-Medium.ttf'] = GILROY_MEDIUM;
  pdfMake.vfs['Gilroy-Semibold.ttf'] = GILROY_SEMIBOLD;

  pdfMake.fonts = {
    Gilroy: {
      normal: 'Gilroy-Medium.ttf',
      bold: 'Gilroy-Semibold.ttf',
      italics: 'Gilroy-Medium.ttf',
      bolditalics: 'Gilroy-Semibold.ttf'
    }
  };
}
