closeMobileTooltip(element) {
  // Сначала ищем в актуальном DOM — это надёжнее после ре-рендера Vue
  const activeTooltip = document.querySelector('.js--tooltip-parent.tooltip-active');
  
  // Если не нашли в DOM, пробуем использовать переданный элемент (фолбэк)
  const target = activeTooltip || (element?.closest?.('.js--tooltip-parent.tooltip-active'));
  
  if (!target) {
  
    console.log('closeMobileTooltip: не нашли активный тултип');
    return;
  }

  target.classList.remove('tooltip-active');
  
  if (typeof bodyUnlockMobileFilter === 'function') {
    bodyUnlockMobileFilter(target);
  }
},
