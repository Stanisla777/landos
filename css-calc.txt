import '../scss/style.scss';
// Регистрируем типы приложений
vueAppManager.registerAppType('form-rating', FormRatingUniversal);
vueAppManager.registerAppType('telegram-block', TelegramBlock);
vueAppManager.registerAppType('simple-rating', SimpleRatingUniversal);
vueAppManager.registerAppType('usefulness-survey', UsefulnessSurvey);

//Словарь ленивых компонентов (только те, которые грузятся по требованию)
const lazyComponents = {
  'housing-cooperative': () => import('./vue-apps/HousingCooperativeRequest/index.vue')
};

// Функция для создания экземпляра приложения
function createAppInstance(container, appType, index) {
  const instanceId = container.dataset.instanceId || `${appType}-${index}`;

  const props = { instanceId };
  Object.keys(container.dataset).forEach(key => {
    if (key !== 'vueApp' && key !== 'instanceId') {
      const camelKey = key.replace(/-([a-z])/g, g => g[1].toUpperCase());
      props[camelKey] = container.dataset[key];
    }
  });

  const containerId = `vue-app-${instanceId}`;
  container.id = containerId;

  vueAppManager.createInstance(appType, containerId, props);
}

document.addEventListener('DOMContentLoaded', async () => {

const containers = document.querySelectorAll('[data-vue-app]');

  if (containers.length === 0) {
    console.log('ℹ️ Элементы data-vue-app не найдены');
    return;
  }

  // Группируем контейнеры по типу
  const containersByType = {};
  containers.forEach(container => {
    const appType = container.dataset.vueApp;
    if (!containersByType[appType]) {
      containersByType[appType] = [];
    }
    containersByType[appType].push(container);
  });

  // Сначала загружаем ленивые компоненты, если они есть на странице
  for (const [appType, typeContainers] of Object.entries(containersByType)) {
    // Если это ленивый компонент и он ещё не зарегистрирован
    if (lazyComponents[appType] && !vueAppManager.appTypes.has(appType)) {
      console.log(`📦 Ленивая загрузка: ${appType}`);
      try {
        const module = await lazyComponents[appType]();
        vueAppManager.registerAppType(appType, module.default || module);
      } catch (error) {
        console.error(`❌ Ошибка загрузки ${appType}:`, error);
        continue;
      }
    }

    // Монтируем все контейнеры этого типа
    typeContainers.forEach((container, idx) => {
      createAppInstance(container, appType, idx);
    });
  }

  vueAppManager.info();
})

const observer = new MutationObserver(async (mutations) => {
  for (const mutation of mutations) {
    for (const node of mutation.addedNodes) {
      if (node.nodeType === 1 && node.matches?.('[data-vue-app]')) {
        const appType = node.dataset.vueApp;

        // Если это ленивый компонент и он ещё не зарегистрирован
        if (lazyComponents[appType] && !vueAppManager.appTypes.has(appType)) {
          console.log(`📦 Динамическая ленивая загрузка: ${appType}`);
          try {
            const module = await lazyComponents[appType]();
            vueAppManager.registerAppType(appType, module.default || module);
          } catch (error) {
            console.error(`❌ Ошибка загрузки ${appType}:`, error);
            continue;
          }
        }

        // Считаем сколько уже есть таких элементов для индекса
        const existingCount = document.querySelectorAll(`[data-vue-app="${appType}"]`).length;
        createAppInstance(node, appType, existingCount - 1);
      }
    }
  }
});

observer.observe(document.body, {
  childList: true,
  subtree: true
});

window.vueAppManager = vueAppManager;
