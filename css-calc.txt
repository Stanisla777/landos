в корне папки vue-apps лежит файл vueAppManager.js с таким содержанием:
import { createApp } from 'vue';

class VueAppManager {
  constructor() {
    this.appTypes = new Map();     // Зарегистрированные типы приложений
    this.instances = new Map();    // Созданные экземпляры
  }

  // Регистрация нового типа приложения
  registerAppType(type, component) {
    this.appTypes.set(type, component);
    console.log(`✅ Зарегистрирован тип приложения: ${type}`);
  }

  // Создание экземпляра приложения
  createInstance(type, containerId, props = {}) {
    const component = this.appTypes.get(type);
    if (!component) {
      console.error(`❌ Тип приложения "${type}" не зарегистрирован`);
      return null;
    }

    const container = document.getElementById(containerId);
    if (!container) {
      console.error(`❌ Контейнер #${containerId} не найден`);
      return null;
    }

    // Защита от двойной инициализации
    if (container.__vue_app__) {
      console.log(`⚠️ Экземпляр ${type} в #${containerId} уже существует`);
      return container.__vue_app__;
    }

    try {
      const app = createApp(component, props);
      app.mount(container);

      // Сохраняем информацию об экземпляре
      const instanceInfo = {
        app,
        containerId,
        type,
        props,
        createdAt: new Date().toISOString()
      };

      if (!this.instances.has(type)) {
        this.instances.set(type, new Map());
      }
      this.instances.get(type).set(containerId, instanceInfo);

      console.log(`✅ Создан экземпляр ${type} в #${containerId}`);
      return app;

    } catch (error) {
      console.error(`❌ Ошибка создания ${type}:`, error);
    }
  }

  // Получить все экземпляры определённого типа
  getInstances(type) {
    return this.instances.get(type) || new Map();
  }

  // Удалить экземпляр
  destroyInstance(type, containerId) {
    const instances = this.instances.get(type);
    if (instances && instances.has(containerId)) {
      const { app } = instances.get(containerId);
      app.unmount();
      instances.delete(containerId);
      console.log(`✅ Удалён экземпляр ${type} из #${containerId}`);
    }
  }

  // Очистить все экземпляры
  destroyAll() {
    this.instances.forEach((instances, type) => {
      instances.forEach(({ app }) => app.unmount());
    });
    this.instances.clear();
    console.log('✅ Все экземпляры удалены');
  }

  // Отладка: показать состояние менеджера
  info() {
    console.log('📊 Состояние менеджера:');
    this.instances.forEach((instances, type) => {
      console.log(`  ${type}: ${instances.size} экземпляров`);
      instances.forEach((info, id) => {
        console.log(`    - #${id} (${info.createdAt})`);
      });
    });
  }
}

export default new VueAppManager();
