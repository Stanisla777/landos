            this.ocenkaMobiErrorText = 'Недостаточно данных, попробуйте изменить запрос';
            this.showError();
            return false;
          }
          if (response.data === null || response.status !== 200) {
            if (response.data.description) {
              this.ocenkaMobiErrorText = response.data.description;
            }
            this.showError();
            return false;
          }

          if (response.data !== null && response.status === 200 && response.data.result !== null) {
            this.resetData();

            const result = response.data.result;
            const stats = result.stats || {};
            const bld = result.bld || {};

            // === 1. Адрес: берём из bld.address или flat.address ===
            this.bullet.address = result.flat?.address || result.bld?.address || result.address || '';
            console.log('Установлен адрес:', this.bullet.address); // для отладки

            // === 2. Цены ===
            this.bullet.priceMin = stats.median ?? stats.average ?? stats.min ?? 0;
            this.bullet.price = stats.price ?? 0;
            this.bullet.price_range = Array.isArray(stats.priceRange) ? stats.priceRange : [0, 0];

            // === 3. Валидация ===
            if (!this.bullet.price || !this.bullet.price_range.length) {
              this.ocenkaMobiErrorText = 'Недостаточно данных о цене. Попробуйте изменить запрос';
              this.showError();
              return;
            }

            // === 4. Станции ===
            if (bld.stations && Array.isArray(bld.stations)) {
              this.bullet.stations = bld.stations;
            }

            // === 5. Показываем и заполняем ===
            console.log('Показываем bullet, адрес:', this.bullet.address);
            this.bullet.showBullet = true;
            this.setData(result);
          }

        })
        .catch((error) => {
          console.log(error);
          if(error&&(error.request.status===403 || error.request.status===405)){
            this.ocenkaMobiErrorText = 'Сервер не отвечает, попробуйте перезагрузить страницу';
            this.showErrorSearch=false
            setTimeout(()=>{
              this.showError();
            },500)

          }
          this.ocenkaMobiErrorText = this.ocenkaMobiErrorTextDefault;
          this.showError();
        })
        .then(() => {
          this.showPreloader = false;
        });
    },

    getInfraApi(token) {
      this.showPreloader = true;
      this.ocenkaMobiErrorText = this.ocenkaMobiErrorTextDefault;
      const grecaptchaInput = document.getElementById('mobi-ocenka-subscribe');
      grecaptchaInput.value = token;
      localaxios.get(localApiUrl, {
        params: {
          method: 'search/infra',
          lat: this.bldPos.lat,
          lng: this.bldPos.lng,
          'smart-token': token
        },
        headers: {
          'X-Bitrix-Csrf-Token': window.BX.bitrix_sessid(),
        }
      })
        .then((response) => {
          if ((response.data === null) || (response.status !== 200) || (!(Object.prototype.hasOwnProperty.call(response.data, 'result') && Object.prototype.hasOwnProperty.call(response.data.result, 'social')))) {
            if (response.data.description) {
              this.ocenkaMobiErrorText = response.data.description;
            }
            this.showError();
            return false;
          }
          this.social = response.data.result.social;
          this.setInfraBulletData(response.data.result.social[0].items[0]);
          this.handlerInfraBullet(true);
          this.setObjectToMap();
        })
        .catch((error) => {
          this.ocenkaMobiErrorText = this.ocenkaMobiErrorTextDefault;
          this.showError();
          // handle error
          console.log(error);
        })
        .then(() => {
          this.showPreloader = false;
        });
    },

      const pos = data.bld?.pos;
      if (pos && typeof pos.lat === 'number' && typeof pos.lng === 'number') {
        const coords = [pos.lat, pos.lng];

        if (typeof window.ymaps !== 'undefined' && window.ymaps.ready) {
          // Карта уже загружена — инициализируем сразу
          window.ymaps.ready(() => {
            this.initMap(coords);
          });
        } else {
          // Карта ещё не готова — пробуем позже
          setTimeout(() => {
            if (typeof window.ymaps !== 'undefined' && window.ymaps.ready) {
              window.ymaps.ready(() => {
                this.initMap(coords);
              });
            }
          }, 1000);
        }
      }

      // === Остальные данные ===
      this.bldPhotos = data.bld?.photos || null;

      // Вкладка "О доме" (id: 0)
      const aboutBld = this.tabs[0]?.content || [];

        aboutBld[0].data = data.bld?.cadNum || '';

        aboutBld[7].data = data.bld?.elevatorCount || '';
        aboutBld[8].data = data.bld?.uFloors || '';
        aboutBld[9].data = data.bld?.bldType || '';
        aboutBld[10].data = data.bld?.wallMaterial || '';
        aboutBld[11].data = data.bld?.wearout || '';
        aboutBld[12].data = data.bld?.bldProject || '';
        aboutBld[13].data = data.bld?.floorMaterial || '';

      }

      if (data.flat) {
        this.flat = true;
        const aboutFlat = this.tabs[1]?.content || [];
        if (aboutFlat.length) {
          aboutFlat[0].data = data.flat.premisesId || '';
          aboutFlat[1].data = data.flat.parcelId || '';

          aboutFlat[6].data = data.flat.floor || '';

          }
        }
      }
    },

    initMap(dataPos) {
      if (typeof window.ymaps === 'undefined' || !window.ymaps.ready) {
        console.warn('️ Яндекс.Карты не загружены, карта недоступна');
        return;
      }
      // eslint-disable-next-line no-unused-vars
      const newAdress = this.getNewAdress;
      // eslint-disable-next-line no-undef
      let pos = dataPos;
      let zoomData = 16;
      let showObj = true;
      if (!pos) {
        pos = centerPos;
        zoomData = 12;
        showObj = false;
      }
      // eslint-disable-next-line no-unused-vars
      const newBullet = this.bullet;
      // eslint-disable-next-line no-undef,no-unused-vars

      const ActiveFilter = this.idActiveFilter;

        // eslint-disable-next-line no-undef,no-unused-vars
        myPlacemark = {
          geometry: {
            type: 'Point',
            coordinates: pos
          },
          properties: {
            // iconContent: this.bullet.address
            iconContent: newBullet.address
          }
        }, {
          preset: 'islands#blackStretchyIcon',
          iconColor: 'green'
        }
        // eslint-disable-next-line no-shadow,no-undef,no-const-assign
        map = new ymaps.Map('map', {
          center: pos,
          controls: ['smallMapDefaultSet'],
          zoom: zoomData
        });
        if (showObj) {
          // eslint-disable-next-line no-undef
          map.geoObjects.add(new ymaps.GeoObject({
            geometry: {
              type: 'Point',
              coordinates: pos
            },
            properties: {
              iconContent: newBullet.address
            }
          }, {
            preset: 'islands#blackStretchyIcon',
            iconColor: 'green'
          }));
        }
        map.container.fitToViewport();
        map.geoObjects.events.add('click', (e) => {
          if (typeof socialE[ActiveFilter] !== 'undefined') {
            const objectsArr = socialE[ActiveFilter].items;
            const thisPlacemark = e.get('target');
            // eslint-disable-next-line no-underscore-dangle

              ballonNumber = 0;

            this.bulletInfra = bulletContent;
            this.setRoute(objectsArr[ballonNumber].pos);
          }
        });

        map.events.add('click', (e) => {
          const coords = e.get('coords');
          map.geoObjects.removeAll();

    async setRoute(objectsPos) {

      map.geoObjects.remove(multiRoute);
      // eslint-disable-next-line no-undef

        referencePoints: [
          [this.bldPos.lat, this.bldPos.lng],
          [objectsPos.lat, objectsPos.lng]
        ],
        params: {
          routingMode: 'pedestrian',
          reverseGeocoding: false
        }
      }, {
        wayPointVisible: false,
        routeActiveStrokeWidth: 6,
        routeActiveStrokeStyle: 'dot',
        routeActiveStrokeColor: '#FED141'
      });
      map.geoObjects.add(multiRoute);
      this.bulletInfra.link = `https://yandex.ru/maps/?ll=${objectsPos.lng},${objectsPos.lat}&z=16&l=map`;

      setTimeout(this.getRouteData, 600);

    compare(a, b) {
      if (a.type === b.type) {
        return 0;
      }
      if (b.type === 'Город') {
        return 1;
      }
      if (a.type === 'Город') {
        return -1;
      }
      return 0;
    },
    handlerAddress() {
      if (this.inputs[0].value.length > 2) {
        axios.jsonp(this.streetApi + this.inputs[0].value)
          .then((res) => {
            this.searchList = [];
            if (res.result) {
              res.result.sort(this.compare);
              for (let i = 0; i < res.result.length; i += 1) {
                let { fullName } = res.result[i];
                if (fullName) {
                  fullName = fullName.split(',');
                  fullName.shift();
                  for (let j = 0; j < fullName.length; j += 1) {
                    fullName[j] = fullName[j].trim();
                  }
                  this.searchList.push(fullName.join(', '));
                  if (this.searchList.length) {
                    this.handlerSelectAddress(true);
                  }
                }
              }
            }
          })
          .catch((error) => {
            console.log(error);
          });
      }
    },
