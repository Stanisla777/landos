.then((res) => {
            if (res.data.code === 200 && res.data.result) {
              // Сохраняем ссылки в Vuex
              store.dispatch('ActionAnswerLink', res.data.result.answerLink);
              store.dispatch('ActionAnswersId', res.data.result.answersId);
              store.dispatch('ActionAnswersExpired',0)
            }
            if (res.data.code!==200) {
              if (res.data.description) {
                store.dispatch('ActionDescriptionAfterSand', res.data.description);
              }
            }
            if (res.data.code) {
              this.answer_code = res.data.code;
            }
            resolve();
          })
