/**
 * Управляет видимостью контейнера стрелок и состоянием вложенного слайдера
 */
function updateActiveArrows(parent, mainSwiper, activeIndex) {
  const currentSlide = mainSwiper.slides[activeIndex];
  if (!currentSlide) return;

  const nestedSwiperEl = currentSlide.querySelector('.js--slider-construtor');
  if (!nestedSwiperEl) return;

  const dataArrowsAttr = nestedSwiperEl.getAttribute('data-arrows');
  if (!dataArrowsAttr) return;

  let arrowConfig;
  try {
    arrowConfig = JSON.parse(dataArrowsAttr);
  } catch (e) {
    return;
  }

  const arrowClass = arrowConfig.arrow_left || arrowConfig.arrow_right;
  if (!arrowClass) return;

  const className = arrowClass.replace(/^\./, '');
  const targetArrow = parent.querySelector(`.${className}`);
  if (!targetArrow) return;

  const activeArrows = targetArrow.closest('.js--swiper-arrow-prise-container');
  if (!activeArrows) return;

  // 1. Скрываем все контейнеры, сбрасываем залипания
  parent.querySelectorAll('.js--swiper-arrow-prise-container').forEach(container => {
    container.classList.add('display-none');
    container.classList.remove('unactive');
  });
  activeArrows.classList.remove('display-none');

  // 2. Ждём, пока конструктор инициализирует вложенный Swiper
  let attempts = 0;
  const checkAndSetup = () => {
    attempts++;
    if (nestedSwiperEl.swiper) {
      applyNestedSliderLogic(nestedSwiperEl.swiper, activeArrows);
      return;
    }
    if (attempts < 40) { // Ждём до 4 секунд (40 * 100ms)
      setTimeout(checkAndSetup, 100);
    }
  };
  checkAndSetup();
}

/**
 * Применяет логику вложенного слайдера: включает свайп, ставит nested, управляет .unactive
 */
function applyNestedSliderLogic(swiper, activeArrows) {
  // 🔑 КРИТИЧЕСКИ ВАЖНО для вложенных слайдеров:
  swiper.params.nested = true;
  swiper.enable(); // Перезапускаем обработчики мыши/тача (без этого свайп часто ломается)
  swiper.update(); // Корректно пересчитываем размеры после показа

  // Даём браузеру применить стили и проставить disabled-классы
  setTimeout(() => {
    const slides = swiper.slides;
    const spv = swiper.params.slidesPerView || 1;
    const total = slides ? slides.length : 0;

    // Считаем неактивные стрелки
    let disabledCount = 0;
    activeArrows.querySelectorAll('.swiper-slide-arrow').forEach(btn => {
      if (btn.classList.contains('swiper-button-disabled') || 
          btn.classList.contains('disabled-non-disappear') || 
          btn.disabled) {
        disabledCount++;
      }
    });

    // Логика скрытия: 1 слайд / всё помещается / обе стрелки отключены
    if (total <= 1 || total <= spv || disabledCount >= 2) {
      activeArrows.classList.add('unactive');
      swiper.allowTouchMove = false;
    } else {
      activeArrows.classList.remove('unactive');
      swiper.allowTouchMove = true;
    }
  }, 60);
}
