computed: {
  // === УПРОЩЁННО: Адрес валиден, если Dadata нашла хотя бы одну подсказку ===
  isAddressValid() {
    return this.isAddressFromDadata;
  },

  isFormValid() {
    // 1. Все обязательные поля заполнены
    const allFilled = this.inputs.every(inp => !inp.validate || inp.value);
    
    // 2. Адрес "похож на валидный" (Dadata что-то нашла)
    return allFilled && this.isAddressValid;
  },
  
  // ... остальные computed без изменений ...
},

-----------------
handlerSendForm(argument) {
  // 1. Валидация полей (пустые поля)
  this.validateAllInputs();
  
  if (!this.isValid) {
    return; // 🛑 Пустые поля
  }
  
  // 2. Проверка: нашла ли Dadata хоть что-то по этому адресу?
  if (!this.isAddressValid) {
    this.inputs[0].isNotValid = true;
    this.ocenkaMobiErrorText = 'Введите корректный адрес (город, улица, дом)';
    this.showError();
    return; // 🛑 Адрес не похож на валидный
  }
  
  // ✅ Всё ок → отправляем (даже если пользователь не выбрал подсказку явно)
  this.captchaInit();
  this.send_argument = argument;
  this.btnSendPress = true;
  
  const grecaptchaInput = document.getElementById('mobi-ocenka-subscribe');
  let recaptchaKey = null;
  if (typeof conf !== 'undefined') recaptchaKey = conf.smartcaptcha_key;
  
  if ((typeof recaptchaKey !== 'undefined') && recaptchaKey !== null) {
    window.smartCaptcha.execute(this.captcha_id, argument);
  } else {
    this.apiConnect('');
  }
},
