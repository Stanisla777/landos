// Кастомный импортёр для SASS – перенаправляет пути из node_modules в src/assets
const sassImporter = {
  canonicalize(url, options) {
    // Проверяем, что обрабатываемый файл находится внутри node_modules
    // и что путь начинается с ../../assets/
    if (options.from && options.from.includes('node_modules') && url.startsWith('../../assets/')) {
      // Заменяем '../../assets/' на абсолютный путь к папке src/assets/
      const newPath = path.resolve(__dirname, 'src', url.replace(/^\.\.\/\.\.\//, ''));
      // Возвращаем canonical URL (формат file://)
      return new URL(`file://${newPath}`);
    }
    // Если путь не подходит – возвращаем null (SASS будет искать стандартным способом)
    return null;
  },
  load(canonicalUrl) {
    // Загружаем содержимое файла по полученному canonical URL
    const fs = require('fs');
    const filePath = canonicalUrl.pathname;
    if (fs.existsSync(filePath)) {
      return {
        contents: fs.readFileSync(filePath, 'utf8'),
        syntax: 'scss' // указываем синтаксис
      };
    }
    return null;
  }
};

}
