store.js
const getDefaultConfirmationResponse = () => ({
  application_status_code: '',
  application_status_header: '',
  application_status_description: '',
  application_status_data: '',
  application_status_can_download:false
});

export default createStore({
  state:{
    confirmation_code_response: getDefaultConfirmationResponse(),
  }

getters:{
  CONFIRMATIONCODERESPONSE(state){
      return state.confirmation_code_response
    }
},
mutation:{

mutationSendCodeConfirmation(state,received_perem){

      axios({
        method: 'post',
        url: '/api/local/program450/statusrequest/check/',
        headers: {
          'Content-Type': 'application/json; charset=UTF-8',
          'X-Bitrix-Csrf-Token': window.BX?.bitrix_sessid?.(),
        },
        data: {
          code: received_perem,
        },
      })
        .then((response) => {
          const data = response.data.result;
          if (data.data.status) {
            //код статуса заявления
            state.confirmation_code_response['application_status_code'] = data.data.status;
          }
          if (data.canDownloadPrintForm) {
            //будет на 3 щаге кнопка скачивания или нет
            state.confirmation_code_response['application_status_can_download'] = data.canDownloadPrintForm;
          }
})
.then(() => {
   state.btn_load = false
   state.step_one = false
   state.step_two = false
   state.step_three = true
   state.timer_begin = false

        })

}


}
