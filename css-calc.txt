// vh.js
let isInitialized = false;

export default function initVH() {
  if (isInitialized) return;
  isInitialized = true;

  const setVH = () => {
    document.documentElement.style.setProperty('--vh', `${window.innerHeight * 0.01}px`);
  };

  setVH();
  window.addEventListener('resize', setVH);
}
