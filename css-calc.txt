mounted() {
  this.initMask();

  if (this.answers && this.region_not === false) {
    const incomeStr = this.answers.familyIncome; // "390000.45"
    
    // Парсим как число
    const valueNumber = parseFloat(incomeStr);
    
    if (!isNaN(valueNumber) && valueNumber >= 0) {
      // ✅ ПЕРЕДАЁМ ЧИСЛО ЧЕРЕЗ unmaskedValue
      this.maskRef.unmaskedValue = valueNumber;
    }
  }
}
