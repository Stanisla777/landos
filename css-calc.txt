<template>
  <div class="test-n-slider" :class="addClasses">
    <Slider
      v-model="answerValue"
      ref="slider"
      :marks="marksConfig"
      :labels="true"
      :min="minValue"
      :max="maxValue"
      :disabled="isAnswered"
      :tooltip="false"
      :height="15"
      :dot-size="32"
      :dot-style="{ height: '32px', width: '32px', background: '#fff', border: '2px solid #000' }"
      :label-style="{ marginTop: '20px', fontSize: '14px' }"
      :process-style="{ background: '#ccc' }"
      @update:modelValue="onSliderMove"
      @change="onSliderRelease"
    />
    <div class="test-n-slider__value">{{ currentValueLabel }}</div>
  </div>
</template>

<script>
import Slider from '@vueform/slider';

export default {
  name: 'TextNSlider',
  components: { Slider },
  props: {
    options: { type: Object, required: true },
    isAnswered: { type: Boolean, default: false }
  },
  data() {
    return {
      answerValue: null,
      isAnswerValueCorrect: null,
      allowedValues: [] // кэшируем допустимые значения для быстрого поиска
    }
  },
  computed: {
    minValue() {
      return +Object.keys(this.options.answers)[0];
    },
    maxValue() {
      return +Object.keys(this.options.answers).at(-1);
    },
    // Преобразуем { "5000": "Плохо", "23000": "Норм" } → { 5000: {label: "Плохо"}, ... }
    marksConfig() {
      const marks = {};
      Object.entries(this.options.answers).forEach(([key, label]) => {
        // Ключи из Object.entries — строки! Преобразуем в числа:
        marks[+key] = { label, style: { fontSize: '12px' } };
      });
      return marks;
    },
    addClasses() {
      return {
        'correct-answer': this.isAnswerValueCorrect === true,
        'uncorrect-answer': this.isAnswerValueCorrect === false
      }
    },
    currentValueLabel() {
      return this.answerValue 
        ? this.options.answers[this.answerValue] 
        : this.options.answers[this.minValue];
    }
  },
  created() {
    // Кэшируем массив допустимых значений (чисел) при создании компонента
    this.allowedValues = Object.keys(this.options.answers).map(Number);
  },
  watch: {
    isAnswered(newVal) {
      if (newVal) {
        this.checkAnswerValue();
        this.prepareDataForSend();
      } else {
        this.setDefault();
        this.answerValue = this.allowedValues[0];
      }
    }
  },
  methods: {
    // Вызывается при каждом движении ползунка (опционально, если нужна реакция в реальном времени)
    onSliderMove(value) {
      // Можно добавить визуальный фидбек, но не менять answerValue
      // console.log('Движение:', value);
    },
    
    // Вызывается при отпускании ползунка — здесь делаем "примагничивание"
    onSliderRelease() {
      if (!this.answerValue && this.answerValue !== 0) return;
      
      // Находим ближайшее допустимое значение
      const closest = this.findClosestValue(this.answerValue);
      
      // Если значение изменилось — принудительно устанавливаем ближайшее
      if (closest !== this.answerValue) {
        this.answerValue = closest;
      }
      
      // Эмитим события только после "примагничивания"
      this.$emit('valuesIsChosen');
      this.$nextTick(() => this.setActiveLabel());
    },
    
    // Поиск ближайшего значения в массиве
    findClosestValue(current) {
      return this.allowedValues.reduce((prev, curr) => {
        return Math.abs(curr - current) < Math.abs(prev - current) ? curr : prev;
      });
    },
    
    setActiveLabel() {
      const sliderEl = this.$refs.slider?.$el;
      if (!sliderEl) return;
      
      // Снимаем активный класс со всех меток
      const labelEls = sliderEl.querySelectorAll('.slider-mark-label');
      labelEls.forEach((labelEl) => {
        labelEl.classList.remove('active');
      });

      // Добавляем активный класс нужной метке
      if (this.answerValue !== null && this.answerValue !== undefined) {
        // Ищем по текстовому содержимому, так как data-value может не работать
        const targetLabel = this.options.answers[this.answerValue];
        const activeLabel = Array.from(labelEls).find(
          el => el.textContent.trim() === targetLabel
        );
        if (activeLabel) activeLabel.classList.add('active');
      }
    },
    
    checkAnswerValue() {
      this.isAnswerValueCorrect = this.answerValue === this.options.correctAnswer;
    },
    
    setDefault() {
      this.answerValue = null;
      this.isAnswerValueCorrect = null;
      this.$emit('valuesIsUnchosen');
    },
    
    prepareDataForSend() {
      const data = {
        answerValue: this.answerValue,
        isAnswerValueCorrect: this.isAnswerValueCorrect
      };
      this.$emit('readyForDataSending', data);
    }
  },
  mounted() {
    // Устанавливаем первое значение из списка как дефолт
    this.answerValue = this.allowedValues[0];
    this.$nextTick(() => {
      this.setActiveLabel();
      
      // 🔍 Диагностика: проверяем, отрисовались ли маркеры
      const sliderEl = this.$refs.slider?.$el;
      if (sliderEl) {
        console.log('🔍 marksConfig:', this.marksConfig);
        console.log('🔍 allowedValues:', this.allowedValues);
        console.log('🔍 Маркеры в DOM:', sliderEl.querySelectorAll('.slider-mark').length);
        console.log('🔍 Лейблы маркеров:', 
          Array.from(sliderEl.querySelectorAll('.slider-mark-label')).map(el => el.textContent.trim())
        );
      }
    });
  }
}
</script>
