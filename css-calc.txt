handleBlurFix() {
  setTimeout(() => {
    if (!this.maskRef || !this.$refs.incomeInput) return;

    const unmasked = this.maskRef.unmaskedValue; // например: "22.02"
    if (!unmasked || unmasked === '0' || unmasked === '') return;

    // Парсим как число с плавающей точкой
    const num = parseFloat(unmasked);
    if (isNaN(num)) return;

    let finalNum = num;
    if (num > this.stgMax) {
      finalNum = this.stgMax;
    }

    // 🔑 ВАЖНО: передаём ЧИСЛО напрямую в typedValue — imask сам отформатирует его правильно
    this.maskRef.typedValue = finalNum;
    this.maskRef.updateValue?.(); // безопасный вызов, если метод есть

    this.dataFieldIncome = this.maskRef.value;
  }, 10);
}
