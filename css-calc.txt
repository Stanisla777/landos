export default function disappearAdditionalBarScroll() {
  const headerContainer = document.querySelector('.js--another-thing-header');
  const infoBar = document.querySelector('.js--information-bar-top-page');
  const header = headerContainer?.querySelector('header');
  const firstBlock = document.querySelector('.js--another-thing-first-block');

  if (!headerContainer || !infoBar || !header) return;

  let ticking = false;
  let isFixed = false;
  let resizeTimer = null; // Таймер для debounce

  // Создаем заглушку
  const placeholder = document.createElement('div');
  placeholder.style.cssText = 'display:none;width:100%;';
  headerContainer.parentNode.insertBefore(placeholder, headerContainer.nextSibling);

  function updateHeaderPosition() {
    const rect = infoBar.getBoundingClientRect();
    const shouldBeFixed = rect.bottom <= 0;

    if (shouldBeFixed === isFixed) {
      ticking = false;
      return;
    }

    if (shouldBeFixed) {
      const headerHeight = header.offsetHeight;
      
      placeholder.style.display = 'block';
      placeholder.style.height = headerHeight + 'px';
      
      headerContainer.classList.add('header-position-fixed');
      firstBlock?.classList.add('style-first-block');
      
      isFixed = true;
    } else {
      headerContainer.classList.remove('header-position-fixed');
      firstBlock?.classList.remove('style-first-block');
      
      placeholder.style.display = 'none';
      placeholder.style.height = '0';
      
      isFixed = false;
    }

    ticking = false;
  }

  // Обработчик resize с debounce
  function handleResize() {
    // Очищаем предыдущий таймер при каждом новом событии resize
    if (resizeTimer) {
      clearTimeout(resizeTimer);
    }

    // Запускаем новый таймер — функция выполнится только через 100мс
    // после ПОСЛЕДНЕГО события resize
    resizeTimer = setTimeout(() => {
      if (isFixed) {
        placeholder.style.height = header.offsetHeight + 'px';
      }
      resizeTimer = null; // Сбрасываем таймер после выполнения
    }, 100);
  }

  window.addEventListener('scroll', () => {
    if (!ticking) {
      window.requestAnimationFrame(updateHeaderPosition);
      ticking = true;
    }
  }, { passive: true });

  window.addEventListener('resize', handleResize);

  // Первичная проверка
  updateHeaderPosition();
}
