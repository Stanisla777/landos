const headerContainer = document.querySelector('.js--another-thing-header');
    const infoBar = document.querySelector('.js--information-bar-top-page');

    if (!headerContainer || !infoBar) return;

    // Создаем наблюдатель
    const observer = new IntersectionObserver(
        (entries) => {
            entries.forEach((entry) => {
                // entry.boundingClientRect.bottom — нижняя граница плашки
                // entry.isIntersecting — виден ли элемент
                if (!entry.isIntersecting && entry.boundingClientRect.bottom <= 0) {
                    // Плашка ушла за верх
                    headerContainer.classList.add('header-position-fixed');
                } else {
                    // Плашка видна
                    headerContainer.classList.remove('header-position-fixed');
                }
            });
        },
        {
            threshold: 0, // Срабатывает при любом пересечении
            rootMargin: '0px 0px 0px 0px' // Без отступов
        }
    );

    // Начинаем наблюдение за плашкой
    observer.observe(infoBar);
