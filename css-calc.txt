updateModalHeight() {
  setTimeout(() => {
    const modal = this.$refs.selectList;
    const listContainer = this.$refs.wrSearch;
    if (!modal || !listContainer) return;

    modal.style.height = '';
    listContainer.style.paddingBottom = ''; // Сброс

    const visualHeight = (window.visualViewport && window.visualViewport.height) || window.innerHeight;
    const windowHeight = window.innerHeight;

    if (this.isMobile) {
      if (windowHeight - visualHeight > 100) {
        // 🔑 Клавиатура открыта
        modal.style.minHeight = `${windowHeight}px`;
        
        // Добавляем отступ = высота клавиатуры + запас
        const keyboardHeight = windowHeight - visualHeight;
        listContainer.style.paddingBottom = `${keyboardHeight + 20}px`;
      } else {
        // Клавиатуры нет — обычный режим
        modal.style.minHeight = `${visualHeight}px`;
        listContainer.style.paddingBottom = '20px'; // Базовый отступ
      }
    }
  }, 50);
},
