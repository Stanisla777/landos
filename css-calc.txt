.catch((error) => {
  const errorCode = error.response?.data?.error?.code;
  const timeleft = error.response?.data?.result?.timeleft || 0; // Получаем время из ответа
  
  state.btn_load = false;
  handleApiError(state, errorCode, timeleft); // Передаём timeleft
  commit('clearPolling');
});
