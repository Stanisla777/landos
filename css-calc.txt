function moveTooltipIconIntoParagraph() {
  document.querySelectorAll('.row-title-container-with-tooltip').forEach(container => {
    const p = container.querySelector('.calc-tax-deduc-new__row-title.big');
    const desktopIcon = container.querySelector('.content-note.desctop');
    const mobileIcon = container.querySelector('.content-note.mobile');

    if (!p) return;

    // 1. Перемещаем иконки внутрь <p> (как раньше)
    if (desktopIcon && !p.contains(desktopIcon)) {
      p.appendChild(desktopIcon);
    }
    if (mobileIcon && !p.contains(mobileIcon)) {
      p.appendChild(mobileIcon);
    }

    // 2. Оборачиваем ПОСЛЕДНЕЕ СЛОВО + ИКОНКИ в span с nowrap
    //    — но только если ещё не обёрнуто
    if (p.querySelector('.tooltip-last-wrap')) return;

    // Получаем текст ДО иконок
    let textBeforeIcons = '';
    const nodes = Array.from(p.childNodes);
    for (let node of nodes) {
      if (node.nodeType === Node.TEXT_NODE) {
        textBeforeIcons += node.textContent;
      } else if (node.classList.contains('content-note')) {
        break; // дальше идут иконки — останавливаемся
      }
    }

    const words = textBeforeIcons.trim().split(' ');
    if (words.length < 1) return;

    const lastWord = words.pop();
    const restText = words.join(' ');

    // Создаём обёртку
    const wrap = document.createElement('span');
    wrap.className = 'tooltip-last-wrap';
    wrap.style.whiteSpace = 'nowrap';
    wrap.textContent = lastWord + '\u00A0'; // слово + неразрывный пробел

    // Перемещаем иконки ВНУТРЬ обёртки
    const icons = p.querySelectorAll('.content-note');
    icons.forEach(icon => wrap.appendChild(icon));

    // Собираем всё обратно
    p.innerHTML = restText + ' ';
    p.appendChild(wrap);
  });
}
