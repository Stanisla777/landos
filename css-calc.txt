function fullnessFieldToActiveButton() {
  const forms = document.querySelectorAll('.js--fullness-field');

  forms.forEach(form => {
    const requiredCheckboxes = form.querySelectorAll('input[required].js--validate-mask--check');
    const submitButton = form.querySelector('.js--button-status-form');

    // Функция для проверки, все ли чекбоксы отмечены
    const updateButtonState = () => {
      const allChecked = Array.from(requiredCheckboxes).every(cb => cb.checked);
      if (allChecked) {
        submitButton.classList.remove('disabled');
      } else {
        submitButton.classList.add('disabled');
      }
    };

    // Назначаем обработчик каждому чекбоксу в текущей форме
    requiredCheckboxes.forEach(cb => {
      cb.addEventListener('change', updateButtonState);
    });

    // Инициализация состояния кнопки при загрузке
    updateButtonState();
  });
}
