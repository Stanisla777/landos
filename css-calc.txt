callback() {
  const el = this.$el;

  if (el?.hasAttribute('data-formsubmit') && 
      el.getAttribute('data-formsubmit') === 'success') {

    this.$nextTick(() => {
      // === 1. Сначала сбрасываем навигацию ===
      this.step_1 = true;
      
      // === 2. Сбрасываем ВСЁ состояние (явно) ===
      this.show_warn = true;
      this.error_step_1 = true;
      this.error_step_2 = true;
      
      // Счётчики Шага 1
      this.fullness_input = 0;
      this.fullness_input_name = 0;
      this.fullness_input_mail = 0;
      this.fullness_input_tel = 0;
      this.fullness_select = 0;
      this.fullness_input_radio = 0;
      
      // Счётчики Шага 2
      this.fullness_radio = 0;
      this.all_checked = true;

      // === 3. Сброс UI: Селекты ===
      const selects = this.$el.querySelectorAll('.js--izhs-select.js--selected');
      selects.forEach(select => {
        select.classList.remove('js--selected', 'input_error');
        const input = select.querySelector('input.js--select-input');
        const hidden = select.querySelector('input.js--select-input-value');
        if (input) {
          input.value = '';
          input.removeAttribute('data-search');
          input.classList.remove('active');
        }
        if (hidden) hidden.value = '';
        select.querySelectorAll('.js--select-item.active-item')
          .forEach(item => item.classList.remove('active-item'));
      });

      // === 4. Сброс радио-групп Шага 2 ===
      const radioGroups = this.$el.querySelectorAll('.js--izhs-radio-group-step-2.js--selected');
      radioGroups.forEach(group => group.classList.remove('js--selected'));
      
      // === 5. Возвращаем радио "Нет" (Шаг 1) — ВАЖНО: через Vue! ===
      const defaultRadio = this.$el.querySelector('#right_1');
      if (defaultRadio) {
        // 1. Меняем DOM
        defaultRadio.checked = true;
        // 2. ✅ Триггерим событие, чтобы сработал @change и обновилось состояние Vue
        defaultRadio.dispatchEvent(new Event('change', { bubbles: true }));
      }

      // === 6. ПЕРЕсчитываем валидацию ПОСЛЕ обновления DOM ===
      // Используем setTimeout 0, чтобы гарантировать, что браузер применил все изменения
      setTimeout(() => {
        this.updateFullnessInput();
        this.buttonActivationLastStep();
        
        // === 7. Сбрасываем флаг для следующей отправки ===
        el.setAttribute('data-formsubmit', '');
        
        // 🔍 Отладка (можно убрать после проверки)
        console.log('[FormReset] Final state:', {
          step_1: this.step_1,
          error_step_1: this.error_step_1,
          error_step_2: this.error_step_2,
          fullness_input: this.fullness_input,
          fullness_radio: this.fullness_radio,
          show_warn: this.show_warn
        });
      }, 0);
    });
  }
},

changeRightParticipate(event) {
  const element = event?.currentTarget || event?.target;
  if (!element) return;
  
  const val_attr = element.value;

  if (val_attr === 'yes') {
    this.show_warn = false;
    element.closest('.js--izhs-radio-group')?.classList.add('js--selected');
    this.fullness_input_radio = 1;
  } else {
    this.show_warn = true;
    element.closest('.js--izhs-radio-group')?.classList.remove('js--selected');
    this.fullness_input_radio = 0;
  }

  this.fullness_input = this.fullness_input_name + this.fullness_input_tel +
    this.fullness_input_mail + this.fullness_input_radio + this.fullness_select;
},

buttonActivationLastStep() {
  // Ждём, если шаг 2 ещё не отрендерился (v-show)
  const array = this.$el.querySelectorAll('.js--izhs-radio-group-step-2');
  
  // Если элементов нет — значит, шаг 2 скрыт, считаем, что валидация не пройдена
  if (array.length === 0) {
    this.error_step_2 = true;
    return;
  }
  
  this.fullness_radio = 0;
  array.forEach(item => {
    if (item.classList.contains('js--selected')) {
      this.fullness_radio += 1;
    }
  });

  this.error_step_2 = array.length !== this.fullness_radio;
},
