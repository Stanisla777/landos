/**
 * Вспомогательная функция: показывает стрелки для активного слайда
 * и добавляет/удаляет класс .unactive в зависимости от количества слайдов
 */
function updateActiveArrows(parent, mainSwiper, activeIndex) {
  const currentSlide = mainSwiper.slides[activeIndex];
  if (!currentSlide) return;

  const nestedSwiperEl = currentSlide.querySelector('.js--slider-construtor');
  if (!nestedSwiperEl) return;

  const dataArrowsAttr = nestedSwiperEl.getAttribute('data-arrows');
  if (!dataArrowsAttr) return;

  let arrowConfig;
  try {
    arrowConfig = JSON.parse(dataArrowsAttr);
  } catch (e) {
    console.warn('Не удалось распарсить data-arrows:', dataArrowsAttr);
    return;
  }

  const arrowClass = arrowConfig.arrow_left || arrowConfig.arrow_right;
  if (!arrowClass) return;

  const className = arrowClass.replace(/^\./, '');
  const targetArrow = parent.querySelector(`.${className}`);
  if (!targetArrow) return;

  const activeArrows = targetArrow.closest('.js--swiper-arrow-prise-container');
  if (!activeArrows) return;

  // ▼▼▼ Скрываем все контейнеры, показываем активный ▼▼▼
  const allContainers = parent.querySelectorAll('.js--swiper-arrow-prise-container');
  allContainers.forEach(container => {
    container.classList.add('display-none');
    // Важно: убираем unactive со всех, чтобы не осталось "залипших" состояний
    container.classList.remove('unactive');
  });
  activeArrows.classList.remove('display-none');

  // ▼▼▼ Функция проверки с повторными попытками ▼▼▼
  let attempts = 0;
  const maxAttempts = 20; // 20 * 100ms = 2 секунды максимум
  
  const checkAndUpdateArrows = () => {
    attempts++;
    
    // Если вложенный слайдер ещё не инициализирован — пробуем позже
    if (!nestedSwiperEl.swiper) {
      if (attempts < maxAttempts) {
        setTimeout(checkAndUpdateArrows, 100);
      } else {
        console.warn('Не удалось дождаться инициализации вложенного слайдера для слайда', activeIndex);
      }
      return;
    }

    const nestedSwiper = nestedSwiperEl.swiper;

    // Принудительно обновляем слайдер и его размеры
    nestedSwiper.update();
    nestedSwiper.updateSize();
    if (nestedSwiper.params.autoHeight) {
      nestedSwiper.updateAutoHeight();
    }

    // Даём браузеру время на перерисовку перед проверкой классов
    requestAnimationFrame(() => {
      // Проверяем состояние стрелок
      const arrowButtons = activeArrows.querySelectorAll('.swiper-slide-arrow');
      let disabledCount = 0;
      
      arrowButtons.forEach(btn => {
        // Проверяем все возможные классы/состояния disabled
        if (btn.classList.contains('swiper-button-disabled') || 
            btn.classList.contains('disabled-non-disappear') ||
            btn.hasAttribute('disabled')) {
          disabledCount++;
        }
      });

      // Также проверяем, сколько слайдов реально видно
      const slidesPerView = nestedSwiper.params.slidesPerView || 1;
      const totalSlides = nestedSwiper.slides ? nestedSwiper.slides.length : 0;
      
      // Логика .unactive: скрываем стрелки, если некуда крутить
      if (totalSlides <= 1 || totalSlides <= slidesPerView || disabledCount >= 2) {
        activeArrows.classList.add('unactive');
        nestedSwiper.allowTouchMove = false;
        if (nestedSwiper.params.simulateTouch !== undefined) {
          nestedSwiper.params.simulateTouch = false;
        }
      } else {
        activeArrows.classList.remove('unactive');
        nestedSwiper.allowTouchMove = true;
        if (nestedSwiper.params.simulateTouch !== undefined) {
          nestedSwiper.params.simulateTouch = true;
        }
      }
    });
  };

  // Запускаем проверку
  checkAndUpdateArrows();
}
