// Настройки devServer для разработки
config.devServer = {
  port: 8080,
  open: true,
  hot: true,
  liveReload: false,
  
  // Добавляем отслеживание EJS-файлов
  watchFiles: {
    paths: ['src/ejs/**/*.ejs'],
    options: {
      usePolling: true,
      interval: 1000
    }
  },
  
  static: {
    directory: path.join(__dirname, 'dist'),
    watch: true,
    publicPath: '/'
  },
  
  client: {
    overlay: {
      errors: true,
      warnings: false
    },
    progress: true,
    reconnect: true
  },
  
  devMiddleware: {
    publicPath: '/',
    writeToDisk: false
  },
  
  // Добавляем простое уведомление в консоль
  onBeforeSetupMiddleware: (devServer) => {
    devServer.app.use((req, res, next) => {
      res.on('finish', () => {
        if (req.url.includes('.ejs')) {
          console.log('📝 EJS файл изменён — нажмите F5 для обновления');
        }
      });
      next();
    });
  }
};
