//rsbuild.config.js
const { defineConfig } = require('@rsbuild/core');
const { pluginPug } = require('@rsbuild/plugin-pug');
const path = require('path');
const fs = require('fs');

// Находим все pug файлы в views
const pugViewsDir = path.resolve(__dirname, 'src/pug/views');
const pugFiles = fs.existsSync(pugViewsDir)
  ? fs.readdirSync(pugViewsDir).filter(f => f.endsWith('.pug'))
  : [];

// Создаем entry points для JS и SCSS
const entry = {
  // JS entry points
  all: './src/js/index.js',
  'general-js': './src/js/general.js',
  'instruction-js': './src/js/instruction.js',
  
  // SCSS entry points
  'general-style': './src/scss/general-style.scss',
  'instruction': './src/scss/instruction.scss',
};

// Добавляем pug файлы в entry (нужно для генерации HTML)
pugFiles.forEach(file => {
  const name = file.replace('.pug', '');
  entry[name] = path.join(pugViewsDir, file);
});

module.exports = defineConfig({
  // Плагины
  plugins: [
    pluginPug({
      pretty: true, // Красивый HTML для разработки
      basedir: path.resolve(__dirname, 'src/pug'),
    }),
  ],
  
  // Входные точки
  source: {
    entry: entry,
    // Алиасы (как в resolve.alias)
    alias: {
      vue$: 'vue/dist/vue.runtime.esm-bundler.js',
    },
  },
  
  // Настройки вывода
  output: {
    distPath: {
      root: 'dist',
      js: 'js',
      css: 'css',
      font: 'fonts',
      image: 'img',
    },
    filename: {
      js: process.env.NODE_ENV === 'production' 
        ? '[name].[contenthash].bundle.js' 
        : '[name].bundle.js',
      css: '[name].css',
    },
    publicPath: process.env.NODE_ENV === 'production' ? '/dist/' : '/',
    cleanDistPath: true, // Заменяет clean: true
    // Копирование статических файлов (заменяет CopyRspackPlugin)
    copy: [
      { from: './src/fonts', to: 'fonts' },
      { from: './src/img', to: 'img' },
    ],
    // Минификация для production
    minify: process.env.NODE_ENV === 'production',
  },
  
  // Настройки HTML генерации
  html: {
    // Шаблон по умолчанию для всех страниц
    template: ({ entryName }) => {
      // Ищем соответствующий pug файл
      const pugFile = pugFiles.find(f => f.replace('.pug', '') === entryName);
      if (pugFile) {
        return path.join(pugViewsDir, pugFile);
      }
      // Для js entry points используем index.pug
      return path.join(pugViewsDir, 'index.pug');
    },
    // Какие чанки подключать для каждой страницы
    chunks: (entryName) => {
      if (entryName === 'instruction') {
        return ['instruction', 'instruction-js', 'vendor-script', 'common'];
      } else if (entryName === 'index' || entryName === 'all') {
        return ['all', 'general-style', 'general-js', 'vendor-script', 'common'];
      }
      return 'all';
    },
    inject: 'body',
    scriptLoading: 'defer',
  },
  
  // Настройки для инструментов (загрузчики и т.д.)
  tools: {
    // Кастомная настройка цепочки сборщика
    bundlerChain: (chain) => {
      // Правило для SCSS (заменяет module.rules для scss)
      chain.module
        .rule('scss')
        .test(/\.(sass|scss)$/)
        .include.add(path.resolve(__dirname, 'src/scss')).end()
        .use('sass-loader')
        .loader('sass-loader')
        .options({ sourceMap: true });
      
      // Ваш кастомный Vue лоадер
      chain.module
        .rule('vue')
        .test(/\.vue$/)
        .use('vue-compiler-loader')
        .loader(path.resolve(__dirname, 'loaders/vue-compiler-loader.js'));
      
      // Правило для JS с babel
      chain.module
        .rule('js')
        .test(/\.js$/)
        .exclude.add(/node_modules/).end()
        .use('babel-loader')
        .loader('babel-loader')
        .options({
          presets: ['@babel/preset-env'],
        });
      
      // Source maps для js
      chain.module
        .rule('source-map')
        .test(/\.js$/)
        .enforce('pre')
        .use('source-map-loader')
        .loader('source-map-loader');
    },
    
    // Настройка PostCSS
    postcss: (config) => {
      config.postcssOptions.plugins = [
        require('autoprefixer'),
        ...(process.env.NODE_ENV === 'production' 
          ? [require('cssnano')({ 
              preset: ['default', { 
                discardComments: { removeAll: true },
                normalizeWhitespace: true,
              }] 
            })] 
          : [])
      ];
      return config;
    },
  },
  
  // Настройки оптимизации и разделения кода
  performance: {
    chunkSplit: {
      strategy: 'split-by-experience',
      forceSplitting: {
        // Вендоры (заменяет vendor cacheGroup)
        vendor: /[\\/]node_modules[\\/](axios|vue-axios|swiper|imask)[\\/]/,
        // Общие чанки (заменяет common cacheGroup)
        common: {
          minChunks: 2,
          priority: 5,
        },
      },
    },
    // Отключаем предупреждения о размере
    hints: false,
  },
  
  // Настройки разработки
  dev: {
    port: 8080,
    open: true,
    hmr: true, // Hot Module Replacement
    liveReload: true,
    watchFiles: {
      paths: ['src/pug/**/*.pug', 'src/**/*'],
      options: {
        usePolling: true,
        interval: 1000,
      },
    },
    client: {
      overlay: {
        errors: true,
        warnings: false,
      },
      progress: true,
    },
  },
  
  // Серверные настройки (заменяют devServer)
  server: {
    publicDir: {
      name: 'dist',
      watch: true,
    },
  },
  
  // Source maps
  output: {
    // ... (это уже было выше, объединяем)
    sourceMap: {
      js: process.env.NODE_ENV === 'production' ? false : 'source-map',
      css: true,
    },
  },
});
