export function anotherThingPrizes() {
  const swiper = new Swiper('.js--another-thing-2025-prizes', {
    init: true,
    loop: false,
    slidesPerView: 1,
    spaceBetween: 16,
    simulateTouch: false,
    allowTouchMove: false,
    autoHeight: true,
    resizeObserver: true,
    observeParents: true,
    observer: true,
    observeSlideChildren: true,

    pagination: {
      // el: '.js--swiper-pagination-question',
      // clickable: true,
    },
    navigation: {
      // nextEl: '.js--swiper-arrow-right-question',
      // prevEl: '.js--swiper-arrow-left-question',
    },
    breakpoints: {},

    on: {
      beforeInit: () => {
        // ничего не делаем
      },

      afterInit: function (el) {
        // После инициализации сразу скрываем все стрелки, 
        // кроме тех, что относятся к активному слайду
        const parent = el.el.closest('.js--prizes-container');
        if (!parent) return;

        // Скрываем все контейнеры стрелок
        parent.querySelectorAll('.js--swiper-arrow-prise-container').forEach(container => {
          container.classList.add('display-none');
        });

        // Показываем только стрелки для текущего слайда
        const activeIndex = el.activeIndex;
        const activeArrows = parent.querySelector(
          `.js--swiper-arrow-prise-container[data-swiper-slide-index="${activeIndex}"]`
        );
        if (activeArrows) {
          activeArrows.classList.remove('display-none');
          // Обновляем состояние стрелок вложенного слайдера, если он уже инициализирован
          const nestedSwiperEl = el.slides[activeIndex]?.querySelector('.js--slider-construtor');
          if (nestedSwiperEl?.swiper) {
            nestedSwiperEl.swiper.update();
          }
        }
      },

      resize: function (el) {
        // можно добавить el.update(), если нужно
      },

      slideChange: function (el) {
        const parent = el.el.closest('.js--prizes-container');
        if (!parent) return;

        // Скрываем все стрелки
        parent.querySelectorAll('.js--swiper-arrow-prise-container').forEach(container => {
          container.classList.add('display-none');
        });

        // Показываем стрелки для активного слайда
        const activeIndex = el.activeIndex;
        const activeArrows = parent.querySelector(
          `.js--swiper-arrow-prise-container[data-swiper-slide-index="${activeIndex}"]`
        );
        if (activeArrows) {
          activeArrows.classList.remove('display-none');
          
          // Важно: обновляем вложенный слайдер, чтобы он пересчитал disabled-состояние стрелок
          const nestedSwiperEl = el.slides[activeIndex]?.querySelector('.js--slider-construtor');
          if (nestedSwiperEl?.swiper) {
            // Небольшая задержка, чтобы вложенный слайдер успел отрендериться
            setTimeout(() => {
              nestedSwiperEl.swiper.update();
              if (nestedSwiperEl.swiper.params.autoHeight) {
                nestedSwiperEl.swiper.updateAutoHeight();
              }
            }, 50);
          }
        }
      }
    }
  });

  // === Обработчики пагинации (ваша логика, без изменений) ===
  const controls = document.querySelectorAll('.js--anotgher-thing-prizes-pag');
  
  controls.forEach(control => {
    control.addEventListener('click', () => {
      const slideIndex = parseInt(control.dataset.index, 10);
      console.log(slideIndex);
      
      // Переход на слайд
      swiper.slideTo(slideIndex);
      
      // Обновляем активный класс на кнопках пагинации
      controls.forEach(c => c.classList.remove('active'));
      control.classList.add('active');

      // Логика для мобильной пагинации
      if (control.closest('.js--pagination-parent-mobile')) {
        const mobileBtn = control.closest('.js--pagination-parent-mobile').querySelector('.js--pagination-btn-mobile p');
        if (mobileBtn) {
          mobileBtn.textContent = control.querySelector('p')?.textContent || '';
        }
        control.closest('.js--pagination-parent-mobile').classList.remove('active');
        if (typeof RemoveClassBody === 'function') {
          RemoveClassBody();
        }
      }
    });
  });

  // Синхронизация активного класса пагинации после инициализации
  swiper.on('afterInit', () => {
    const realIndex = swiper.realIndex;
    controls.forEach((control, index) => {
      if (index === realIndex) {
        control.classList.add('active');
      } else {
        control.classList.remove('active');
      }
    });
  });
}
