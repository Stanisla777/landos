        { vote: "Y" },

      sending_data_server.checBox = this.massive_checkbox;

      axios({
        method: 'post',
        url: '/local/api/tests/',
        headers: {
          "Content-type": "text/html; charset=UTF-8"
        },
        data: sending_data_server,
      })
        .then((res) => {

                item.closest('.test-window').setAttribute('style', `height:${height_el + 147}px`);
                resolve();

      } else if (percentage_responses > 40 && percentage_responses <= 60) {
        this.smile_3 = true;
      } else if (percentage_responses > 60 && percentage_responses <= 80) {
        this.smile_4 = true;
      } else if (percentage_responses > 80 && percentage_responses < 100) {

      } else if (percentage_responses === 100) {
