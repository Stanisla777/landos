export default function initSwitcherChoice() {
  const parentContainer = document.querySelector('.js--for-swither-container');
  
  if (!parentContainer) return;

  parentContainer.addEventListener('click', (event) => {
    // 1. Находим сам таб, по которому кликнули (или его родителя)
    const clickedTab = event.target.closest('.ds-switcher__item');
    
    // Если клик был не по табу, а где-то рядом в контейнере — ничего не делаем
    if (!clickedTab) return;

    // 2. Получаем значение атрибута tabindex (вернёт строку, например, "0", "1", "2")
    const tabIndex = clickedTab.getAttribute('tabindex');
    
    console.log('Клик по табу с tabindex:', tabIndex);

    // ==========================================
    // 3. ЛОГИКА ПЕРЕКЛЮЧЕНИЯ КОНТЕНТА
    // ==========================================
    
    // Предположим, что ваши блоки с контентом из PHP имеют класс .js-tab-content 
    // и атрибут data-tab, соответствующий tabindex (например: data-tab="0")
    // <div class="js-tab-content" data-tab="0">Контент тем</div>
    // <div class="js-tab-content" data-tab="1">Контент сервисов</div>
    
    const allContentBlocks = parentContainer.querySelectorAll('.js-tab-content');
    
    // Скрываем все блоки контента
    allContentBlocks.forEach(block => {
      block.style.display = 'none'; // или block.classList.remove('is-active')
    });

    // Находим блок, который соответствует нажатому табу
    const activeContentBlock = parentContainer.querySelector(`.js-tab-content[data-tab="${tabIndex}"]`);
    
    // Показываем нужный блок
    if (activeContentBlock) {
      activeContentBlock.style.display = 'block'; // или activeContentBlock.classList.add('is-active')
    }

    // ==========================================
    // 4. (Опционально) Визуальное переключение самих табов
    // ==========================================
    // Если компонент DsSwitcher сам не добавляет класс 'active' при клике, 
    // можно сделать это вручную:
    const allTabs = parentContainer.querySelectorAll('.ds-switcher__item');
    allTabs.forEach(tab => tab.classList.remove('is-active')); // убираем активный класс у всех
    clickedTab.classList.add('is-active'); // добавляем нажатому
  });
}
