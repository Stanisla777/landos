<template>
  <div class="test-n-slider" :class="addClasses">
    <!-- Vue 3: v-model работает как раньше, но внутри используется modelValue -->
    <Slider
      v-model="answerValue"
      ref="slider"
      :marks="marksConfig"
      :min="minValue"
      :max="maxValue"
      :disabled="isAnswered"
      :tooltip="false"
      :height="15"
      :dot-size="32"
      :dot-style="{ height: '32px', width: '32px', background: '#fff', border: '2px solid #000' }"
      :label-style="{ marginTop: '20px', fontSize: '14px' }"
      :process-style="{ background: '#ccc' }"
      @change="onSliderChange"
    />
    <div class="test-n-slider__value">{{ currentValueLabel }}</div>
  </div>
</template>

<script>
import Slider from '@vueform/slider';
import '@vueform/slider/themes/default.css'; // или ваша тема

export default {
  name: 'TextNSlider',
  components: { Slider },
  props: {
    options: { type: Object, required: true },
    isAnswered: { type: Boolean, default: false }
  },
  data() {
    return {
      answerValue: null,
      isAnswerValueCorrect: null,
    }
  },
  computed: {
    minValue() {
      return +Object.keys(this.options.answers)[0];
    },
    maxValue() {
      return +Object.keys(this.options.answers).at(-1);
    },
    // Преобразуем {0: "Плохо", 5: "Норм", 10: "Отлично"} → {0: {label: "Плохо"}, ...}
    marksConfig() {
      const marks = {};
      Object.entries(this.options.answers).forEach(([key, label]) => {
        marks[key] = { label, style: { fontSize: '12px' } };
      });
      return marks;
    },
    addClasses() {
      return {
        'correct-answer': this.isAnswerValueCorrect === true,
        'uncorrect-answer': this.isAnswerValueCorrect === false
      }
    },
    currentValueLabel() {
      return this.answerValue 
        ? this.options.answers[this.answerValue] 
        : this.options.answers[this.minValue];
    }
  },
  watch: {
    isAnswered(newVal) {
      if (newVal) {
        this.checkAnswerValue();
        this.prepareDataForSend();
      } else {
        this.setDefault();
        this.answerValue = Object.keys(this.options.answers)[0];
      }
    }
  },
  methods: {
    onSliderChange() {
      if (this.answerValue) {
        this.$emit('valuesIsChosen');
        // Vue 3: DOM-манипуляции только после $nextTick
        this.$nextTick(() => this.setActiveLabel());
      }
    },
    setActiveLabel() {
      // @vueform/slider рендерит метки с data-value
      const sliderEl = this.$refs.slider?.$el;
      if (!sliderEl) return;
      
      const labelEls = sliderEl.querySelectorAll('[data-value]');
      labelEls.forEach((labelEl) => {
        labelEl.classList.remove('active');
      });

      if (this.answerValue !== null) {
        const activeLabel = sliderEl.querySelector(`[data-value="${this.answerValue}"]`);
        if (activeLabel) activeLabel.classList.add('active');
      }
    },
    checkAnswerValue() {
      this.isAnswerValueCorrect = this.answerValue === this.options.correctAnswer;
    },
    setDefault() {
      this.answerValue = null;
      this.isAnswerValueCorrect = null;
      this.$emit('valuesIsUnchosen');
    },
    prepareDataForSend() {
      const data = {
        answerValue: this.answerValue,
        isAnswerValueCorrect: this.isAnswerValueCorrect
      };
      this.$emit('readyForDataSending', data);
      // $forceUpdate в Vue 3 работает, но редко нужен
    }
  },
  mounted() {
    this.answerValue = Object.keys(this.options.answers)[0];
    this.$nextTick(() => this.setActiveLabel());
  }
}
</script>
