// Устанавливаем начальные значения через ref (только один раз)
  this.$nextTick(() => {
    if (this.inputFioValue && this.$refs.fioInput) {
      this.$refs.fioInput.value = this.inputFioValue;
    }
    if (this.inputEmailValue && this.$refs.emailInput) {
      this.$refs.emailInput.value = this.inputEmailValue;
    }
    if (this.inputPhoneValue && this.$refs.phoneInput) {
      this.$refs.phoneInput.value = this.inputPhoneValue;
    }
  });
