  this.error_input_mail = this.input_mail.length === 0 || !isValidFormat || isTooLong;

  // üëá –ö–õ–Æ–ß–ï–í–ê–Ø –ß–ê–°–¢–¨: –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —Å–∫—Ä—ã–≤–∞–µ–º –æ—à–∏–±–∫–∏ –ø—Ä–∏ –∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏–∏
  if (isTooLong) {
    this.error_mail_too_long = true;
    this.error_mail = false;

    this.error_mail_too_long = false;

  }
}
