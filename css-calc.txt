// modules/scroll-depth-tracker-clever-data.js

const CONFIG = {
  thresholds: [25, 50, 75, 100],
  eventPrefix: 'scroll_',
  throttleMs: 200,
  // Селектор блоков, составляющих "полезный контент"
  contentSelector: '.js--calculated-scroll-area',
  // Отступ от конца последнего блока, при котором считаем, что прочитано 100%
  // 0 = когда конец блока коснулся верха экрана
  // 100 = когда до конца блока осталось 100px
  offsetBottomPx: 0 
};

const state = {
  sent: new Set(),
  maxReached: 0,
  lastCheck: 0,
  geometry: null 
};

/**
 * Рассчитывает границы и высоту полезного контента.
 * Использует getBoundingClientRect для максимальной совместимости с iOS/Safari.
 */
function calculateGeometry() {
  const elements = document.querySelectorAll(CONFIG.contentSelector);
  
  if (!elements || elements.length === 0) {
    return null;
  }

  let minTop = Infinity;
  let maxBottom = -Infinity;
  let totalHeight = 0;

  elements.forEach(el => {
    // rect.top - позиция относительно верха видимой области
    const rect = el.getBoundingClientRect();
    
    // Абсолютная позиция на странице
    const absoluteTop = window.scrollY + rect.top;
    const absoluteBottom = absoluteTop + rect.height;

    if (absoluteTop < minTop) minTop = absoluteTop;
    if (absoluteBottom > maxBottom) maxBottom = absoluteBottom;
    
    // Суммируем высоты (если нужно строго только контент без разрывов)
    // Но для учета разрыва в 40px лучше использовать разницу maxBottom - minTop
    totalHeight += rect.height;
  });

  // Вариант А: Учитываем разрывы между блоками (Рекомендуется для вашего случая с 40px)
  const effectiveContentHeight = maxBottom - minTop;

  // Вариант Б: Только сумма высот (если бы разрывы были огромными и их надо было исключить)
  // const effectiveContentHeight = totalHeight; 

  return {
    start: minTop,                // Начало первого блока
    end: maxBottom,               // Конец последнего блока
    height: effectiveContentHeight // Полная высота зоны (включая мелкие разрывы)
  };
}

function sendToCleverData(eventName) {
  if (typeof window.dmpkitdl !== 'undefined' && window.dmpkitdl !== null) {
    try {
      window.dmpkitdl.push({ event: eventName });
      // console.log('[CleverData] Sent:', eventName); 
    } catch (e) {
      console.warn('[ScrollDepth] Error sending to dmpkitdl:', e);
    }
  }
}

function getScrollPercent() {
  // 1. Получаем геометрию (из кэша или пересчитываем)
  if (!state.geometry) {
    state.geometry = calculateGeometry();
  }
  
  if (!state.geometry || state.geometry.height <= 0) return 0;

  const scrollTop = window.scrollY || document.documentElement.scrollTop;
  
  const { start, height } = state.geometry;

  // 2. Сколько пикселей контента уже прошло над верхней границей экрана?
  // Если scrollTop < start, значит мы еще не дошли до начала статьи (скроллим хедер)
  const scrolledIntoContent = scrollTop - start;

  // 3. Расчет процента
  if (scrolledIntoContent < 0) {
    return 0; // Еще не начали читать
  }

  // Эффективная высота для 100% (с учетом желаемого отступа от конца)
  const targetHeight = height - CONFIG.offsetBottomPx;
  
  if (targetHeight <= 0) {
     return scrolledIntoContent >= 0 ? 100 : 0;
  }

  const percent = (scrolledIntoContent / targetHeight) * 100;

  return Math.min(100, Math.max(0, Math.round(percent)));
}

function checkScrollDepth() {
  const now = Date.now();
  if (now - state.lastCheck < CONFIG.throttleMs) return;
  state.lastCheck = now;

  // Если геометрия еще не посчитана или сброшена (например, после resize)
  if (!state.geometry) {
    state.geometry = calculateGeometry();
  }

  // Дополнительная защита для iOS: если высота 0, возможно, контент еще не отрендерился
  if (!state.geometry || state.geometry.height === 0) return;

  const currentPercent = getScrollPercent();

  for (const threshold of CONFIG.thresholds) {
    if (threshold > state.maxReached && currentPercent >= threshold) {
      const eventName = CONFIG.eventPrefix + threshold;
      if (!state.sent.has(threshold)) {
        sendToCleverData(eventName);
        state.sent.add(threshold);
        state.maxReached = threshold;
      }
      break;
    }
  }
}

export default function scrollDepthTrackerCleverData() {
  // Инициализация с небольшой задержкой, чтобы DOM и стили применились
  // Особенно важно для iOS Safari, где рендеринг может запаздывать
  setTimeout(() => {
    state.geometry = calculateGeometry();
    checkScrollDepth();
  }, 100);

  let ticking = false;

  const onScroll = () => {
    if (!ticking) {
      window.requestAnimationFrame(() => {
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  const onResize = () => {
    // При ресайзе (поворот экрана, изменение панелей браузера) сбрасываем кэш
    state.geometry = null; 
    
    if (!ticking) {
      window.requestAnimationFrame(() => {
        state.geometry = calculateGeometry();
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  window.addEventListener('scroll', onScroll, { passive: true });
  window.addEventListener('resize', onResize, { passive: true });

  return () => {
    window.removeEventListener('scroll', onScroll);
    window.removeEventListener('resize', onResize);
  };
}
