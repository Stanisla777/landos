        <span class="test-n-slider__mark-label">{{ item.label }}</span>

<script>
import Slider from '@vueform/slider';

export default {
  name: 'TextNSlider',
  components: { Slider },
  props: {
    options: { type: Object, required: true },
    isAnswered: { type: Boolean, default: false }
  },
  data() {
    return {
      selectedIndex: 0,
      answerValue: null,
      isAnswerValueCorrect: null
    }
  },
  computed: {
    // Преобразуем объект {10000: "10k", 20000: "20k"} → массив [{value:10000, label:"10k"}, ...]
    answersList() {
      if (!this.options.answers) return [];
      return Object.entries(this.options.answers).map(([val, label]) => ({
        value: +val,
        label
      }));
    },
    addClasses() {
      return {
        'correct-answer': this.isAnswerValueCorrect === true,
        'uncorrect-answer': this.isAnswerValueCorrect === false
      }
    },
    currentValueLabel() {
      const current = this.answersList[this.selectedIndex];
      return current ? current.label : this.answersList[0]?.label;
    }
  },
  watch: {
    answerValue(){
      console.log('this.answerValue');
      console.log(this.answerValue);
    },
    isAnswered(val) {
      if (val) {
        this.checkAnswerValue();
        this.prepareDataForSend();
      } else {
        this.setDefault();
        this.selectedIndex = 0;
      }
    }
  },
  methods: {
    // Равномерное распределение: 0%, 33.3%, 66.6%, 100%
    getEvenPosition(index) {
      if (this.answersList.length <= 1) return 0;
      return (index / (this.answersList.length - 1)) * 100;
    },

    // Срабатывает при отпускании ползунка
    onSliderRelease() {
      if (this.isAnswered) return;

      // Микрозадержка: даём библиотеке завершить внутренние расчеты drag-end
      setTimeout(() => {
        const nearest = Math.round(this.selectedIndex);

        // Обновляем только если реально изменилось
        if (nearest !== this.selectedIndex) {
          this.selectedIndex = nearest;
        }

        this.answerValue = this.answersList[nearest]?.value;
        this.$emit('valuesIsChosen');
      }, 0);
    },

    // Клик по маркеру
    selectByIndex(index) {
      if (this.isAnswered) return;
      this.selectedIndex = index;
      this.onSliderRelease();
    },

    checkAnswerValue() {
      this.isAnswerValueCorrect = this.answerValue === this.options.correctAnswer;
      console.log('this.options.correctAnswer');
      console.log(this.options.correctAnswer);
    },
    setDefault() {
      this.selectedIndex = 0;
      this.answerValue = this.answersList[0]?.value;
      this.isAnswerValueCorrect = null;
      this.$emit('valuesIsUnchosen');
    },
    prepareDataForSend() {
      this.$emit('readyForDataSending', {
        answerValue: this.answerValue,
        isAnswerValueCorrect: this.isAnswerValueCorrect
      });
    }
  },
  mounted() {
    this.selectedIndex = 0;
    this.answerValue = this.answersList[0]?.value;
  }
}
