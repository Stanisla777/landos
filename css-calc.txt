.catch((error) => {
  const errorCode = error.response?.data?.error?.code;
  const timeleft = error.response?.data?.result?.timeleft || 0; // Получаем время из ответа
  
  state.btn_load = false;
  handleApiError(state, errorCode, timeleft); // Передаём timeleft
  commit('clearPolling');
});


else if (errorCode === 'P450S-19') {
  if (timeleft !== null) {
    // С таймером
    state.timeleft_certificate = timeleft;
    state.modal_content = {
      ...state.modal_variations.certificate_could_not_generated_available_through,
      description: `Повторный запрос справки будет доступен через ${timeleft} секунд`
    };
  } else {
    // Без таймера (общая ошибка)
    state.modal_content = state.modal_variations.certificate_could_not_generated;
  }
  state.modal_status = true;
}
