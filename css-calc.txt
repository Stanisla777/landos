function handleShareClick(e) {
  e.preventDefault();
  const trigger = e.currentTarget;
  const wrapper = trigger.closest('.js--soc-list-wrapper-share');
  if (!wrapper) return;

  // Закрываем текущий активный popup (если есть)
  if (activeDropdown) {
    hideDropdown(activeDropdown);
  }

  // ✅ Ищем popup в кэше по wrapper
  let dropdown = dropdownCache.get(wrapper);
  if (!dropdown) {
    // Создаём popup впервые для этой кнопки
    dropdown = createShareDropdown(wrapper);
    // Вставляем в глобальный контейнер
    ensurePopupContainer().appendChild(dropdown);
    // Сохраняем в кэш (WeakMap — безопасно)
    dropdownCache.set(wrapper, dropdown);
  }

  // Показываем popup под текущей кнопкой
  showDropdown(dropdown, trigger);
}
