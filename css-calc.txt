watch: {
    screenWidth(newWidth) {
      if (newWidth <= 480) {
      
        const cookiesClosed = this.getCookieCookies('cookiewindowclosed');
        if (cookiesClosed !== 'yes') {
          this.isHidden = true; // Скрываем, только если куки ещё нет
        }
        // Если кука есть — блок остаётся видимым (isHidden не меняем)
      } else {
        // Десктоп: всегда показываем
        this.isHidden = false;
      }
    }
  }
