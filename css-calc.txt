parseJsonToArray(st) {
  if (typeof st === 'string') {
    try {
      const parsed = JSON.parse(st);
      
      // Если это объект с ключами — извлекаем пары ключ-значение в порядке появления в строке
      if (parsed && typeof parsed === 'object' && !Array.isArray(parsed)) {
        // Регулярка ищет пары "ключ":"значение" в исходной строке
        // Поддерживает экранированные кавычки внутри значений
        const regex = /"([^"]+)"\s*:\s*"((?:[^"\\]|\\.)*)"/g;
        const result = [];
        let match;
        
        while ((match = regex.exec(st)) !== null) {
          const key = match[1];    // ID региона, например "279"
          const value = match[2];  // Название региона
          result.push({ id: key, name: value });
        }
        
        // Если нашли пары через регексп — возвращаем массив объектов
        if (result.length > 0) return result;
        
        // Фоллбэк: если регексп не сработал, возвращаем массив значений (но порядок может быть нарушен)
        return Object.values(parsed).map(name => ({ id: null, name }));
      }
      
      // Если уже массив — возвращаем как есть
      return Array.isArray(parsed) ? parsed : [parsed];
    }
    catch (e) {
      console.warn('Ошибка парсинга JSON в parseJsonToArray:', e);
      return Array.isArray(st) ? st : [st];
    }
  }
  // Если пришла не строка — возвращаем как есть
  return Array.isArray(st) ? st : [st];
}
