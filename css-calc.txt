// Позиционирование popup’а с адаптацией по краям экрана
function positionDropdown(trigger, dropdown) {
  const rect = trigger.getBoundingClientRect();
  const viewportWidth = window.innerWidth;

  // ✅ Определяем: десктоп и нет mobile-share
  const isMobileOrTablet = isMobileDevice();
  const isNarrow = window.matchMedia('(max-width: 480px)').matches;
  const isMobileShare = dropdown.classList.contains('mobile-share');

  if (isMobileOrTablet || isNarrow || isMobileShare) {
    // На мобилке/планшете — не трогаем (уже управляется CSS)
    return;
  }

  // Получаем ширину popup (важно: должен быть visible для getBoundingClientRect)
  dropdown.style.display = 'block';
  dropdown.style.opacity = '0'; // невидим, но в потоке
  const dropdownRect = dropdown.getBoundingClientRect();
  const dropdownWidth = dropdownRect.width;
  dropdown.style.display = 'none';
  dropdown.style.opacity = '';

  // Позиция по умолчанию: под кнопкой, слева от неё
  let top = rect.top + rect.height + 10;
  let left = rect.left;

  // ✅ Проверяем: помещается ли popup справа?
  const rightEdge = left + dropdownWidth;
  if (rightEdge > viewportWidth - 10) {
    // Не помещается → выравниваем ПО ПРАВОМУ КРАЮ кнопки
    left = rect.right - dropdownWidth;
    // Гарантируем, чтобы не ушёл за левый край
    if (left < 0) left = 0;
  }

  // Учитываем прокрутку документа (хотя для fixed — не нужно, но для consistency)
  const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
  const scrollLeft = window.pageXOffset || document.documentElement.scrollLeft;

  dropdown.style.transform = `translate(${left + scrollLeft}px, ${top + scrollTop}px)`;
}
