export default function testBuild() {
  const testElements = document.querySelectorAll('tests');
  console.log('🔍 testBuild: found', testElements.length, '<tests> elements');
  
  testElements.forEach((item) => {
    const id = item.getAttribute('id');
    const views = item.getAttribute('views');
    const vueContainer = document.createElement('div');
    vueContainer.setAttribute('data-vue-app', 'tests-build');
    vueContainer.setAttribute('data-id-test', id);
    vueContainer.setAttribute('data-view-test', views);

    const parent = item.parentNode;
    const favoritesBlock = parent?.querySelector('.block-add-favorites');
    console.log('📦 Before replace: .block-add-favorites exists?', !!favoritesBlock);

    parent.replaceChild(vueContainer, item);
    
    console.log('📦 After replace: .block-add-favorites exists?', !!parent?.querySelector('.block-add-favorites'));
  });
  
  header();
  console.log('✅ testBuild + header() finished');
}
