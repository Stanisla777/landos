// test-build.js (обновлённая версия для Vue 3)
export default function testBuild() {
  const testElements = document.querySelectorAll('tests');
  
  testElements.forEach((item) => {
    const id = item.getAttribute('id');
    const views = item.getAttribute('views');
    
    // Создаём контейнер с правильными атрибутами для менеджера
    const vueContainer = document.createElement('div');
    vueContainer.setAttribute('data-vue-app', 'tests-build');
    vueContainer.setAttribute('data-id-test', id);
    vueContainer.setAttribute('data-view-test', views);
    
    // Заменяем старый элемент новым
    item.parentNode.replaceChild(vueContainer, item);
  });
}
