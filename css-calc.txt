handlerSendForm(argument) {
  this.captchaInit();
  this.send_argument = argument;
  this.btnSendPress = true;

  // 1. Запускаем валидацию полей (покажет красные рамки и "Заполните поле")
  this.validateAllInputs();

  // 2. Если какое-то обязательное поле пустое → останавливаемся
  if (!this.isValid) {
    return; // Запрос НЕ уходит, пользователь видит ошибки на полях
  }

  // 3. Если адрес не найден в Dadata → подсвечиваем поле и показываем ошибку
  if (!this.isAddressFromDadata) {
    this.inputs[0].isNotValid = true; // Красная рамка на поле адреса
    this.ocenkaMobiErrorText = 'Выберите адрес из выпадающего списка';
    this.showError();
    return; // Запрос НЕ уходит
  }

  // Всё валидно
  const grecaptchaInput = document.getElementById('mobi-ocenka-subscribe');
  let recaptchaKey = null;
  if (typeof conf !== 'undefined') {
    recaptchaKey = conf.smartcaptcha_key;
  }

  if ((typeof recaptchaKey !== 'undefined') && recaptchaKey !== null) {
    window.smartCaptcha.execute(this.captcha_id, argument);
  } else {
    // Для локальной разработки без капчи
    this.apiConnect('');
  }
},
