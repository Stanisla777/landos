// Глобальные переменные
let activeTooltip = null;
let activeTrigger = null;
let tooltipContainer = null;
let isBodyLocked = false;
let tooltipCloseTimer = null; // Таймер для закрытия тултипа
let wasNarrow = window.matchMedia('(max-width: 480px)').matches; // ← НОВАЯ ПЕРЕМЕННАЯ

// Обработчик изменения размера окна
function handleResize() {
  const isNarrow = window.matchMedia('(max-width: 480px)').matches;
  
  // Если произошёл переход через границу 480px — закрываем тултип
  // (структура тултипа кардинально меняется: десктопная ↔ мобильная)
  if (isNarrow !== wasNarrow && activeTooltip) {
    hideTooltip(activeTooltip);
  }
  
  wasNarrow = isNarrow;
}

// Инициализация
export default function contentNoteNew() {
  // === Закрываем тултип при скролле (только для десктопа) ===
  const handleScroll = () => {
    const isNarrow = window.matchMedia('(max-width: 480px)').matches;

    // Для мобильной версии НЕ закрываем при скролле
    if (isNarrow && activeTooltip && activeTooltip.classList.contains('mobile-tooltip')) {
      return;
    }

    if (activeTooltip) {
      hideTooltip(activeTooltip);
    }
  };

  // Добавляю обработчики событий
  document.addEventListener('mouseover', handleMouseOver);
  document.addEventListener('mouseout', handleMouseOut);
  document.addEventListener('mouseout', handleTooltipMouseOut);
  document.addEventListener('click', handleClickOutside);
  window.addEventListener('resize', handleResize);

  // === ДОБАВЛЯЕМ ОБРАБОТЧИК ИЗМЕНЕНИЯ МЕДИА-ЗАПРОСА ===
  const mediaQuery = window.matchMedia('(max-width: 480px)');
  // Для поддержки старых браузеров
  if (mediaQuery.addEventListener) {
    mediaQuery.addEventListener('change', handleResize);
  } else if (mediaQuery.addListener) {
    mediaQuery.addListener(handleResize);
  }

  // === НОВЫЙ: Закрываем тултип мгновенно при клике на иконку с модалкой (только десктоп) ===
  document.addEventListener('click', (e) => {
    // Проверяем, что это не мобильное устройство
    if (isMobileDevice()) return;

    // Проверяем, что кликнули на иконку тултипа
    const trigger = e.target.closest('.js--content-note-new');
    if (!trigger) return;

    // Проверяем, есть ли класс .js--modal-opener у иконки или её родителя
    const hasModalOpener = trigger.classList.contains('js--modal-opener') ||
      trigger.closest('.js--modal-opener');

    if (hasModalOpener && activeTooltip) {
      // Мгновенно закрываем тултип без задержки
      hideTooltip(activeTooltip);
    }
  });

  // === Закрытие по кнопке в мобильном тултипе (с обработкой клика на дочерние элементы) ===
  document.addEventListener('click', (e) => {
    // Ищем ближайший элемент с классом кнопки закрытия
    const closeBtn = e.target.closest('.js--content-note-text-new-close');
    if (closeBtn && activeTooltip) {
      hideTooltip(activeTooltip);
    }
  });

  // Скролл страницы (только для десктопа)
  window.addEventListener('scroll', handleScroll, { passive: true });
  // Прокрутка колёсиком мыши (только для десктопа)
  document.addEventListener('wheel', handleScroll, { passive: true });

  // Для мобильных устройств использую клик вместо наведения (тач-устройства)
  const isIOS = (/(iPad|iPhone|iPod)/.test(navigator.userAgent) && !window.MSStream) ||
    (navigator.userAgent.includes('Mac') && 'ontouchend' in document);

  if (isIOS || isMobileDevice()) {
    document.addEventListener('touchstart', handleClick, { passive: false });
    // Свайп на мобильных - НЕ закрываем тултип
    // document.addEventListener('touchmove', handleScroll, { passive: true });
  }

  // Обработчик для закрытия по клику на затемнённый фон контейнера
  document.addEventListener('click', (e) => {
    // Закрытие по клику на затемнённый фон контейнера
    if (e.target === tooltipContainer && tooltipContainer.classList.contains('mobile-tooltip')) {
      hideTooltip(activeTooltip);
    }
  });
}
