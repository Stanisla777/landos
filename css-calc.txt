handleBlurFix() {
  setTimeout(() => {
    if (!this.maskRef) return;

    const unmasked = this.maskRef.unmaskedValue;
    if (!unmasked || unmasked === '0' || unmasked === '') return;

    const num = parseFloat(unmasked);
    if (isNaN(num)) return;

    let finalNum = num;
    if (num > this.stgMax) {
      finalNum = this.stgMax;
    }

    // 🔑 Ключевой момент: устанавливаем unmaskedValue как строку
    this.maskRef.unmaskedValue = String(finalNum);

    // Обновляем ваше data-поле
    this.dataFieldIncome = this.maskRef.value;
  }, 10);
}
