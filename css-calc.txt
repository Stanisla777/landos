const { defineConfig } = require('@rsbuild/core');
const { pluginPug } = require('@rsbuild/plugin-pug');
const path = require('path');
const fs = require('fs');

// Находим все pug файлы в views
const pugViewsDir = path.resolve(__dirname, 'src/pug/views');
const pugFiles = fs.existsSync(pugViewsDir)
  ? fs.readdirSync(pugViewsDir).filter(f => f.endsWith('.pug'))
  : [];

// Создаем entry points для JS и SCSS
const entry = {
  // JS entry points
  all: './src/js/index.js',
  'general-js': './src/js/general.js',
  'instruction-js': './src/js/instruction.js',
  
  // SCSS entry points
  'general-style': './src/scss/general-style.scss',
  'instruction': './src/scss/instruction.scss',
};

module.exports = defineConfig({
  // Плагины
  plugins: [
    pluginPug({
      pretty: true,
      basedir: path.resolve(__dirname, 'src/pug'),
    }),
  ],
  
  // Входные точки
  source: {
    entry: entry,
  },
  
  // Настройки разрешения модулей
  resolve: {
    alias: {
      vue$: 'vue/dist/vue.runtime.esm-bundler.js',
    },
    extensions: ['.js', '.vue', '.json', '.pug'],
  },
  
  // Настройки вывода
  output: {
    distPath: {
      root: 'dist',
      js: 'js',
      css: 'css',
      font: 'fonts',
      image: 'img',
    },
    filename: {
      js: process.env.NODE_ENV === 'production' 
        ? '[name].[contenthash].bundle.js' 
        : '[name].bundle.js',
      css: '[name].css',
    },
    publicPath: process.env.NODE_ENV === 'production' ? '/dist/' : '/',
    cleanDistPath: true,
    copy: [
      { from: './src/fonts', to: 'fonts' },
      { from: './src/img', to: 'img' },
    ],
    sourceMap: {
      js: process.env.NODE_ENV === 'production' ? false : 'source-map',
      css: true,
    },
  },
  
  // Настройки HTML генерации
  html: {
    template: ({ entryName }) => {
      // Для CSS entry points используем соответствующий pug
      if (entryName === 'general-style') {
        return path.join(pugViewsDir, 'index.pug');
      }
      if (entryName === 'instruction') {
        return path.join(pugViewsDir, 'instruction.pug');
      }
      // Для JS entry points
      const pugFile = pugFiles.find(f => f.replace('.pug', '') === entryName);
      if (pugFile) {
        return path.join(pugViewsDir, pugFile);
      }
      // По умолчанию
      return path.join(pugViewsDir, 'index.pug');
    },
    inject: 'body',
    scriptLoading: 'defer',
  },
  
  // Настройки для инструментов
  tools: {
    bundlerChain: (chain) => {
      // Правило для SCSS
      chain.module
        .rule('scss')
        .test(/\.(sass|scss)$/)
        .include.add(path.resolve(__dirname, 'src/scss')).end()
        .use('sass-loader')
        .loader('sass-loader')
        .options({ sourceMap: true });
      
      // Ваш кастомный Vue лоадер
      chain.module
        .rule('vue')
        .test(/\.vue$/)
        .use('vue-compiler-loader')
        .loader(path.resolve(__dirname, 'loaders/vue-compiler-loader.js'));
      
      // Правило для JS с babel
      chain.module
        .rule('js')
        .test(/\.js$/)
        .exclude.add(/node_modules/).end()
        .use('babel-loader')
        .loader('babel-loader')
        .options({
          presets: ['@babel/preset-env'],
        });
      
      // Source maps для js
      chain.module
        .rule('source-map')
        .test(/\.js$/)
        .enforce('pre')
        .use('source-map-loader')
        .loader('source-map-loader');
    },
    
    postcss: (config) => {
      config.postcssOptions.plugins = [
        require('autoprefixer'),
        ...(process.env.NODE_ENV === 'production' 
          ? [require('cssnano')({ 
              preset: ['default', { 
                discardComments: { removeAll: true },
                normalizeWhitespace: true,
              }] 
            })] 
          : [])
      ];
      return config;
    },

    // 👇 ДОБАВЛЕННЫЙ БЛОК для кастомных cacheGroups (vendor и common)
    rspack: (config) => {
      // Добавляем кастомные cacheGroups
      if (!config.optimization) config.optimization = {};
      if (!config.optimization.splitChunks) config.optimization.splitChunks = {};
      if (!config.optimization.splitChunks.cacheGroups) config.optimization.splitChunks.cacheGroups = {};
      
      config.optimization.splitChunks.cacheGroups = {
        vendor: {
          test: /[\\/]node_modules[\\/](axios|vue-axios|swiper|imask)[\\/]/,
          name: 'vendor-script',
          chunks: (chunk) => ['general-js', 'instruction-js'].includes(chunk.name),
          enforce: true,
          priority: 10,
        },
        common: {
          name: 'common',
          chunks: 'all',
          minChunks: 2,
          priority: 5,
          reuseExistingChunk: true,
        },
      };
      
      return config;
    },
  },
  
  // Настройки производительности
  performance: {
    // Отключаем предупреждения о размере
    hints: false,
    // Настройки разделения кода
    chunkSplit: {
      strategy: 'split-by-module',
      // Можно также использовать 'all-in-one', 'single-vendor', 'split-by-experience'
    },
  },
  
  // Настройки разработки
  dev: {
    port: 8080,
    open: true,
    hmr: true,
    liveReload: true,
    watchFiles: {
      paths: ['src/pug/**/*.pug', 'src/**/*'],
      options: {
        usePolling: true,
        interval: 1000,
      },
    },
    client: {
      overlay: {
        errors: true,
        warnings: false,
      },
      progress: true,
    },
  },
  
  // Серверные настройки
  server: {
    publicDir: {
      name: 'dist',
      watch: true,
    },
  },
});
