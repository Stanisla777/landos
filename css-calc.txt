if (isProduction) {
  config.mode = 'production';
  config.devtool = false;
  config.optimization.minimize = true;
  
  // 👇 КЛЮЧЕВОЕ: убираем хеши, оставляем те же имена, что и в development
  config.output.filename = 'js/[name].bundle.js';  // Без [contenthash]
  config.output.publicPath = '/dist/';
  
  // 👇 Отключаем генерацию HTML в production (если не нужна)
  config.plugins = config.plugins.filter(plugin => 
    !(plugin instanceof rspack.HtmlRspackPlugin)
  );

}
