tools: {
  cssLoader: {
    url: false,
  },

  postcss: (config) => {
    config.postcssOptions.plugins = [
      require('autoprefixer'),
      ...(process.env.NODE_ENV === 'production'
        ? [require('cssnano')({
          preset: ['default', {
            discardComments: { removeAll: true },
            normalizeWhitespace: true,
          }]
        })]
        : [])
    ];
    return config;
  },

  rspack: (config) => {
    // Добавляем поддержку require.ensure для ленивой загрузки
    if (!config.module) config.module = {};
    if (!config.module.parser) config.module.parser = {};
    config.module.parser.javascript = {
      ...config.module.parser.javascript,
      requireEnsure: true,
    };

    // 👇 РАДИКАЛЬНОЕ ОТКЛЮЧЕНИЕ ГЕНЕРАЦИИ HTML ДЛЯ ENTRY ТОЧЕК
    // 1. Удаляем все entry, которые не должны генерировать HTML
    if (config.entry) {
      // Создаем новый объект entry, оставляя только те, что реально нужны как входные точки для JS/CSS
      const newEntry = {};
      Object.keys(config.entry).forEach(entryName => {
        // Оставляем все entry, но помечаем, что они не должны создавать HTML
        newEntry[entryName] = {
          import: config.entry[entryName].import || config.entry[entryName],
          // Явно запрещаем создавать HTML для этой entry
          html: false,
        };
      });
      config.entry = newEntry;
    }

    // 2. Глобально запрещаем генерацию HTML через output
    if (!config.output) config.output = {};
    if (!config.output.filename) config.output.filename = {};
    config.output.filename.html = false;

    // 3. Удаляем автоматически создаваемые HTML-плагины, если они есть
    if (config.plugins) {
      config.plugins = config.plugins.filter(plugin => {
        // Фильтруем плагины, оставляем только те, что не являются встроенными HTML-генераторами
        // Это примерный подход, может потребоваться уточнение
        return !plugin.constructor || plugin.constructor.name !== 'HtmlRspackPlugin';
      });
    }

    // 4. Добавляем ТОЛЬКО наши кастомные HtmlRspackPlugin
    const { HtmlRspackPlugin } = require('@rspack/core');
    
    const htmlPlugins = pugFiles.map(file => {
      const name = file.replace('.pug', '');
      return new HtmlRspackPlugin({
        template: path.join(pugViewsDir, file),
        filename: `${name}.html`,
        inject: true,
        scriptLoading: 'defer',
        minify: false,
        // Не указываем chunks - подключаются все скрипты
      });
    });

    // Добавляем наши плагины
    config.plugins = [
      ...(config.plugins || []),
      ...htmlPlugins,
    ];

    return config;
  },
},
