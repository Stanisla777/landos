createSectionBox(content) {
  const padding = 16;
  const blockHeight = 180;
  const borderWidth = 1;
  const borderColor = '#c8c8cb';
  const borderRadius = 8;

  // Копируем исходный объект и добавляем margin
  const contentWithPadding = {
    ...content,
    margin: [padding, padding, padding, padding]
  };

  return {
    stack: [
      {
        canvas: [
          {
            type: 'rect',
            x: 0,
            y: 0,
            w: 515,
            h: blockHeight,
            r: borderRadius,
            lineWidth: borderWidth,
            color: borderColor
          }
        ],
        margin: [0, 0, 0, -blockHeight]
      },
      contentWithPadding
    ],
    margin: [0, 0, 0, 16]
  };
}
