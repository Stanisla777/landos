function initCommentsComplainDelegation() {
  // Защита от повторной инициализации
  if (window.commentsComplainDelegationInitialized) return;
  window.commentsComplainDelegationInitialized = true;

  document.addEventListener('click', (e) => {
    const complainBtn = e.target.closest('.comments__item-button-complain');
    if (!complainBtn) return;
    
    e.preventDefault();
    
    if (complainBtn.dataset.id) {
      const complainInput = document.querySelector('.comments__complain-comment');
      const complainOverlay = document.querySelector('.comments__complain-overlay');
      
      if (complainInput && complainOverlay) {
        complainInput.value = complainBtn.dataset.id;
        complainOverlay.classList.add('comments__complain-overlay_active');
        
        if (!document.body.classList.contains('modal-opened')) {
          document.body.classList.add('modal-opened', 'body-modal-not-remove-else', 'body-modal-not-remove');
          AddClassBody();
        }
      }
    }
  });
}

function commentsComplain() {
  // ❌ УДАЛИТЬ блок с кнопками "Пожаловаться" — теперь это делегирование
  
  // ✅ ОСТАВИТЬ обработчики закрытия попапа (они статичны)
  const buttonClassClose = ['.comments__complain-form-button_exit', '.comments__complain-popup-close'];
  buttonClassClose.forEach((el) => {
    const element = document.querySelector(el);
    if (element) {
      // Защита от повторной привязки
      if (element.dataset.closeHandlerAttached) return;
      element.dataset.closeHandlerAttached = 'true';
      
      element.addEventListener('click', (e) => {
        e.preventDefault();
        document.querySelector('.comments__complain-comment').value = '';
        document.querySelector('.comments__complain-overlay').classList.remove('comments__complain-overlay_active');
        document.body.classList.remove('modal-opened');
        if (!document.body.classList.contains('non-scroll-for-body') && document.body.classList.contains('body-additional-class')) {
          RemoveClassBody();
        }
        document.body.classList.remove('body-modal-not-remove');
        document.body.classList.remove('body-modal-not-remove-else');
      });
    }
  });
}
