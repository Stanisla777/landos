methods: {
  // ===== ВСПОМОГАТЕЛЬНЫЙ: Закрытие мобильного тултипа =====
  closeMobileTooltip(element) {
    const target = element || document.querySelector('.js--tooltip-parent.tooltip-active');
    if (!target) return;
    
    const parent = target.closest?.('.js--tooltip-parent');
    if (parent) {
      parent.classList.remove('tooltip-active');
      if (typeof bodyUnlockMobileFilter === 'function') {
        bodyUnlockMobileFilter(parent);
      }
    }
  },

  // ===== 🔗 Обработчик для соцсетей (ВК, ТГ, МАКС...) =====
  handleSocialClick(event) {
    // На мобильном: закрываем тултип с небольшой задержкой, чтобы ссылка успела открыться
    setTimeout(() => {
      this.closeMobileTooltip(event?.currentTarget);
    }, 100);
    // Ссылка открывается автоматически благодаря href + target="_blank"
  },

  // ===== 📋 Обработчик для "Скопировать ссылку" =====
  async handleCopyClick(event) {
    const element = event?.currentTarget;
    try {
      if (this.answerLink?.link) {
        await this.copyLinkManually(this.answerLink.link, element, () => {
          // 🔥 Закрываем тултип через 1 сек, когда пользователь увидел "Скопировано"
          setTimeout(() => {
            this.closeMobileTooltip(element);
          }, 1000);
        });
      }
    } catch (error) {
      console.error('Ошибка при копировании:', error);
    }
  },

  // ===== ✉️ Обработчик для Email (с проверкой класса) =====
  handleEmailClick(event) {
    // Проверяем, что клик был именно по кнопке из блока "поделиться"
    const isFromShare = event?.target?.closest?.('.js--sand-to-mail');
    
    if (isFromShare) {
      // Закрываем мобильный тултип перед открытием модального окна
      this.closeMobileTooltip(event?.currentTarget);
    }
    
    // Вызываем оригинальный openModal (который используется и в других местах)
    this.openModal(event);
  },

  // ===== 📋 Копирование с поддержкой колбэка =====
  async copyLinkManually(link, element, onAfterAnimation = null) {
    try {
      if (navigator.clipboard) {
        await navigator.clipboard.writeText(link);
        this._updateCopyText(element, 'Скопировано');
        
        // Возврат текста через 3 секунды
        setTimeout(() => {
          this._updateCopyText(element, 'Скопировать ссылку');
        }, 3000);
        
        // Вызов колбэка (для закрытия тултипа)
        if (typeof onAfterAnimation === 'function') {
          onAfterAnimation();
        }
      } else {
        this._fallbackCopy(link, element, onAfterAnimation);
      }
    } catch {
      this._fallbackCopy(link, element, onAfterAnimation);
    }
  },

  _fallbackCopy(link, element, onAfterAnimation = null) {
    const textarea = document.createElement('textarea');
    textarea.value = link;
    textarea.style.position = 'fixed';
    textarea.style.opacity = '0';
    document.body.appendChild(textarea);
    textarea.focus();
    textarea.select();
    document.execCommand('copy');
    document.body.removeChild(textarea);
    
    this._updateCopyText(element, 'Скопировано');
    
    setTimeout(() => {
      this._updateCopyText(element, 'Скопировать ссылку');
    }, 3000);
    
    if (typeof onAfterAnimation === 'function') {
      onAfterAnimation();
    }
  },

  _updateCopyText(element, text) {
    const p = element?.querySelector?.('p');
    if (p) {
      p.textContent = text;
    }
  },

  // ===== 📧 Ваш существующий openModal (не меняем, он используется в других местах) =====
  // openModal(event) { ... ваш код ... }
}
