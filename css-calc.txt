function moveTooltipIconIntoParagraph() {
  document.querySelectorAll('.row-title-container-with-tooltip').forEach(container => {
    const p = container.querySelector('.calc-tax-deduc-new__row-title.big');
    const desktopIcon = container.querySelector('.content-note.desctop');
    const mobileIcon = container.querySelector('.content-note.mobile');

    if (!p || (!desktopIcon && !mobileIcon)) return;

    // Перемещаем иконки внутрь <p>
    if (desktopIcon) p.appendChild(desktopIcon);
    if (mobileIcon) p.appendChild(mobileIcon);

    // Оборачиваем последнее слово + иконки в span с nowrap
    const words = p.textContent.trim().split(' ');
    if (words.length < 2) return;

    const lastWord = words.pop();
    const rest = words.join(' ');

    // Создаём обёртку
    const wrapper = document.createElement('span');
    wrapper.className = 'tooltip-last';
    wrapper.style.whiteSpace = 'nowrap';
    wrapper.innerHTML = `${lastWord}&nbsp;`;

    // Перемещаем иконки ВНУТРЬ обёртки
    if (desktopIcon) wrapper.appendChild(desktopIcon);
    if (mobileIcon) wrapper.appendChild(mobileIcon);

    // Собираем всё обратно
    p.innerHTML = rest + ' ';
    p.appendChild(wrapper);
  });
}
