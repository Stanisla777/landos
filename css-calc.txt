initRealtySlider() {
  this.realtySlider = noUiSlider.create(this.$refs.mortgagePrice, {
    start: [Math.round(this.start)], // Округляем начальное значение
    connect: 'lower',
    step: 1, // Только целые шаги
    range: {
      min: Math.ceil(this.stgMin), // Минимальное целое
      max: Math.floor(this.stgMax) // Максимальное целое
    },
  })

  this.realtySlider.on('start', () => {
    this.input_salary = false
    // При начале движения сохраняем дробную часть
    const currentValue = parseFloat(this.mask_interest.unmaskedValue);
    this.decimalPart = currentValue % 1; // Сохраняем дробную часть
  });

  this.realtySlider.on('slide', (values, handle) => {
    if (this.isUpdating) return
    this.isUpdating = true

    const integerValue = Math.round(values[handle]); // Округляем до целого
    const newValue = integerValue + (this.decimalPart || 0); // Добавляем сохраненную дробную часть
    
    this.updateInputValue(newValue)

    this.isUpdating = false
  });

  this.realtySlider.on('set', (values, handle) => {
    if (this.isUpdating) return
    this.isUpdating = true

    const integerValue = Math.round(values[handle]);
    const newValue = integerValue + (this.decimalPart || 0);
    
    this.updateInputValue(newValue)
    this.updateStoreValue(newValue)

    this.isUpdating = false
    this.decimalPart = null; // Сбрасываем сохраненную дробную часть
  });
},

data() {
  return {
    // ...
    decimalPart: null // Для хранения дробной части при движении ползунка
  }
}
