lastSlide() {
  // 🔑 Самый надёжный способ: isEnd = true, если это последний слайд
  if (this.mySwiper?.isEnd) {
    this.btn_result = true;
  } else {
    this.btn_result = false;
  }
  
  // Альтернатива (если нужно больше контроля):
  // const currentIndex = this.mySwiper?.activeIndex ?? 0;
  // const totalSlides = this.mySwiper?.slides?.length ?? 0;
  // this.btn_result = (currentIndex + 1 === totalSlides);
},

slideChange: () => {
        // 🔑 Ключевое: проверяем, последний ли слайд, после каждого перехода
        this.lastSlide();
      },
