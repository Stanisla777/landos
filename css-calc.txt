// Удаляю форму
function removeReplyForm(form) {
  if (!form) return;
  
  // Если есть поддержка анимаций и переходов
  if (form.style.transition) {
    form.style.transition = 'opacity 0.2s ease';
    form.style.opacity = '0';
    setTimeout(() => {
      if (form.parentNode) {
        form.remove();
      }
    }, 200);
  } else {
    // Мгновенное удаление, если анимация не нужна
    form.remove();
  }
}

function sendReply(item, formData, errors) {
  let request = BX.ajax.runComponentAction('local:comments.form', 'save', {
    mode: 'class',
    data: formData
  });

  request.then(function(responce){
    // 1. Показываем сообщение об успехе
    commentsPopupShow(responce.data.title, responce.data.message, responce.status);
    
    // 2. Сбрасываем форму (на всякий случай, перед удалением)
    item.reset();
    
    // 3. Сбрасываем состояние кнопки
    const button = item.querySelector('.form-comments__button');
    if (button) {
      button.classList.add('form-comments__button_disabled');
      button.classList.remove('submitting');
      button.setAttribute('disabled', 'disabled');
    }
    
    // 4. 🔥 УДАЛЯЕМ форму после успешной отправки
    removeReplyForm(item);
    
    // 5. Сбрасываем локальные ошибки (хотя форма уже удалена)
    errors = {
      name: false,
      text: false,
      check: false
    };
    
    // 6. Обновляем счётчики и список комментариев
    updateCountComments(responce);
    updateCommentsList(responce.data.id, formData.get('parent_id'));
    
  }, function (responce) {
    // ❗ При ошибке форму НЕ удаляем — пользователь должен исправить и отправить снова
    commentsPopupShow('Ошибка', responce.errors, responce.status);
    
    // Разблокируем кнопку, если она была в состоянии submitting
    const button = item.querySelector('.form-comments__button');
    if (button) {
      button.classList.remove('submitting');
      // Состояние disabled/классов уже управляется через валидацию
    }
  });
}
