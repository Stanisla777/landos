script.
      // Загружаем футер
      fetch('/src/includes/footer.html')
        .then(response => response.text())
        .then(data => {
          document.getElementById('footer-placeholder').innerHTML = data;
        })
        .catch(error => {
          console.error('Ошибка загрузки футера:', error);
          document.getElementById('footer-placeholder').innerHTML = '<footer>Футер временно недоступен</footer>';
        });
