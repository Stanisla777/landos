watch: {
  stgMax(newVal) {
    if (this.realtySlider) {
      this.realtySlider.updateOptions({
        range: {
          min: this.stgMin,
          max: newVal
        }
      });
      
      // Также обновите текущее значение, если оно превышает новый максимум
      const currentValue = this.realtySlider.get();
      if (parseInt(currentValue) > newVal) {
        this.realtySlider.set(newVal);
        Storage.dispatch('ActionInitialPayment', newVal);
      }
    }
  },
  // остальные watchers...
}
