<template>
  <div v-cloak>
    <v-component-it-company />
    <v-component-success v-if="status_success" />
    <v-component-failure-not-found
      v-if="status_failure"
      :phone="phone"
      :phone_formatted="phone_formatted"
    />
    <v-component-failure-other-error
      v-if="status_failure_other_error"
      :phone="phone"
      :phone_formatted="phone_formatted"
    />
  </div>
</template>

<script>
import vComponentItCompany from './components/v-component-it-company.vue';
import vComponentSuccess from './components/v-component-success.vue';
import vComponentFailureNotFound from './components/v-component-failure-not-found.vue';
import vComponentFailureOtherError from './components/v-component-failure-other-error.vue';
import store from './store';

export default {
  name: 'ITMortgageSurveys',
  components: {
    vComponentItCompany,
    vComponentSuccess,
    vComponentFailureNotFound,
    vComponentFailureOtherError,
  },
  props: {
    titleHint: {
      type: String,
      default: '',
    },
  },
  data() {
    return {
      phone: '',
      phone_formatted: '',
    };
  },
  computed: {
    status_failure() {
      return store.getters.STATUS_FAILURE;
    },
    status_success() {
      return store.getters.STATUS_SUCCESS;
    },
    status_failure_other_error() {
      return store.getters.STATUS_FAILURE_OTHER_ERROR;
    },
  },
  mounted() {
    if (typeof window.conf !== 'undefined' && window.conf !== null) {
      if (window.conf.hasOwnProperty('UF_PHONE')) {
        this.phone = window.conf.UF_PHONE;
      }
      if (window.conf.hasOwnProperty('UF_PHONE_formatted')) {
        this.phone_formatted = window.conf.UF_PHONE_formatted;
      }
    }
  },
};
</script>

<style scoped>
[v-cloak] {
  display: none !important;
}
</style>
