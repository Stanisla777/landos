// === НОВЫЙ: Закрытие по кнопке в мобильном тултипе ===
document.addEventListener('click', (e) => {
  // Ищем ближайший элемент с классом кнопки закрытия
  const closeBtn = e.target.closest('.js--content-note-text-new-close');
  if (closeBtn && activeTooltip) {
    hideTooltip(activeTooltip);
  }
});
