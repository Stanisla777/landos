<script>
export default {
  name: 'ShareButtons',
  
  props: {
    links: {
      type: Object,
      required: true,
      default: () => ({})
    },
    copyLabel: {
      type: String,
      default: 'Скопировать ссылку'
    },
    maxLabel: {
      type: String,
      default: 'MAX'  // ← Единое место для названия соцсети
    }
  },

  emits: ['share', 'copy', 'email'],

  methods: {
    async copyText(text) {
      try {
        if (navigator.clipboard) {
          await navigator.clipboard.writeText(text)
        } else {
          const textarea = document.createElement('textarea')
          textarea.value = text
          textarea.style.position = 'fixed'
          textarea.style.opacity = '0'
          document.body.appendChild(textarea)
          textarea.focus()
          textarea.select()
          document.execCommand('copy')
          document.body.removeChild(textarea)
        }
        this.$emit('copy', text)
      } catch (e) {
        console.error('Copy failed:', e)
      }
    },

    onCopy() {
      if (this.links.link) {
        this.copyText(this.links.link)
      }
    },

    onShare(network, url) {
      this.$emit('share', { network, url })
      window.open(url, '_blank')
    }
  }
}
</script>
