const headerContainer = document.querySelector('.js--another-thing-header');
    const infoBar = document.querySelector('.js--information-bar-top-page');

    if (!headerContainer || !infoBar) return;

    // Небольшой порог, чтобы класс добавлялся чуть раньше, 
    // до того как плашка полностью исчезнет
    const THRESHOLD = 2;

    function updateHeaderPosition() {
        const rect = infoBar.getBoundingClientRect();
        
        // Добавляем класс чуть раньше (за 2px до полного исчезновения)
        if (rect.bottom <= THRESHOLD) {
            if (!headerContainer.classList.contains('header-position-fixed')) {
                headerContainer.classList.add('header-position-fixed');
            }
        } else {
            if (headerContainer.classList.contains('header-position-fixed')) {
                headerContainer.classList.remove('header-position-fixed');
            }
        }
    }

    // Убираем requestAnimationFrame — проверяем на каждом событии скролла
    // Это дает мгновенную реакцию без задержек
    window.addEventListener('scroll', updateHeaderPosition, { passive: true });
    
    // Первичная проверка
    updateHeaderPosition();
