input(
        type="text"
        inputmode="numeric"
        placeholder="Введите сумму"
        v-model="dataFieldIncome"
        ref="realtyInput"
        @input="numberFormattingThousandths(stgMax, $event)"
      
      )

numberFormattingThousandths(count,e) {
      const element = e.currentTarget;
      const target = e.target
      let position = target.selectionStart;
      const val = parseFloat(this.dataFieldIncome.replace(/\s/g, ''))
      this.dataFieldIncome = new Intl.NumberFormat("ru-RU").format(val);

      if (e.inputType==="deleteContentBackward"){
        target.selectionEnd = position;
      }
      if (this.dataFieldIncome=='не число') {
        this.dataFieldIncome=''
      }
      if (this.dataFieldIncome.length > 0 && this.dataFieldIncome[0] === '0'){
        this.dataFieldIncome = '1' +  this.dataFieldIncome.slice(1)
      }

      if (this.dataFieldIncome.replace(/\s/g, '') > count){
        this.dataFieldIncome=(count).toFixed(0)
          .toString().replace(/(\d)(?=(\d{3})+$)/g, '$1 ')
      }
    },
