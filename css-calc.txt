copyLink(event){
      const element = event.currentTarget;
      let perem = ''
      element.addEventListener('click', async () => {
        if (element.classList.contains('js--instead-copyable')) {
          try {
            perem = element.textContent;
            await navigator.clipboard.writeText(element.textContent);
            // eslint-disable-next-line no-param-reassign
            element.textContent = 'Текст скопирован!';
            setTimeout(()=>{
              element.textContent = perem;
              perem=''
            },3000)
          }
          



          catch (err) {
            if (element.closest('.js--wrapper-inserting-copied')) {
              element.closest('.js--wrapper-inserting-copied')
                .classList
                .add('active');
            }
          }
        }

      });

    }
