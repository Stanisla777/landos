<template>
  <div class="form-rating">
    <!-- Динамически скомпилированный шаблон -->
    <component :is="compiledTemplate" v-bind="templateData" />
  </div>
</template>

<script>
import { compile, h } from 'vue';
import IMask from 'imask';
import Cookies from 'js-cookie';

export default {
  name: 'FormRatingWrapper',
  props: {
    instanceIndex: {
      type: Number,
      default: 0
    },
    // Принимаем HTML-шаблон от бэкенда
    template: {
      type: String,
      default: ''
    }
  },
  data() {
    return {
      variant_placeholder: [
        'Что разочаровало?',
        'Какие изменения вы бы предложили?',
        'Чем стоит дополнить?',
        'Что показалось сложным?',
        'Поделитесь своими эмоциями!'
      ],
      place_holder: '',
      array_star: [],
      rate: 0,
      textValue: '',
      show_error: false,
      length_error: false,
      max_length: 300,
      btn_active: false,
      form_show: true
    };
  },
  computed: {
    cookie_key() {
      return `was_useful_rating_${window.location.pathname}_${this.instanceIndex}`;
    },
    // Данные для шаблона
    templateData() {
      return {
        variant_placeholder: this.variant_placeholder,
        place_holder: this.place_holder,
        array_star: this.array_star,
        rate: this.rate,
        textValue: this.textValue,
        show_error: this.show_error,
        length_error: this.length_error,
        max_length: this.max_length,
        btn_active: this.btn_active,
        form_show: this.form_show
      };
    },
    // Компилируем шаблон
    compiledTemplate() {
      if (!this.template) return null;
      
      try {
        // Создаём рендер-функцию из строки шаблона
        const render = compile(this.template, {
          mode: 'function'
        });
        
        return {
          render: (ctx) => render(ctx),
          // Передаём методы в шаблон
          methods: {
            changeGrade: this.changeGrade,
            showError: this.showError,
            endFocus: this.endFocus,
            beginFocus: this.beginFocus,
            textKeyDown: this.textKeyDown,
            textInput: this.textInput,
            localStorageClick: this.localStorageClick,
            submitForm: this.submitForm
          }
        };
      } catch (error) {
        console.error('Ошибка компиляции шаблона:', error);
        return null;
      }
    }
  },
  methods: {
    initArrayStar() {
      this.array_star = [
        `form-rating-input-${this.instanceIndex}-1`,
        `form-rating-input-${this.instanceIndex}-2`,
        `form-rating-input-${this.instanceIndex}-3`,
        `form-rating-input-${this.instanceIndex}-4`,
        `form-rating-input-${this.instanceIndex}-5`
      ];
    },

    changeGrade(el, index) {
      this.rate = this.array_star.length - index;
      this.place_holder = this.variant_placeholder[this.rate - 1] || '';
    },

    showError() {
      this.show_error = true;
    },

    endFocus() {
      this.show_error = false;
      this.length_error = false;
    },

    beginFocus() {
      this.show_error = false;
      this.length_error = false;
    },

    textKeyDown(el) {
      this.show_error = false;
      const element = el.currentTarget;
      if (element.value.length === this.max_length) {
        this.length_error = true;
      }
    },

    inputIMask() {
      const textArea = this.$el.querySelector('textarea');
      if (!textArea) return false;
      IMask(textArea, {
        mask: /\S\s?$/
      });
      return true;
    },

    textInput(el) {
      const element = el.currentTarget;
      this.btn_active = element.value.trim().length > 0;
      if (element.value.length <= this.max_length) {
        this.length_error = false;
      }
    },

    localStorageLoading() {
      let local_stage = Cookies.get(this.cookie_key);
      if ((local_stage !== undefined) && (parseInt(local_stage) !== 0)) {
        this.rate = parseInt(local_stage);
        this.form_show = false;
      }
    },

    localStorageClick() {
      Cookies.set(this.cookie_key, this.rate);
    },

    submitForm(el) {
      el.preventDefault();
      el.stopPropagation();
      
      const form = el.currentTarget.closest('form');
      if (!form) return;
      
      let formData = new URLSearchParams(Array.from(new FormData(form))).toString();
      
      if (window.BX) {
        BX.ajax({
          url: '/local/ajax/form.result.new.php',
          data: formData,
          method: 'POST',
          dataType: 'json',
          timeout: 30,
          async: true,
          processData: true,
          scriptsRunFirst: true,
          emulateOnload: true,
          start: true,
          cache: false,
          onsuccess: (result) => {
            if (result && result.result) {
              if (typeof window.dmpkitdl !== "undefined") {
                window.dmpkitdl.push({event: 'evaluation-fos-send'});
              }
              this.localStorageClick();
              this.form_show = false;
            }
          },
          onfailure: console.warn
        });
      }
    }
  },
  mounted() {
    this.initArrayStar();
    this.inputIMask();
    this.localStorageLoading();
  }
};
</script>

import '../scss/style.scss';
import { createApp } from 'vue';

// Импорты Vue-компонентов
import TelegramBlock from './modules/TelegramBlock.vue';
import FormRatingWrapper from './modules/FormRatingWrapper.vue';

// ============================================
// ХРАНИЛИЩЕ ИНИЦИАЛИЗИРОВАННЫХ КОМПОНЕНТОВ
// ============================================
const initializedApps = new WeakSet();

// ============================================
// ФУНКЦИЯ ДЛЯ БЕЗОПАСНОГО МОНТИРОВАНИЯ
// ============================================
function mountVueApp(component, element, props = {}) {
  // Защита от двойной инициализации
  if (element.__vue_app__ || initializedApps.has(element)) {
    console.log('⚠️ Компонент уже инициализирован, пропускаем');
    return;
  }

  try {
    // Создаём и монтируем приложение
    const app = createApp(component, props);
    app.mount(element);
    
    // Помечаем как инициализированный
    initializedApps.add(element);
    element.__vue_app__ = app;
    
    console.log('✅ Vue компонент смонтирован:', element);
    return app;
    
  } catch (error) {
    console.error('❌ Ошибка монтирования Vue компонента:', error);
  }
}

// ============================================
// СПИСОК ВСЕХ VUE-КОМПОНЕНТОВ ДЛЯ МОНТИРОВАНИЯ
// ============================================
/*
Каждый компонент описывается объектом с полями:
- selector: CSS-селектор для поиска элемента
- component: импортированный Vue-компонент
- getProps: функция, собирающая пропсы с DOM-элемента
*/
const vueComponents = [
  // Компонент TelegramBlock
  {
    selector: 'telegram-block',
    component: TelegramBlock,
    getProps: (el) => ({
      qr: el.getAttribute('qr') || '',
      mobilicon: el.getAttribute('mobilicon') || '',
      link: el.getAttribute('link') || '',
      description: el.getAttribute('description') || ''
    })
  },
  
  // Компонент FormRating (с компиляцией шаблона)
  {
    selector: '.form-rating',
    component: FormRatingWrapper,
    getProps: (el, index) => ({
      instanceIndex: index,
      template: el.innerHTML  // Передаём содержимое как шаблон
    })
  }
  
  // Сюда будут добавляться новые компоненты
  // {
  //   selector: 'other-component',
  //   component: OtherComponent,
  //   getProps: (el) => ({ ... })
  // }
];

// ============================================
// ИНИЦИАЛИЗАЦИЯ ПРИ ЗАГРУЗКЕ DOM
// ============================================
document.addEventListener('DOMContentLoaded', () => {
  console.log('🔍 Поиск Vue компонентов для монтирования...');
  
  // Перебираем все зарегистрированные компоненты
  vueComponents.forEach(({ selector, component, getProps }) => {
    // Находим все элементы по селектору
    const elements = document.querySelectorAll(selector);
    
    if (elements.length === 0) {
      console.log(`ℹ️ Элементы "${selector}" не найдены на странице`);
      return;
    }

    // Для каждого найденного элемента создаём экземпляр
    elements.forEach((element, index) => {
      // Создаём уникальный ID для элемента (если нужно)
      if (!element.id) {
        element.id = `${selector.replace(/\W/g, '')}-${index}`;
      }

      // Получаем пропсы для этого экземпляра
      const props = getProps ? getProps(element, index) : {};
      
      // Монтируем приложение
      mountVueApp(component, element, props);
    });
  });
});

// ============================================
// НАБЛЮДАТЕЛЬ ЗА ДИНАМИЧЕСКИ ДОБАВЛЯЕМЫМИ ЭЛЕМЕНТАМИ
// ============================================
const observer = new MutationObserver((mutations) => {
  mutations.forEach((mutation) => {
    mutation.addedNodes.forEach((node) => {
      // Проверяем, что это элемент (nodeType 1)
      if (node.nodeType === 1) {
        // Проверяем по всем зарегистрированным селекторам
        vueComponents.forEach(({ selector, component, getProps }) => {
          if (node.matches?.(selector)) {
            console.log(`🔄 Обнаружен новый ${selector}, монтируем...`);
            
            // Считаем сколько уже есть таких элементов
            const existingCount = document.querySelectorAll(selector).length;
            
            // Получаем пропсы для нового экземпляра
            const props = getProps ? getProps(node, existingCount - 1) : {};
            
            // Монтируем приложение
            mountVueApp(component, node, props);
          }
        });
      }
    });
  });
});

// Запускаем наблюдение за изменениями в DOM
observer.observe(document.body, {
  childList: true,  // отслеживаем добавление/удаление элементов
  subtree: true     // отслеживаем во всех потомках
});

// ============================================
// ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ДЛЯ ОТЛАДКИ (опционально)
// ============================================
if (process.env.NODE_ENV === 'development') {
  // Функция для принудительного перемонтирования
  window.__remountVueApps = () => {
    console.log('🔄 Принудительное перемонтирование Vue приложений...');
    
    // Очищаем хранилище
    vueComponents.forEach(({ selector }) => {
      document.querySelectorAll(selector).forEach(el => {
        if (el.__vue_app__) {
          el.__vue_app__.unmount();
          initializedApps.delete(el);
          delete el.__vue_app__;
        }
      });
    });
    
    // Перезапускаем инициализацию
    document.dispatchEvent(new Event('DOMContentLoaded'));
  };
  
  console.log('ℹ️ Для перемонтирования используйте window.__remountVueApps()');
}
