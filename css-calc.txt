/* eslint-disable */
import axios from 'axios';
import store from "../../FamilyMortgageCalculator/store";
import bodyUnlockMobileFilter from '../../../modules/redesign-site/body-unlock-mobile-filter';
export function factoryShareMixin({store}) {
  return {
    data() {
      return {
        isAnimatingHolidays:false,
        flag_send_results:true,
        captcha_id:null,
        answer_code:null,
        lastSentAnswers: null, // Для хранения последних отправленных данных
        isSending: false,     // Флаг отправки запроса
        screenWidth : window.innerWidth,
        screenHeight: window.innerHeight,
      };
    },
    methods:{
      AddClassBody() {
        document.body.classList.add('body-modal');
        document.body.classList.add('body-additional-class');
        document.body.setAttribute('style', `top:-${window.scrollY}px;position: fixed;`);
        document.ontouchmove = (e) => {
          e.preventDefault();
        };
      },
      RemoveClassBody() {
        if (!document.body.classList.contains('body-modal-modals')) {
          document.body.classList.remove('body-modal');
          document.body.classList.remove('body-additional-class');
        }
        const scrollY = document.body.style.top;
        document.body.style.position = '';
        document.body.style.top = '';
        window.scrollTo(0, parseInt(scrollY || '0') * -1);
      },
      closeMobileTooltip(element) {
        const activeTooltip = document.querySelector('.js--tooltip-parent.tooltip-active');
        const target = activeTooltip || (element?.closest?.('.js--tooltip-parent.tooltip-active'));

        if (!target) {
          return;
        }

        target.classList.remove('tooltip-active');

        if (typeof bodyUnlockMobileFilter === 'function') {
          bodyUnlockMobileFilter(target);
        }
      },
      closeMobileTooltipMail(element) {
        const activeTooltip = document.querySelector('.js--tooltip-parent.tooltip-active');
        const target = activeTooltip || (element?.closest?.('.js--tooltip-parent.tooltip-active'));
        if (!target) {
          return;
        }
        target.classList.remove('tooltip-active');
      },


      async openModalMail(event) {
        try {
          if (this.hasAnswersChanged()) {
            await this.sendResults();
          }

          // Закрываем мобильную шторку перед открытием модального окна
          if (event?.currentTarget) {
            setTimeout(()=> {
              this.closeMobileTooltipMail(event.currentTarget);
            },1000)

          }

          this.showModal();
        } catch (error) {
          console.error('Ошибка при открытии модального окна:', error);
        }
      },

      showModal() {
        store.dispatch('ActionOpenModalMailStatus', true);
        if (!document.body.classList.contains('body-modal')) {
          this.AddClassBody()
        }
        document.body.classList.add('js--open-modal')
      },

      // Обработчик клика на соцсети (VK, TG, WP)
      async handleLinkClick(type, event) {
        try {
          const needSend = this.hasAnswersChanged();

          if (needSend) {
            await this.sendResults();
          }

          const link = this.answerLink?.[type];
          if (link) {
            window.open(link, '_blank');
          }

          //Закрываем мобилку, если не отправляли данные (или после отправки)
          // Передаём элемент, чтобы точно найти нужный родитель
          if (event?.currentTarget) {
            this.closeMobileTooltip(event.currentTarget);
          }

        } catch (error) {
          console.error('Ошибка при открытии ссылки:', error);
        }
      },
      // Обработчик клика на копирование
      // Обработчик клика на "Скопировать ссылку"
      async handleCopyClick(event) {
        const element = event?.currentTarget;
        try {
          if (this.hasAnswersChanged()) {
            await this.sendResults();
          }

          if (this.answerLink?.link) {
            //Колбэк просто закрывает шторку.
            // Тайминг (1 сек для закрытия, 3 сек для текста) — внутри copyLinkManually
            this.copyLinkManually(this.answerLink.link, element, () => {
              this.closeMobileTooltip(element);
            });
          }
        } catch (error) {
          console.error('Ошибка при копировании:', error);
        }
      },

      // Проверка, изменились ли данные
      hasAnswersChanged() {
        if (!this.lastSentAnswers) return true;
        return JSON.stringify(this.answersToSand) !== JSON.stringify(this.lastSentAnswers);
      },
      // Отправка результатов
      // Отправка результатов
      async sendResults() {
        if (this.isSending) return;
        this.isSending = true;

        try {
          // Получаем текущую ссылку (если answerLink или link не существует — null)
          let initialLink = null;
          if (this.answerLink && typeof this.answerLink.link !== 'undefined') {
            initialLink = this.answerLink.link;
          }
          await this.sendingResult('result');
          this.lastSentAnswers = JSON.parse(JSON.stringify(this.answersToSand));
          // Ждём обновления ссылки
          await this.waitForAnswerLinkUpdate(initialLink);
        } catch (error) {
          console.error('Ошибка при отправке результатов:', error);
          throw error;
        } finally {
          this.isSending = false;
        }
      },
      async waitForAnswerLinkUpdate(oldLink, timeout = 5000) {
        const start = Date.now();
        const checkInterval = 50;

        return new Promise((resolve, reject) => {
          const interval = setInterval(() => {
            let currentLink = null;
            // Проверяем, существует ли answerLink и его поле link
            if (this.answerLink && typeof this.answerLink.link !== 'undefined' && this.answerLink.link !== null) {
              currentLink = this.answerLink.link;
            }
            // Если появилась новая ссылка (и она отличается от старой)
            if (currentLink && currentLink !== oldLink) {
              clearInterval(interval);
              resolve();
            }
            // Таймаут
            if (Date.now() - start > timeout) {
              clearInterval(interval);
              reject(new Error('Таймаут ожидания обновления answerLink'));
            }
          }, checkInterval);
        });
      },

      // Унифицированная функция копирования
      copyLinkManually(link, element, onAfterAnimation = null) {
        try {
          if (navigator.clipboard) {
            navigator.clipboard.writeText(link).then(() => {
              this._updateCopyText(element, 'Скопировано');

              // Таймер 1: Возврат текста через 3 секунды
              setTimeout(() => {
                this._updateCopyText(element, 'Скопировать ссылку');
              }, 3000);

              // Таймер 2: Закрытие шторки через 1 секунду (если передан колбэк)
              if (typeof onAfterAnimation === 'function') {
                setTimeout(() => {
                  onAfterAnimation();
                }, 1000);
              }
            }).catch(() => {
              this._fallbackCopy(link, element, onAfterAnimation);
            });
          } else {
            this._fallbackCopy(link, element, onAfterAnimation);
          }
        } catch {
          this._fallbackCopy(link, element, onAfterAnimation);
        }
      },

// Резервный способ копирования
      _fallbackCopy(link, element, onAfterAnimation = null) {
        const textarea = document.createElement('textarea');
        textarea.value = link;
        textarea.style.position = 'fixed';
        textarea.style.opacity = '0';
        document.body.appendChild(textarea);
        textarea.focus();
        textarea.select();
        document.execCommand('copy');
        document.body.removeChild(textarea);

        this._updateCopyText(element, 'Скопировано');

        // Таймер 1: Возврат текста через 3 секунды
        setTimeout(() => {
          this._updateCopyText(element, 'Скопировать ссылку');
        }, 3000);

        // Таймер 2: Закрытие шторки через 1 секунду
        if (typeof onAfterAnimation === 'function') {
          setTimeout(() => {
            onAfterAnimation();
          }, 1000);
        }
      },

// Вспомогательный метод для обновления текста
      _updateCopyText(element, text) {
        const p = element?.querySelector?.('p');
        if (p) {
          p.textContent = text;
        }
      },
      // Инициализация капчи
      sendingResult(flag) {
        this.captchaInit(flag);
        const recaptchaKey = typeof conf !== 'undefined' ? conf.smartcaptcha_key : null;
        if (recaptchaKey) {
          window.smartCaptcha.execute(this.captcha_id);
        }
      },
      captchaInit(flag) {
        this.captcha_id = window.smartCaptcha.render('yandex-captcha-family-calculator', {
          sitekey: conf.smartcaptcha_key,
          invisible: true,
          callback: (token) => {
            if (flag === 'result') {
              this.sendingResultToApi(token);
            } else if (flag === 'mail') {
              this.sendingMailToApi(token);
            }
          },
        });
      },
      // Отправка данных в API
      sendingResultToApi(token) {
        const data = {
          calculatorId: this.calculatorId,
          answers: this.answersToSand,
          'smart-token': token,
        };
        return new Promise((resolve, reject) => {
          axios({
            method: 'post',
            url: '/api/local/calculator/answers/',
            headers: {
              'Content-type': 'application/json; charset=UTF-8',
              'X-Bitrix-Csrf-Token': window.BX.bitrix_sessid(),
            },
            data,
          })
            .then((res) => {
              if (res.data.code === 200 && res.data.result) {
                // Сохраняем ссылки в Vuex
                store.dispatch('ActionAnswerLink', res.data.result.answerLink);
                store.dispatch('ActionAnswersId', res.data.result.answersId);
                store.dispatch('ActionAnswersExpired',0)
              }
              if (res.data.code!==200) {
                if (res.data.description) {
                  store.dispatch('ActionDescriptionAfterSand', res.data.description);
                }
              }
              if (res.data.code) {
                this.answer_code = res.data.code;
              }
              resolve();
            })
            .catch((error) => {
              if (error.response) {
                if (error.response.data !== undefined) {
                  console.log(error.response);
                  store.dispatch('ActionDescriptionAfterSand', error.response.data.description);
                }
              }
              console.log(error);
              reject(error);
            });
        });
      },
      sendingMailToApi(token) {
        const modal_main_content = this.$refs.dataModalMail.querySelector('.js--modal-main-content');
        const modal_success_content = this.$refs.dataModalMail.querySelector('.js--modal-wr-success');
        const modal_failed_content = this.$refs.dataModalMail.querySelector('.js--modal-wr-failed');
        let data = {
          "userEmail": this.mask_mail.value,
          "smart-token": token,
          "answersId": this.answersId,
          "agree": this.check_agree_advertisement
        };
        axios({
          method: 'post',
          url: '/api/local/calculator/answers/sendmail/',
          headers: {
            "Content-type": "application/json; charset=UTF-8",
            'X-Bitrix-Csrf-Token': window.BX.bitrix_sessid(),
          },
          data: data
        })
          .then((res) => {
            if (modal_main_content) {
              modal_main_content.classList.add('unactive');
            }
            if (modal_success_content) {
              modal_success_content.classList.add('active');
            }
          })
          .catch((error) => {
            if (error.response.data !== undefined && error.response.data.description !== undefined) {
              console.log(error.response);
              this.description_after_sand_mail = error.response.data.description;
            }
            console.log(error);
            if (modal_main_content) {
              modal_main_content.classList.add('unactive');
            }
            if (modal_failed_content) {
              modal_failed_content.classList.add('active');
            }
          });
      },
    }
  }
}
