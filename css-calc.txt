clearFieldError(inputId) {
    const field = this.inputs.find(inp => inp.id === inputId);
    if (field) {
      field.isNotValid = false;
    }
  },

searchError() {
  if (this.inputs[0].value.length > 8) {
    if (this.$refs.searchResults[0]?.querySelectorAll('ul li').length === 0 && this.isShowError !== true) {
      this.showErrorSearch = true;
      // ✅ Если Dadata ничего не нашла, скрываем "Заполните поле" — поле-то заполнено!
      this.inputs[0].isNotValid = false;
    } else {
      this.showErrorSearch = false;
    }
  } else {
    this.showErrorSearch = false;
  }
},

computed: {
  // === НОВОЕ: Адрес валиден, только если он точно есть в подсказках Dadata ===
  isAddressValid() {
    const address = this.inputs[0].value?.trim();
    if (!address) return false;
    
    // Проверяем точное совпадение с одной из подсказок
    return this.searchList.includes(address);
  },
  
  // === Обновляем isFormValid ===
  isFormValid() {
    // 1. Все обязательные поля заполнены
    const allFilled = this.inputs.every(inp => !inp.validate || inp.value);
    
    // 2. Адрес должен быть валидным (выбран из подсказок)
    return allFilled && this.isAddressValid;
  },
  
  // ... остальные computed ...
},

handlerSendForm(argument) {
  // 1. Валидация полей
  this.validateAllInputs();
  
  if (!this.isValid) {
    return; // 🛑 Пустые поля
  }
  
  // 2. Проверка адреса: должен точно совпадать с подсказкой
  if (!this.isAddressValid) {
    this.inputs[0].isNotValid = true;
    this.ocenkaMobiErrorText = 'Выберите адрес из выпадающего списка';
    this.showError();
    return; // 🛑 Адрес не выбран → платный запрос НЕ уходит
  }
  
  // ✅ Всё ок → отправляем
  this.captchaInit();
  this.send_argument = argument;
  this.btnSendPress = true;
  
  // ... остальной код отправки ...
},
