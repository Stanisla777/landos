testFont() {
  import('@/utils/manrope-fonts.js').then(fonts => {
    pdfMake.fonts = {
      TestFont: {
        normal: fonts.manropeMediumBase64
      }
    };

    const docDefinition = {
      defaultStyle: { font: 'TestFont', fontSize: 12 },
      content: [{ text: 'Тест: Привет, Мир! 123', fontSize: 16 }]
    };

    pdfMake.createPdf(docDefinition).download('test-font.pdf');
  });
}
