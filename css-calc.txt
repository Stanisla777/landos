@click="onTrackClick"    <!-- ← Добавили -->
  ref="slider"

 onTrackClick(event) {
    if (this.isAnswered) return;

    // Вычисляем, куда кликнули относительно ширины трека
    const trackRect = this.$refs.slider.$el.getBoundingClientRect();
    const clickX = event.clientX - trackRect.left;
    const percent = Math.max(0, Math.min(1, clickX / trackRect.width));
    
    // Переводим процент в индекс и округляем до ближайшего маркера
    const maxIndex = this.answersList.length - 1;
    const nearest = Math.round(percent * maxIndex);
    
    // Если значение изменилось — обновляем (запускает CSS-анимацию)
    if (nearest !== this.selectedIndex) {
      this.selectedIndex = nearest;
    }
    
    this.answerValue = this.answersList[nearest]?.value;
    this.$emit('valuesIsChosen');
  },
