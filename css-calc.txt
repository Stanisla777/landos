mounted() {
  this.initMask(); // создаёт this.maskRef

  if (this.answers && this.region_not === false) {
    const valueObject = parseFloat(this.answers.familyIncome); // ← лучше parseFloat, если могут быть дроби
    if (!isNaN(valueObject) && valueObject > 0) {
      this.maskRef.value = valueObject; // imask сам отформатирует: 390000 → "390 000"
      // this.dataFieldIncome обновится автоматически через событие 'accept'
    }
  }
}
