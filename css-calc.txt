filters: {
    capitalize(str) {
      if (!str) return '';
      return str.charAt(0).toUpperCase() + str.slice(1)
    },
    format_decimal:(val) => {
      if (val!==undefined && val!==null){
        return new Intl.NumberFormat("ru-RU").format(Math.abs(val.toFixed(2)));
      }

    },
    format_month:(val) => {
      return val===1?'Январь':
        val===2?'Февраль':
          val===3?'Март':
            val===4?'Апрель':
              val===5?'Май':
                val===6?'Июнь':
                  val===7?'Июль':
                    val===8?'Август':
                      val===9?'Сентябрь':
                        val===10?'Октябрь':
                          val===11?'Ноябрь':
                            val===12?'Декабрь':''
    },
    replace_point:(val) =>{
      if (val == null) return '';
      return String(val).replace('.',',')
    }
  },
