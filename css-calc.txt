/**
 * Вспомогательная функция: показывает стрелки для активного слайда
 * и добавляет/удаляет класс .unactive в зависимости от количества слайдов
 */
function updateActiveArrows(parent, mainSwiper, activeIndex) {
  // 1. Находим вложенный слайдер в активном слайде главного
  const currentSlide = mainSwiper.slides[activeIndex];
  if (!currentSlide) return;

  const nestedSwiperEl = currentSlide.querySelector('.js--slider-construtor');
  if (!nestedSwiperEl) return;

  // 2. Читаем data-arrows, чтобы узнать классы стрелок
  const dataArrowsAttr = nestedSwiperEl.getAttribute('data-arrows');
  if (!dataArrowsAttr) return;

  let arrowConfig;
  try {
    arrowConfig = JSON.parse(dataArrowsAttr);
  } catch (e) {
    console.warn('Не удалось распарсить data-arrows:', dataArrowsAttr);
    return;
  }

  // 3. Берём класс левой или правой стрелки (любой, чтобы найти контейнер)
  const arrowClass = arrowConfig.arrow_left || arrowConfig.arrow_right;
  if (!arrowClass) return;

  // Убираем точку в начале, если она есть
  const className = arrowClass.replace(/^\./, '');

  // 4. Находим стрелку по классу и поднимаемся к контейнеру
  const targetArrow = parent.querySelector(`.${className}`);
  if (!targetArrow) return;

  const activeArrows = targetArrow.closest('.js--swiper-arrow-prise-container');
  if (!activeArrows) return;

  // 5. Скрываем все контейнеры стрелок, показываем только активный
  const allContainers = parent.querySelectorAll('.js--swiper-arrow-prise-container');
  allContainers.forEach(container => {
    container.classList.add('display-none');
  });
  activeArrows.classList.remove('display-none');

  // 6. ▼▼▼ КЛЮЧЕВОЕ ИЗМЕНЕНИЕ: не выходим, если слайдер не готов, а ждём его ▼▼▼
  if (!nestedSwiperEl.swiper) {
    // Пробуем ещё раз через 150мс (одна попытка — достаточно для вашего кейса)
    setTimeout(() => {
      if (nestedSwiperEl.swiper) {
        finalizeArrows(nestedSwiperEl.swiper, activeArrows);
      }
    }, 150);
    return;
  }

  // Если слайдер уже готов — обрабатываем сразу
  finalizeArrows(nestedSwiperEl.swiper, activeArrows);
}

/**
 * Отдельная функция: финальная настройка стрелок и свайпа
 * (вынесена, чтобы не дублировать код для основного и отложенного вызова)
 */
function finalizeArrows(swiper, activeArrows) {
  // 🔑 Важно: сбрасываем allowTouchMove в true перед проверкой
  // Это чинит свайп, если он был отключён ранее (например, paginationSlider)
  swiper.allowTouchMove = true;
  
  // Обновляем слайдер (теперь он точно видим и инициализирован)
  swiper.update();
  if (swiper.params.autoHeight) {
    swiper.updateAutoHeight();
  }

  // Небольшая задержка, чтобы браузер успел проставить классы disabled
  setTimeout(() => {
    const arrowButtons = activeArrows.querySelectorAll('.swiper-slide-arrow');
    let disabledCount = 0;

    arrowButtons.forEach(btn => {
      if (btn.classList.contains('swiper-button-disabled')) {
        disabledCount++;
      }
    });

    // Логика .unactive: скрываем стрелки, если некуда крутить
    if (swiper.slides.length <= 1 || disabledCount >= 2) {
      activeArrows.classList.add('unactive');
      swiper.allowTouchMove = false;
    } else {
      activeArrows.classList.remove('unactive');
      swiper.allowTouchMove = true;
    }
  }, 50);
}

// Вместо одного setTimeout — цикл с 3 попытками
let attempts = 0;
const tryInit = () => {
  attempts++;
  if (nestedSwiperEl.swiper) {
    finalizeArrows(nestedSwiperEl.swiper, activeArrows);
    return;
  }
  if (attempts < 3) {
    setTimeout(tryInit, 100);
  }
};
tryInit();
