plugins: [
  store => {
    // Флаг для предотвращения множественных вычислений
    let isCalculating = false;
    
    const calculateAndUpdate = () => {
      if (isCalculating) return;
      isCalculating = true;
      
      // Обновляем answers_to_send
      Vue.set(store.state.answers_to_send, 'loanAmount', store.state.cost_property);
      Vue.set(store.state.answers_to_send, 'initialPayment', store.state.initial_payment);
      Vue.set(store.state.answers_to_send, 'bet', store.state.annualInterestRate);
      Vue.set(store.state.answers_to_send, 'loanTerm', store.state.loanTerm);
      
      // Выполняем расчет только если все параметры валидны
      if (store.state.loanAmount > 0 && store.state.annualInterestRate >= 0 && store.state.loanTerm > 0) {
        store.commit('mutationTotalCalculate');
      }
      
      isCalculating = false;
    };
    
    // Отслеживаем изменения всех необходимых параметров
    store.watch(
      () => [
        store.state.cost_property,
        store.state.initial_payment,
        store.state.annualInterestRate,
        store.state.loanTerm,
        store.state.loanAmount
      ],
      calculateAndUpdate,
      { deep: true, immediate: true }
    );
  }
]
