<template>
  <div v-if="isLoading" class="stories-loading">
    <div class="spinner"></div>
    <div class="loading-text">Загрузка историй...</div>
  </div>
  <Stories v-else v-bind="$attrs" />
</template>

<script>
import { defineAsyncComponent } from 'vue';
import store from './store';

export default {
  name: 'StoriesLazy',
  inheritAttrs: false,
  components: {
    Stories: defineAsyncComponent(() => import('./Stories.vue')),
  },
  data() {
    return {
      isLoading: true,
    };
  },
  async mounted() {
    await store.dispatch('axiosGetParametr');
    this.isLoading = false;
  },
};
</script>
