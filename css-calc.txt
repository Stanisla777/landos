handlerTab(activeId) {
  // ✅ Гарантия, что activeTab всегда валидный индекс
  if (activeId < 0 || activeId >= this.tabs.length) {
    activeId = 0;
  }
  
  for (let i = 0; i < this.tabs.length; i += 1) {
    this.tabs[i].isActive = false;
  }
  this.tabs[activeId].isActive = true;
  this.activeTab = activeId;
},

mounted() {
  this.$nextTick(() => {
    // ✅ Гарантия валидного activeTab при монтировании
    if (!this.tabs[this.activeTab]) {
      this.activeTab = 0;
    }
    setTimeout(this.initMap, 1000);
  });
},
