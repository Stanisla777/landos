import {
  vueAppManager,
  FormRatingUniversal,
  TelegramBlock,
  SimpleRatingUniversal,
  UsefulnessSurvey,
  HousingCooperativeRequest
} from './vue-apps';

// Регистрируем типы приложений
vueAppManager.registerAppType('form-rating', FormRatingUniversal);
vueAppManager.registerAppType('telegram-block', TelegramBlock);
vueAppManager.registerAppType('simple-rating', SimpleRatingUniversal);
vueAppManager.registerAppType('usefulness-survey', UsefulnessSurvey);
vueAppManager.registerAppType('housing-cooperative', HousingCooperativeRequest);

document.addEventListener('DOMContentLoaded', () => {

const containers = document.querySelectorAll('[data-vue-app]');

  containers.forEach((container, index) => {
    const appType = container.dataset.vueApp;
    const instanceId = container.dataset.instanceId || `${appType}-${index}`;

    // Собираем все data-атрибуты как пропсы
    const props = { instanceId };

    // Копируем все data-атрибуты (кроме vue-app и instance-id)
    Object.keys(container.dataset).forEach(key => {
      if (key !== 'vueApp' && key !== 'instanceId') {
        // Преобразуем kebab-case в camelCase (data-star-count → starCount)
        const camelKey = key.replace(/-([a-z])/g, g => g[1].toUpperCase());
        props[camelKey] = container.dataset[key];
      }
    });

    // Создаём уникальный ID для контейнера
    const containerId = `vue-app-${instanceId}`;
    container.id = containerId;

    // Создаём экземпляр
    vueAppManager.createInstance(appType, containerId, props);
  });

  // Показываем статистику
  vueAppManager.info();

})
