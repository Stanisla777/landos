const chokidar = require('chokidar');

// Настройки devServer для разработки
config.devServer = {
  port: 8080,
  open: true,
  hot: true,
  liveReload: true,
  watchFiles: {
    paths: ['src/ejs/**/*.ejs', 'src/pug/**/*.pug', 'src/**/*'],
    options: {
      usePolling: true,
      interval: 1000
    }
  },
  static: {
    directory: path.join(__dirname, 'dist'),
    watch: true,
    publicPath: '/'
  },
  client: {
    overlay: {
      errors: true,
      warnings: false
    },
    progress: true,
    reconnect: true
  },
  devMiddleware: {
    publicPath: '/',
    writeToDisk: false
  },
  //  Добавляем middleware для отслеживания EJS
  onBeforeSetupMiddleware: (devServer) => {
    // Отслеживаем изменения в EJS-файлах
    const watcher = chokidar.watch('src/ejs/**/*.ejs', {
      ignored: /node_modules/,
      persistent: true,
      usePolling: true,
      interval: 500
    });

    watcher.on('change', (filePath) => {
      console.log(` EJS файл изменён: ${filePath}`);
      
      // Отправляем сигнал на перезагрузку
      if (devServer.socketServer) {
        devServer.socketServer.sockets.forEach((socket) => {
          socket.send(JSON.stringify({
            type: 'static-changed'
          }));
        });
      }
    });
  }
};
