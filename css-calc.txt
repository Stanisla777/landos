<template>
  <div class="test-n-slider" :class="addClasses">
    <!-- 
      Слайдер работает с ИНДЕКСАМИ (0, 1, 2, 3...), а не с реальными значениями.
      step="1" и snap="true" делают поведение идентичным радиокнопкам.
    -->
    <Slider
      v-model="selectedIndex"
      :min="0"
      :max="answersList.length - 1"
      :step="1"
      :snap="true"
      :tooltip="false"
      :height="15"
      :dot-size="32"
      :disabled="isAnswered"
      @change="onSliderRelease"
      class="test-n-slider__track"
    />
    
    <!-- Маркеры позиционируются РАВНОМЕРНО -->
    <div class="test-n-slider__marks" :class="{ 'is-disabled': isAnswered }">
      <div
        v-for="(item, index) in answersList"
        :key="item.value"
        class="test-n-slider__mark"
        :class="{ 'is-active': index === selectedIndex }"
        :style="{ left: getEvenPosition(index) + '%' }"
        @click.stop="selectByIndex(index)"
      >
        <span class="test-n-slider__mark-dot"></span>
        <span class="test-n-slider__mark-label">{{ item.label }}</span>
      </div>
    </div>

    <div class="test-n-slider__value">{{ currentValueLabel }}</div>
  </div>
</template>

<script>
import Slider from '@vueform/slider';

export default {
  name: 'TextNSlider',
  components: { Slider },
  props: {
    options: { type: Object, required: true },
    isAnswered: { type: Boolean, default: false }
  },
  data() {
    return {
      selectedIndex: 0,
      answerValue: null,
      isAnswerValueCorrect: null
    }
  },
  computed: {
    // Преобразуем объект {10000: "10k", 20000: "20k"} → массив [{value:10000, label:"10k"}, ...]
    answersList() {
      if (!this.options.answers) return [];
      return Object.entries(this.options.answers).map(([val, label]) => ({
        value: +val,
        label
      }));
    },
    addClasses() {
      return {
        'correct-answer': this.isAnswerValueCorrect === true,
        'uncorrect-answer': this.isAnswerValueCorrect === false
      }
    },
    currentValueLabel() {
      const current = this.answersList[this.selectedIndex];
      return current ? current.label : this.answersList[0]?.label;
    }
  },
  watch: {
    isAnswered(val) {
      if (val) {
        this.checkAnswerValue();
        this.prepareDataForSend();
      } else {
        this.setDefault();
        this.selectedIndex = 0;
      }
    }
  },
  methods: {
    // Равномерное распределение: 0%, 33.3%, 66.6%, 100%
    getEvenPosition(index) {
      if (this.answersList.length <= 1) return 0;
      return (index / (this.answersList.length - 1)) * 100;
    },

    // Срабатывает при отпускании ползунка
    onSliderRelease() {
      // selectedIndex уже целое число благодаря step=1 и snap=true
      const selectedItem = this.answersList[this.selectedIndex];
      if (!selectedItem) return;

      this.answerValue = selectedItem.value;
      this.$emit('valuesIsChosen');
      
      // Опционально: если нужна мгновенная проверка ответа при выборе
      // this.checkAnswerValue(); 
    },

    // Клик по маркеру
    selectByIndex(index) {
      if (this.isAnswered) return;
      this.selectedIndex = index;
      this.onSliderRelease();
    },

    checkAnswerValue() {
      this.isAnswerValueCorrect = this.answerValue === this.options.correctAnswer;
    },
    setDefault() {
      this.selectedIndex = 0;
      this.answerValue = this.answersList[0]?.value;
      this.isAnswerValueCorrect = null;
      this.$emit('valuesIsUnchosen');
    },
    prepareDataForSend() {
      this.$emit('readyForDataSending', {
        answerValue: this.answerValue,
        isAnswerValueCorrect: this.isAnswerValueCorrect
      });
    }
  },
  mounted() {
    this.selectedIndex = 0;
    this.answerValue = this.answersList[0]?.value;
  }
}
</script>
