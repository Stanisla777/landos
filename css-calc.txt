<Slider
  v-model="selectedIndex"
  :min="0"
  :max="answersList.length - 1"
  :step="0.01"          <!-- ✅ Свободное плавное перетаскивание -->
  :tooltip="false"
  :height="15"
  :dot-size="32"
  :disabled="isAnswered"
  @change="onSliderRelease" <!-- Срабатывает только при ОТПУСКАНИИ -->
  class="test-n-slider__track"
/>

onSliderRelease() {
  if (this.isAnswered) return;
  
  // Находим ближайший целый индекс
  const nearestIndex = Math.round(this.selectedIndex);
  
  // Обновляем значение → срабатывает реактивность → слайдер плавно едет к маркеру
  this.selectedIndex = nearestIndex;
  
  // Сохраняем реальное значение ответа
  this.answerValue = this.answersList[nearestIndex]?.value;
  this.$emit('valuesIsChosen');
}
