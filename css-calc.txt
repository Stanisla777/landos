    { loader: 'css-loader', options: { sourceMap: true, url: false } },
    {
      loader: 'postcss-loader',
      options: {
        sourceMap: true,
        postcssOptions: {
          plugins: [require('autoprefixer')]
        }
      }
    },
    {
      loader: 'sass-loader',
      options: {
        api: 'modern-compiler',
        sourceMap: true,
        sassOptions: {
          quietDeps: true,
          silenceDeprecations: [
            'import', 'legacy-js-api', 'global-builtin',

          ]
        }
      }
    }
  ]
},
