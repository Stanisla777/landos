//Проверка на заполненность полей, чтобы сделать кнопку отправки активной
function fullnessFieldToActiveButton() {
  let count_required = 0;
  const array_form = document.querySelectorAll('.js--fullness-field');
  for (let item of array_form) {
    const count_required_field = item.querySelectorAll('input[required]');
    for (let element of count_required_field) {
      element.addEventListener('change', (e) => {
        if (element.hasAttribute('required') && element.classList.contains('js--validate-mask--check')) {
          if (e.target.checked) {
            count_required+=1
          }
          if (!e.target.checked && count_required!==0) {
            count_required-=1
          }

        }
        if(count_required===count_required_field.length) {
          element.closest('.js--fullness-field').querySelector('.js--button-status-form').classList.remove('disabled')
        }
        else {
          element.closest('.js--fullness-field').querySelector('.js--button-status-form').classList.add('disabled')
        }
      });
    }
  }

}
