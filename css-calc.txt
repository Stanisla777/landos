<Slider
  v-model="selectedIndex"
  :min="0"
  :max="answersList.length - 1"
  :step="0.01"
  :tooltip="false"
  :height="15"
  :dot-size="32"
  :disabled="isAnswered"
  @change="onSliderChange"
  @dragstart="onDragStart"
  @dragend="onDragEnd"
  class="test-n-slider__track"
/>


 isDragging: false  // ← Новый флаг

onDragStart() {
    this.isDragging = true;
  },
  
  // ← Срабатывает при окончании перетаскивания
  onDragEnd() {
    this.isDragging = false;
    // Если отпустили после драга — применяем "примагничивание"
    this.snapToNearest();
  },
  
  // ← Срабатывает при клике по треку (не при драге!)
  onSliderChange() {
    // Если сейчас идёт драг — игнорируем, обработаем в onDragEnd
    if (this.isDragging) return;
    
    // Если клик по треку — сразу применяем "примагничивание"
    this.snapToNearest();
  },
