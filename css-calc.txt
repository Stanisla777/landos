document.addEventListener('click', (e) => {
  if (e.target.classList.contains('js--close-mobile-tooltip-2')) {
    const wrapper = e.target.closest('.mobile-tooltip-wrapper');
    if (wrapper) {
      wrapper.remove();
    } else {
      document.querySelectorAll('.mobile-tooltip-wrapper').forEach(el => el.remove());
    }
    RemoveClassBody();
  }
});
