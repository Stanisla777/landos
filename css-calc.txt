methods: {
  // ... остальные методы

  syncCalendarHeight() {
    this.$nextTick(() => {
      if (this.$refs.eventsColumn && this.$refs.calendarColumn) {
        const eventsHeight = this.$refs.eventsColumn.offsetHeight;
        this.$refs.calendarColumn.style.height = `${eventsHeight}px`;
      }
    });
  }
}

mounted() {
  this.parseEvents();
  this.calculateDateBounds();
  this.setDefaultMonth();
  this.setDefaultSelection();
  this.syncCalendarHeight();
  window.addEventListener('resize', this.syncCalendarHeight);
},

beforeDestroy() {
  window.removeEventListener('resize', this.syncCalendarHeight);
},

watch: {
  selectedDate() {
    this.syncCalendarHeight();
  }
}
