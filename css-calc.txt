localStorageLoading() {
  const local_stage = Cookies.get(this.cookie_key);
  
  // Проверка: куки существует, это число и больше 0
  if (local_stage && !isNaN(parseInt(local_stage))) {
    const rating = parseInt(local_stage);
    const parent = this.$refs.formRating;
    
    // Защита: если реф не найден, выходим
    if (!parent) return;
    
    const array_star = parent.querySelectorAll('.js--grade-item');
    
    // Защита: если элементов нет, выходим
    if (!array_star.length) return;
    
    // Ограничиваем рейтинг количеством доступных звёзд
    const starsToCheck = Math.min(rating, array_star.length);
    
    for (let i = 0; i < starsToCheck; i++) {
      const index = (array_star.length - 1) - i;
      // Дополнительная защита от отрицательных индексов
      if (index >= 0 && array_star[index]) {
        array_star[index].checked = true;
      }
    }
    
    this.rate = rating;
    this.form_show = false;
  }
}

changeGrade(event, index) {
  const parent = this.$refs.formRating;
  if (!parent) return;
  
  this.rate = this.array_star.length - index;
  const array_star = parent.querySelectorAll('.js--grade-item');
  
  if (!array_star.length) return;

  // Снимаем выделение со всех
  for (let i = 0; i < array_star.length; i++) {
    if (array_star[i]) array_star[i].checked = false;
  }
  
  // Ставим выделение нужным (с защитой индекса)
  const starsToCheck = Math.min(this.rate, array_star.length);
  for (let i = 0; i < starsToCheck; i++) {
    const idx = (array_star.length - 1) - i;
    if (idx >= 0 && array_star[idx]) {
      array_star[idx].checked = true;
    }
  }

  // ... остальная логика с placeholder и hiddenInput
}
