// modules/scroll-depth-tracker-clever-data.js

const CONFIG = {
  thresholds: [25, 50, 75, 100],
  eventPrefix: 'scroll_',
  throttleMs: 200,
  offsetBottom: 1000 // расчёт процентов высота страницы - offsetBottom
};

const state = {
  sent: new Set(),
  maxReached: 0,
  lastCheck: 0
};

function sendToCleverData(eventName) {
  console.log('eventName');
  console.log(eventName);
  if (typeof window.dmpkitdl !== 'undefined' && window.dmpkitdl !== null) {
    try {
      window.dmpkitdl.push({ event: eventName });
    } catch (e) {
      console.warn('[ScrollDepth] error sending to dmpkitdl:', e);
    }
  }
}

function getScrollPercent() {
  const scrollTop = window.scrollY || document.documentElement.scrollTop;
  const viewportHeight = document.documentElement.clientHeight;
  const docHeight = document.documentElement.scrollHeight;

  const scrollableHeight = docHeight - viewportHeight;

  if (scrollableHeight <= CONFIG.offsetBottom) {
    return scrollTop > 0 ? 100 : 0;
  }

  const effectiveHeight = scrollableHeight - CONFIG.offsetBottom;
  const percent = (scrollTop / effectiveHeight) * 100;

  return Math.min(100, Math.max(0, Math.round(percent)));
}

function checkScrollDepth() {
  const now = Date.now();
  if (now - state.lastCheck < CONFIG.throttleMs) return;
  state.lastCheck = now;

  const currentPercent = getScrollPercent();

  // eslint-disable-next-line no-restricted-syntax
  for (const threshold of CONFIG.thresholds) {
    if (threshold > state.maxReached && currentPercent >= threshold) {
      const eventName = CONFIG.eventPrefix + threshold;
      if (!state.sent.has(threshold)) {
        sendToCleverData(eventName);
        state.sent.add(threshold);
        state.maxReached = threshold;
      }
      break;
    }
  }
}

export default function scrollDepthTrackerCleverData() {
  checkScrollDepth();

  let ticking = false;

  const onScroll = () => {
    if (!ticking) {
      window.requestAnimationFrame(() => {
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  const onResize = () => {
    if (!ticking) {
      window.requestAnimationFrame(() => {
        checkScrollDepth();
        ticking = false;
      });
      ticking = true;
    }
  };

  window.addEventListener('scroll', onScroll, { passive: true });
  window.addEventListener('resize', onResize, { passive: true });

  return () => {
    window.removeEventListener('scroll', onScroll);
    window.removeEventListener('resize', onResize);
  };
}
