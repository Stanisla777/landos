function initCommentsAnswerDelegation() {
  // Защита от повторной инициализации
  if (window.commentsAnswerDelegationInitialized) return;
  window.commentsAnswerDelegationInitialized = true;

  document.addEventListener('click', (e) => {
    const button = e.target.closest('.comments__item-button-answer');
    if (!button) return;
    
    e.preventDefault();
    
    const commentItem = button.closest('.comments__item');
    if (!commentItem) return;
    
    const parentId = commentItem.dataset.id;
    const titleEl = commentItem.querySelector('.comments__item-title');
    const authorName = titleEl && titleEl.textContent ? titleEl.textContent.trim() : '';
    
    const elementIdInput = document.querySelector('.js--form-comments .form-comments__container input[name="elementId"]');
    const captchaId = elementIdInput && elementIdInput.value ? elementIdInput.value : `S${Date.now()}`;
    
    // 1. Закрываем и удаляем другие открытые формы
    document.querySelectorAll('.comments__item-form-answer_active').forEach((form) => {
      const currentReplyForm = commentItem.querySelector('.comments__item-form-answer');
      if (form !== currentReplyForm) {
        removeReplyForm(form);
      }
    });
    
    // 2. Проверяем, есть ли уже форма
    let replyForm = commentItem.querySelector('.comments__item-form-answer');
    
    if (!replyForm) {
      let formHTML = '';
      if (typeof window.REPLY_FORM_TEMPLATE === 'string') {
        formHTML = window.REPLY_FORM_TEMPLATE
          .replace(/{captchaId}/g, captchaId)
          .replace(/{parentId}/g, parentId)
          .replace(/{authorName}/g, authorName);
      }
      
      const itemInfo = commentItem.querySelector('.comments__item-info');
      if (itemInfo) {
        itemInfo.insertAdjacentHTML('beforeend', formHTML);
        replyForm = itemInfo.querySelector('.comments__item-form-answer');
        
        // Инициализация валидации
        if (typeof CommentsFormValidationAnswer === 'function') {
          CommentsFormValidationAnswer();
        }
      }
    }
    
    // 3. Показываем форму
    if (replyForm) {
      replyForm.classList.add('comments__item-form-answer_active');
      
      // Фокус на textarea
      const textarea = replyForm.querySelector('.js--form-comments-text');
      if (textarea) {
        setTimeout(() => textarea.focus(), 100);
      }
    }

document.addEventListener('DOMContentLoaded', function() {
  initCommentsAnswerDelegation(); // ← Добавляем эту строку
  // ... остальные инициализации
});
  });
}
