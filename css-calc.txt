/* eslint-disable */

export default function catalogNewSendFormNativeJs() {
  // === STATE ===
  const state = {
    step_1: true,
    error_step_1: true,
    error_step_2: true,
    fullness_input_name: 0,
    fullness_input_mail: 0,
    fullness_input_tel: 0,
    fullness_texarea_program: 0,
    fullness_texarea_doc: 0,
    fullness_input_radio: 0,
    fullness_select: 0,
    fullness_input: 0,
    show_warn: true,
    all_checked: true,
    hiddenInput: '',
    InputSearhPlace: ''
  };

  // === DOM REFERENCES ===
  const modal = document.querySelector('#modal-catalog-new-request');
  if (!modal) return { destroy: () => {} };

  const step1 = modal.querySelector('.js--step-1');
  const step2 = modal.querySelector('.js--step-2');
  const form = modal.querySelector('.js--housing-cooperative-request');
  const btnContinueStep1 = modal.querySelector('.modal-form-izhs__wr-final-button .btn_s.green_color_white.desabled');
  const btnNextStep1 = modal.querySelector('.modal-form-izhs__wr-final-button .btn_s.black');
  const btnSendStep2 = modal.querySelector('.modal-form-izhs__step-2 .btn_s.green_color_white.desabled');
  const btnSendStep2Black = modal.querySelector('.modal-form-izhs__step-2 .btn_s.black.js--btn-universal-form');
  const btnBackStep2 = modal.querySelector('.modal-form-izhs__step-2 .btn_s.transparent_black_border');

  // === EMAIL REGEX (из Vue-кода) ===
  const emailRegex = /^[a-z0-9!#$%&'*+/=?^_`{|}~-]+(?:\.[a-z0-9!#$%&'*+/=?^_`{|}~-]+)*@(?:[a-z0-9](?:[a-z0-9-]*[a-z0-9])?\.)+[a-z0-9](?:[a-z0-9-]*[a-z0-9])?$/i;

  // === HELPERS: UPDATE UI ===
  function updateStepVisibility() {
    if (step1 && step2) {
      step1.style.display = state.step_1 ? 'block' : 'none';
      step2.style.display = state.step_1 ? 'none' : 'block';
    }
  }

  function updateStep1Buttons() {
    const allFilled = state.fullness_input >= 7; // 7 полей в шаге 1
    state.error_step_1 = !allFilled;
    
    if (btnContinueStep1 && btnNextStep1) {
      if (state.error_step_1) {
        btnContinueStep1.style.display = 'inline-flex';
        btnNextStep1.style.display = 'none';
      } else {
        btnContinueStep1.style.display = 'none';
        btnNextStep1.style.display = 'inline-flex';
      }
    }
  }

  function updateStep2Buttons() {
    const parent = modal.querySelector('.js--step-2');
    if (!parent) return;
    
    const allFields = parent.querySelectorAll('.js--validate-mask');
    const filledFields = parent.querySelectorAll('.js--validate-mask.js-filled');
    
    state.error_step_2 = filledFields.length !== allFields.length;
    
    if (btnSendStep2 && btnSendStep2Black) {
      if (state.error_step_2) {
        btnSendStep2.style.display = 'inline-flex';
        btnSendStep2Black.style.display = 'none';
      } else {
        btnSendStep2.style.display = 'none';
        btnSendStep2Black.style.display = 'inline-flex';
      }
    }
  }

  function updateFullnessInput() {
    state.fullness_input = 
      state.fullness_input_name + 
      state.fullness_input_tel + 
      state.fullness_input_mail + 
      state.fullness_input_radio + 
      state.fullness_select + 
      state.fullness_texarea_program + 
      state.fullness_texarea_doc;
    updateStep1Buttons();
  }

  // === HANDLERS: STEP 1 INPUTS ===
  function handleNameInput(element) {
    state.fullness_input_name = element.value.length > 0 ? 1 : 0;
    updateFullnessInput();
  }

  function handleEmailInput(element) {
    if (element.value.length > 0 && emailRegex.test(element.value)) {
      state.fullness_input_mail = 1;
    } else {
      state.fullness_input_mail = 0;
    }
    updateFullnessInput();
  }

  function handleTelInput(element) {
    state.fullness_input_tel = element.value.length >= 16 ? 1 : 0;
    updateFullnessInput();
  }

  function handleProgramInput(element) {
    state.fullness_texarea_program = element.value.length > 0 ? 1 : 0;
    updateFullnessInput();
  }

  function handleDocInput(element) {
    state.fullness_texarea_doc = element.value.length > 0 ? 1 : 0;
    updateFullnessInput();
  }

  // === HANDLER: STEP 2 FIELDS ===
  function handleStep2Field(element) {
    if (element.type === 'checkbox') {
      element.classList.toggle('js-filled', element.checked);
    } else {
      element.classList.toggle('js-filled', element.value.length > 0);
    }
    updateStep2Buttons();
  }

  // === NAVIGATION ===
  function nextSlide() {
    state.step_1 = false;
    updateStepVisibility();
  }

  function prevSlide() {
    state.step_1 = true;
    updateStepVisibility();
  }

  // === VALIDATION: STEP 1 SUBMIT ===
  function sendingRequestFirstStep(element) {
    const parent = element.closest('.js--step-1');
    if (!parent) return;
    
    const inputs = parent.querySelectorAll('.js--validate-mask');
    
    inputs.forEach(item => {
      if (item.classList.contains('js--validate-mask-select')) {
        if (!item.value) {
          item.closest('.js--izhs-select')?.classList.add('input_error');
        }
      } else if (item.classList.contains('js--validate-mask')) {
        if (!item.value) {
          const label = item.closest('label');
          label?.classList.add('input_error');
          
          if (label && !label.querySelector('.js--input__error_required')) {
            const errorEl = document.createElement('p');
            errorEl.classList.add('input__error', 'js--input__error_required');
            errorEl.textContent = 'Поле обязательно для заполнения';
            label.appendChild(errorEl);
          }
        }
      }
    });
  }

  // === FORM SUBMISSION CALLBACK (MutationObserver) ===
  function onFormSubmitSuccess() {
    const observerOptions = { attributes: true };
    
    const observer = new MutationObserver(() => {
      if (modal.hasAttribute('data-formsubmit') && 
          modal.getAttribute('data-formsubmit') === 'success' && 
          !modal.classList.contains('open')) {
        
        setTimeout(() => {
          // Сброс чекбокса, если есть
          const right1 = modal.querySelector('#right_1');
          if (right1) right1.checked = true;
          
          // Сброс state
          state.show_warn = true;
          state.step_1 = true;
          state.error_step_1 = true;
          state.error_step_2 = true;
          state.fullness_input = 0;
          state.fullness_radio = 0;
          state.hiddenInput = '';
          state.all_checked = true;
          state.InputSearhPlace = '';
          state.fullness_input_name = 0;
          state.fullness_input_mail = 0;
          state.fullness_input_tel = 0;
          state.fullness_select = 0;
          state.fullness_input_radio = 0;
          
          // Сброс UI
          updateStepVisibility();
          updateStep1Buttons();
          updateStep2Buttons();
          
          // Сброс классов js-filled
          modal.querySelectorAll('.js-filled').forEach(el => el.classList.remove('js-filled'));
          
          // Сброс ошибок
          modal.querySelectorAll('.input_error').forEach(el => el.classList.remove('input_error'));
          modal.querySelectorAll('.js--input__error_required').forEach(el => el.remove());
          
          // Сброс значений полей
          modal.querySelectorAll('input, textarea').forEach(el => {
            if (el.type !== 'checkbox') el.value = '';
          });
          
          modal.setAttribute('data-formsubmit', '');
        }, 1000);
      }
    });
    
    observer.observe(modal, observerOptions);
    return () => observer.disconnect();
  }

  // === EVENT LISTENERS SETUP ===
  const listeners = [];

  function addListener(el, event, handler, options) {
    if (el) {
      el.addEventListener(event, handler, options);
      listeners.push({ el, event, handler });
    }
  }

  function setupListeners() {
    // Step 1 inputs
    const nameInput = modal.querySelector('.js--validate-mask-name');
    const emailInput = modal.querySelector('.js--validate-mask-email');
    const telInput = modal.querySelector('.js--validate-mask-tel');
    const programTextarea = modal.querySelector('.modal-form-izhs__step-1 .js--validate-mask-message');
    const docTextarea = modal.querySelectorAll('.modal-form-izhs__step-1 .js--validate-mask-message')[1];

    addListener(nameInput, 'keyup', (e) => handleNameInput(e.target));
    addListener(emailInput, 'keyup', (e) => handleEmailInput(e.target));
    addListener(telInput, 'keyup', (e) => handleTelInput(e.target));
    addListener(programTextarea, 'keyup', (e) => handleProgramInput(e.target));
    addListener(docTextarea, 'keyup', (e) => handleDocInput(e.target));

    // Step 2 fields
    const step2Fields = modal.querySelectorAll('.js--step-2 .js--validate-mask, .js--step-2 input[type="checkbox"]');
    step2Fields.forEach(field => {
      const eventType = field.type === 'checkbox' ? 'change' : 'keyup';
      addListener(field, eventType, (e) => handleStep2Field(e.currentTarget));
    });

    // Navigation buttons
    addListener(btnNextStep1, 'click', nextSlide);
    addListener(btnBackStep2, 'click', prevSlide);
    
    // Step 1 "continue" button (when error)
    if (btnContinueStep1) {
      addListener(btnContinueStep1, 'click', (e) => sendingRequestFirstStep(e.currentTarget));
    }

    // Prevent default form submit if needed (универсальная форма)
    if (form) {
      addListener(form, 'submit', (e) => {
        // Форма отправляется через universal-js, просто предотвращаем стандартное поведение
        e.preventDefault();
      });
    }
  }

  // === INIT ===
  function init() {
    updateStepVisibility();
    updateStep1Buttons();
    updateStep2Buttons();
    setupListeners();
    const destroyObserver = onFormSubmitSuccess();
    
    return {
      destroy: () => {
        listeners.forEach(({ el, event, handler }) => {
          el.removeEventListener(event, handler);
        });
        destroyObserver?.();
      }
    };
  }

  return init();
}
