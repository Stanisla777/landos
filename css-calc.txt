const componentMap = {
  'housing-cooperative': () => import('./vue-apps/HousingCooperativeRequest.vue')
  // Добавляйте новые приложения сюда:
  // 'another-app': () => import('./vue-apps/AnotherApp.vue')
};

// Функция для динамической загрузки компонента
async function loadComponent(appType) {
  const loader = componentMap[appType];
  if (!loader) {
    console.error(`❌ Неизвестный тип приложения: ${appType}`);
    return null;
  }
  try {
    console.log(`📦 Загрузка компонента: ${appType}`);
    const module = await loader();
    return module.default || module;
  } catch (error) {
    console.error(`❌ Ошибка загрузки ${appType}:`, error);
    return null;
  }
}

// Функция для создания экземпляра приложения (без изменений)
function createAppInstance(container, appType, index) {
  const instanceId = container.dataset.instanceId || `${appType}-${index}`;

  const props = { instanceId };
  Object.keys(container.dataset).forEach(key => {
    if (key !== 'vueApp' && key !== 'instanceId') {
      const camelKey = key.replace(/-([a-z])/g, g => g[1].toUpperCase());
      props[camelKey] = container.dataset[key];
    }
  });

  const containerId = `vue-app-${instanceId}`;
  container.id = containerId;

  vueAppManager.createInstance(appType, containerId, props);
}

const observer = new MutationObserver(async (mutations) => {
  for (const mutation of mutations) {
    for (const node of mutation.addedNodes) {
      if (node.nodeType === 1 && node.matches?.('[data-vue-app]')) {
        const appType = node.dataset.vueApp;

        // Если компонент ещё не зарегистрирован — загружаем
        if (!vueAppManager.appTypes.has(appType)) {
          const component = await loadComponent(appType);
          if (component) {
            vueAppManager.registerAppType(appType, component);
          } else {
            continue;
          }
        }

        const existingCount = document.querySelectorAll(`[data-vue-app="${appType}"]`).length;
        createAppInstance(node, appType, existingCount - 1);
      }
    }
  }
});

observer.observe(document.body, {
  childList: true,
  subtree: true
});

window.vueAppManager = vueAppManager;
