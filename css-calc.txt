<template>
  <div class="polls__component-question js--polls-component-radio"
       :class="required === 'Y' ? 'required' : 'optional'">
    <p class="polls__question" v-html="question"></p>
    <div class="polls__list-answers js--list-answers">
      <div class="checkbox-stylized polls__list-answers-item polls__list-checkbox js--list-variant"
           v-for="(answer, key, ind) in variants"
           :key="answer.ID || key"
           :data-sort="answer.ID"
           @click="pass_clickVariant($event, answer.FIELD_WIDTH, (answer['~FIELD_PARAM'] || answer.FIELD_PARAM), answer.COLOR, answer.NAME_INPUT, answer.VALUE_INPUT, answer.ID, id_question)">
        
        <input
          :id="`${id_question}-${item_key}-${key}`"
          :name="`${id_question}-${item_key}`"
          type="checkbox"
          value=""
          @click.stop
          @change="pass_click($event, answer.FIELD_WIDTH, (answer['~FIELD_PARAM'] || answer.FIELD_PARAM), answer.COLOR, answer.NAME_INPUT, answer.VALUE_INPUT, answer.ID, id_question)"
        />
        <label :for="`${id_question}-${item_key}-${key}`" v-html="answer.MESSAGE"></label>
        
        <p class="polls__answer-to-question js--answer-to-question" 
           v-if="answer.COLOR == 1 && views_test == 'Test2'">
          {{ (answer['~FIELD_PARAM'] || answer.FIELD_PARAM) }}
        </p>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'v-conponent-checkbox',
  props: ['question', 'variants', 'item_key', 'required', 'current_answer', 'id_question', 'views_test'],
  methods: {
    pass_click(elem, param_1, param_2, param_3, param_4, param_5, param_6, param_7) {
      const element = elem.currentTarget;
      const data = [param_4, param_5, param_6, param_7, element];
      if (element.closest('.js--polls-component-radio').classList.contains('required')) {
        this.$emit('event_SelectCheckForButton', data);
      }
    },
    pass_clickVariant(elem, param_1, param_2, param_3, param_4, param_5, param_6, param_7) {
      const element = elem.currentTarget;
      const status = !element.querySelector('input').checked;
      const data = [param_4, param_5, param_6, param_7, element];
      
      if (status === true) {
        this.$emit('event_SelectCheckboxAdd', data);
      } else {
        this.$emit('event_SelectCheckboxDelete', data);
      }
      this.$emit('event_clickVariant', [element, status]);
    }
  }
};
</script>
