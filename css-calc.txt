  preloadAllImages({ state }) {
    return new Promise((resolve) => {
      const allStories = state.STORIES.DETAIL;
      const imagesToLoad = [];
      
      // Собираем все URL изображений из всех историй
      allStories.forEach(story => {
        story.slides.forEach(slide => {
          if (slide.image?.url) {
            imagesToLoad.push(slide.image.url);
          }
          if (slide.video?.preimg) {
            imagesToLoad.push(slide.video.preimg);
          }
        });
      });
      
      if (imagesToLoad.length === 0) {
        resolve();
        return;
      }
      
      let loadedCount = 0;
      const totalImages = imagesToLoad.length;
      
      imagesToLoad.forEach(url => {
        const img = new Image();
        img.onload = () => {
          loadedCount++;
          if (loadedCount === totalImages) {
            resolve();
          }
        };
        img.onerror = () => {
          loadedCount++;
          if (loadedCount === totalImages) {
            resolve();
          }
        };
        img.src = url;
      });
    });
  },
}
