    };
  },
  computed: {
    isValid() {
      for (let i = 0; i < this.inputs.length; i += 1) {
        if (this.inputs[i].validate && this.inputs[i].isNotValid) {
          return false;
        }
      }
      return true;
    },
    showFilter() {
      return !!this.social.length;
    }
  },
  filters: {
    format(val) {
      const num = Number(val);
      const sign = Math.sign(num);
      const absNum = Math.abs(num);

      if (absNum >= 1.0e9) {
        return (sign * (absNum / 1.0e9).toFixed(3) + " блн").replace('.', ',');
      } else if (absNum >= 1.0e6) {
        return (sign * (absNum / 1.0e6).toFixed(3) + " млн").replace('.', ',');
      } else if (absNum >= 1.0e3) {
        return (sign * (absNum / 1.0e3).toFixed(1) + " тыс").replace('.', ',');
      }
      return String(val);
    },
    formatSpace(val) {
      return String(val).replace(/(\d)(?=(\d{3})+([^\d]|$))/g, '$1 ').replace('.', ',');
    },
    formatDate(val) {
      return val.split('-').reverse().join('.');
    },
    formatPointComma(val) {
      return String(val).replace('.', ',');
    }
  },
  watch: {
    inputs: {
      handler() {
        if (this.btnSendPress) {
          this.validateAllInputs();
        }
      },
      deep: true
    },
    searchList(newVal) {
      this.showSearchList = !!newVal.length;
    }
  },
  methods: {
    handlerToggleSelect(data) {
      this.openSelect = data;
    },
    setSelectItem(val, title) {
      this.inputs[2].value = val;
      this.inputs[2].plh = title;
      this.handlerToggleSelect(false);
    },
    validInput(curInp) {
      curInp.isNotValid = curInp.value === '';
    },
    validateAllInputs() {
      for (let i = 0; i < this.inputs.length; i += 1) {
        if (this.inputs[i].validate) {
          this.validInput(this.inputs[i]);
        }
      }
    },
    closeBullet() {
      this.bullet.showBullet = false;
    },
    handlerTab(activeId) {
      for (let i = 0; i < this.tabs.length; i += 1) {
        this.tabs[i].isActive = false;
      }
      this.tabs[activeId].isActive = true;
      this.activeTab = activeId;
    },
    resetData() {
      if (this.map !== null) {
        this.map.destroy();
      }
      this.bullet.showBullet = false;
      this.bullet.address = '';
      this.bullet.priceMin = 0;
      this.bullet.price = 0;
      this.bullet.stations = [];
      this.bldPos = {};
      this.bldPhotos = null;
      this.handlerTab(0);
      for (let i = 0; i < this.tabs.length; i += 1) {
        const contentArr = this.tabs[i].content;
        for (let j = 0; j < contentArr.length; j += 1) {
          contentArr[j].data = '';
          if (contentArr[j].link) {
            contentArr[j].linkUrl = '';
          }
        }
      }
      this.social = [];
      this.isShowError = false;
      this.idActiveFilter = 0;
      this.flat = null;
      this.bulletInfra = {
        title: '',
        subtitle: '',
        duration: '',
        link: '',
        photos: []
      };
      this.bulletInfraShow = false;
      this.showSearchList = false;
    },
    initMap(dataPos) {
      const pos = dataPos || [55.7558, 37.6173];
      const zoomData = dataPos ? 16 : 12;
      const showObj = !!dataPos;
      const self = this;

      window.ymaps.ready(function () {
        self.map = new window.ymaps.Map('map', {
          center: pos,
          controls: ['smallMapDefaultSet'],
          zoom: zoomData
        });

        if (showObj) {
          self.map.geoObjects.add(new window.ymaps.GeoObject({
            geometry: { type: 'Point', coordinates: pos },
            properties: { iconContent: self.bullet.address }
          }, {
            preset: 'islands#blackStretchyIcon',
            iconColor: 'green'
          }));
        }

        self.map.container.fitToViewport();

        // Map click event
        self.map.events.add('click', function (e) {
          const coords = e.get('coords');
          self.map.geoObjects.removeAll();

          window.ymaps.geocode(coords).then(function (res) {
            const firstGeoObject = res.geoObjects.get(0);
            self.map.geoObjects.add(new window.ymaps.GeoObject({
              geometry: { type: 'Point', coordinates: coords },
              properties: { iconContent: firstGeoObject.getAddressLine() }
            }, {
              preset: 'islands#blackStretchyIcon',
              iconColor: 'green'
            }));
            self.inputs[0].value = firstGeoObject.getAddressLine();
          });
        });

        // GeoObjects click event for infrastructure
        self.map.geoObjects.events.add('click', function (e) {
          if (self.social[self.idActiveFilter]) {
            const objectsArr = self.social[self.idActiveFilter].items;
            const thisPlacemark = e.get('target');
            const ballonNumber = thisPlacemark.properties._data.balloonContentHeader || 0;
            const bulletContent = objectsArr[ballonNumber];
            self.bulletInfra = bulletContent;
            self.setRoute(objectsArr[ballonNumber].pos);
          }
        });
      });

    setRoute(objectsPos) {
      this.bulletInfraShow = false;
      if (this.multiRoute) {
        this.map.geoObjects.remove(this.multiRoute);
      }

      this.multiRoute = new window.ymaps.multiRouter.MultiRoute({
        referencePoints: [
          [this.bldPos.lat, this.bldPos.lng],
          [objectsPos.lat, objectsPos.lng]
        ],
        params: {
          routingMode: 'pedestrian',
          reverseGeocoding: false
        }
      }, {
        wayPointVisible: false,
        routeActiveStrokeWidth: 6,
        routeActiveStrokeStyle: 'dot',
        routeActiveStrokeColor: '#FED141'
      });

      this.map.geoObjects.add(this.multiRoute);
      this.bulletInfra.link = `https://yandex.ru/maps/?ll=${objectsPos.lng},${objectsPos.lat}&z=16&l=map`;
      this.getRouteData();
      setTimeout(this.getRouteData, 600);
    },
    getRouteData() {
      const activeRoute = this.multiRoute?.getActiveRoute();
      if (activeRoute) {
        this.bulletInfra.duration = activeRoute.properties.get('duration').text;
        this.bulletInfraShow = true;
      } else {
        setTimeout(this.getRouteData, 400);
      }
    },
    setObjectToMap() {
      this.bulletInfraShow = false;
      if (this.myCollection) {
        this.myCollection.removeAll();
      }

      const objectsArr = this.social[this.idActiveFilter].items;
      this.myCollection = new window.ymaps.GeoObjectCollection();

      for (let i = 0; i < objectsArr.length; i += 1) {
        this.myCollection.add(new window.ymaps.Placemark(
          [objectsArr[i].pos.lat, objectsArr[i].pos.lng],
          { balloonContentHeader: i },
          { preset: 'islands#darkBlueCircleDotIcon', openBalloonOnClick: false }
        ));
      }

      this.setRoute(objectsArr[0].pos);
      this.bulletInfra = objectsArr[0];
      this.map.geoObjects.add(this.myCollection);
    },

      const element = event.currentTarget;
      if (!element.classList.contains('is-active')) {
        this.idActiveFilter = id;
        this.setObjectToMap();
      }
    },
    clearFilter() {
      this.idActiveFilter = null;
      this.bulletInfraShow = false;
      if (this.myCollection) {
        this.map.geoObjects.remove(this.myCollection);

      if (this.multiRoute) {

          this.isShowError = false;

            headers: {
              'Content-Type': 'application/json',
              'X-Bitrix-Csrf-Token': window.BX?.bitrix_sessid?.(),
              Accept: 'application/json',
            }
          };

          localaxios.post('/api/local/suggestions/address/', {
            query: this.inputs[0].value,
            count: 8
          }, config)
            .then((response) => {
              this.searchList = [];
              if (response.data.result.suggestions) {
                for (let i = 0; i < response.data.result.suggestions.length; i += 1) {
                  const fullName = response.data.result.suggestions[i].value;
                  if (fullName) {
                    this.searchList.push(fullName);
                    if (this.searchList.length) {
                      this.handlerSelectAddress(true);
                    }
                  }
                }
              }
            })
            .then(this.searchError)
            .catch((error) => {
              console.log(error);
              if (error?.request?.status === 403 || error?.request?.status === 405) {
                this.ocenkaMobiErrorText = 'Сервер не отвечает. Попробуйте перезагрузить страницу';
                this.showErrorSearch = false;
                this.showError();
              }
            });
        }
      }, 500);
    },
    setAddress(data) {
      this.inputs[0].value = data;
      this.handlerSelectAddress(false);
    },
    handlerSelectAddress(data) {
      this.showSearchList = data;
    },
    firstZero(event) {
      const input = event.target;
      this.inputs[1].value = input.value
        .replace(/^0+/gi, '1')
        .replace(',', '.')
        .replace(/[^\d\.]/g, "")
        .replace(/\./, "x")
        .replace(/\./g, "")
        .replace(/x/, ".");
    },
    searchError() {
      if (this.inputs[0].value.length > 8) {
        const results = this.$refs.searchResults?.[0]?.querySelectorAll('ul li')?.length || 0;
        if (results === 0 && !this.isShowError) {
          this.showErrorSearch = true;
        } else {
          this.showErrorSearch = false;
        }
      } else {
        this.showErrorSearch = false;
      }
    },
    clearField() {
      this.inputs[0].value = '';
      this.inputs[1].value = '';
      if (this.$refs.SelectList && this.$refs.SelectList[0]) {
        const firstOption = this.$refs.SelectList[0].querySelector('.select__list-item:first-child');
        if (firstOption) firstOption.click();
      }
      this.bullet.showBullet = false;
      this.bulletInfraShow = false;
      this.showBuildInfo = false;
      this.showSocial = false;
      this.inputs[0].isNotValid = false;
      this.inputs[1].isNotValid = false;
      if (this.map) {
        this.map.geoObjects.removeAll();
      }
    },
    goBack() {
      this.bulletInfraShow = false;
      this.showBuildInfo = false;
      this.bullet.showBullet = true;
      this.showSocial = false;
      if (this.myCollection) {
        this.map.geoObjects.remove(this.myCollection);
      }
      if (this.multiRoute) {
        this.map.geoObjects.remove(this.multiRoute);
      }
    },
    showError() {
      this.isShowError = true;
    },
    handlerSendForm(argument) {
      // Здесь будет логика с капчей и API
      this.btnSendPress = true;
      this.validateAllInputs();
      if (this.isValid) {
        // API call implementation
        console.log('Форма валидна, отправка...');
      }
    },
    handlerBuildInfo(data, token) {
      this.showBuildInfo = data;
      this.showSocial = data;
      if (data) {
        this.$nextTick(() => {
          import('swiper').then(({ Swiper, Navigation, Pagination }) => {
            Swiper.use([Navigation, Pagination]);
            new Swiper('#swiper-appraisal', {
              slidesPerView: 1,
              spaceBetween: 80,
              loop: true,
              observer: true,
              observeParents: true,
              pagination: { el: '.swiper-pagination', clickable: true },
              navigation: { nextEl: '.swiper-button-next', prevEl: '.swiper-button-prev' }
            });
          });
        });
        this.closeBullet();
        scrollIt(document.querySelector('#form'), 400, 'linear');
      }
    }
  },
  mounted() {
    this.$nextTick(() => {
      setTimeout(() => this.initMap(), 1000);
    });
  }
};
