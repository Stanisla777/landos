<script>
(function() {
  const originalAdd = document.body.classList.add;
  document.body.classList.add = function(...classes) {
    if (classes.includes('body-modal') || classes.includes('body-modal-modals')) {
      console.warn('НУ КТО ДОБАВЛЯЕТ ЭТИ КЛАССЫ, А?', classes, new Error().stack);
    }
    return originalAdd.apply(this, classes);
  };
})();
</script>
