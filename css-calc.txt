mutationCalculate(state){
  state.arr_data_small_debt = []
  state.annuity_monthly_payment = 0
  state.differentiated_monthly_payment = []
  state.annuity_debt_plus_interest = 0
  state.annuity_overpayment = 0
  state.payment_annuity_list = []
  state.payment_differentiated_list = []
  state.dateArray = []

  // рассчитываю платежи
  const interest_rate = (state.interest_rate / 12) / 100
  
  // формирую даты
  const month_array = []
  const month_year = []
  var amountOfDatesFromPast = state.credit_time
  
  for (var i = 0; i < amountOfDatesFromPast; i++){
    var dateObj = new Date()
    dateObj.setMonth(dateObj.getMonth() + i)
    var month = dateObj.getUTCMonth() + 1
    month_array.push(month)
    var year = dateObj.getUTCFullYear()
    month_year.push(year)
  }
  
  for(let i=0; i<month_array.length; i++){
    if(month_array[i] === month_array[i+1]){
      month_array[i] = month_array[i] - 1
    }
  }
  for(let i=0; i<month_array.length; i++){
    month_array[i] === 1 ? month_array[i] = 'Январь' : month_array[i] === 2 ? month_array[i] = 'Февраль' : month_array[i] === 3 ? month_array[i] = 'Март'
      : month_array[i] === 4 ? month_array[i] = 'Апрель' : month_array[i] === 5 ? month_array[i] = 'Май' : month_array[i] === 6 ? month_array[i] = 'Июнь'
        : month_array[i] === 7 ? month_array[i] = 'Июль' : month_array[i] === 8 ? month_array[i] = 'Август' : month_array[i] === 9 ? month_array[i] = 'Сентябрь'
          : month_array[i] === 10 ? month_array[i] = 'Октябрь' : month_array[i] === 11 ? month_array[i] = 'Ноябрь' : month_array[i] = 'Декабрь'
  }
  for(let i=0; i<month_array.length; i++){
    state.dateArray.push(month_array[i] + ' ' + month_year[i])
  }

  // ==================== АННУИТЕТНЫЙ ПЛАТЁЖ ====================
  if (state.type_credit === 'ann'){
    // === ИСПРАВЛЕННАЯ ФОРМУЛА АННУИТЕТА ===
    let annuity_ratio
    if (interest_rate === 0) {
      annuity_ratio = 1 / state.credit_time
    } else {
      annuity_ratio = interest_rate / (1 - Math.pow(1 + interest_rate, -state.credit_time))
    }
    
    const annuity_payment_full = state.loan_amount * annuity_ratio
    const annuity_monthly_payment = parseFloat(annuity_payment_full.toFixed(2))
    
    state.annuity_monthly_payment = annuity_monthly_payment
    state.annuity_debt_plus_interest = annuity_monthly_payment * state.credit_time
    state.annuity_overpayment = state.annuity_debt_plus_interest - state.loan_amount
    state.necessary_income = annuity_monthly_payment / 0.6

    // === ФОРМИРОВАНИЕ ГРАФИКА С ЗАЩИТОЙ ===
    let obj_payment = {}
    let main_debt = state.loan_amount // полная точность

    for(let i = 0; i <= state.credit_time - 1; i++){
      if (main_debt <= 0.005) break
      
      const monthly_interest_payment = main_debt * interest_rate
      let principal_payment = annuity_payment_full - monthly_interest_payment
      
      let isLastPayment = false
      if (principal_payment > main_debt) {
        principal_payment = main_debt
        isLastPayment = true
      }
      
      main_debt = main_debt - principal_payment
      
      const payment_amount = isLastPayment 
        ? principal_payment + monthly_interest_payment 
        : annuity_monthly_payment
      
      obj_payment.payment = parseFloat(payment_amount.toFixed(2))
      obj_payment.principal_payment = parseFloat(principal_payment.toFixed(2))
      obj_payment.monthly_interest_payment = parseFloat(monthly_interest_payment.toFixed(2))
      obj_payment.remainder = parseFloat(main_debt.toFixed(2))
      obj_payment.month = state.dateArray[i]
      
      state.payment_annuity_list.push(obj_payment)
      obj_payment = {}
    }
    
    state.share_percent_small_chart = ((state.annuity_overpayment * 100) / state.annuity_debt_plus_interest).toFixed(1)
    state.arr_data_small_debt.push(state.annuity_overpayment)
    state.arr_data_small_debt.push(state.appartment_price)
    if(state.share_percent_small_chart == "NaN"){
      state.share_percent_small_chart = 0
    }
  }

  // ==================== ДИФФЕРЕНЦИРОВАННЫЙ ПЛАТЁЖ ====================
  if (state.type_credit === 'def'){
    // === ИСПРАВЛЕНИЕ: фиксируем equal_payment с полной точностью ===
    const equal_payment_full = state.loan_amount / state.credit_time
    
    // Сначала считаем массив платежей для отображения
    state.differentiated_monthly_payment = []
    for(let i = 0; i <= state.credit_time - 1; i++){
      const remaining_balance = state.loan_amount - equal_payment_full * i
      const interest = remaining_balance * interest_rate
      const principal = equal_payment_full
      const payment = principal + interest
      
      state.differentiated_monthly_payment.push(parseFloat(payment.toFixed(2)))
    }

    // Долг + проценты
    state.annuity_debt_plus_interest = state.differentiated_monthly_payment.reduce((a, b) => parseFloat(a) + parseFloat(b), 0)
    state.annuity_overpayment = state.annuity_debt_plus_interest - state.loan_amount
    state.necessary_income = state.differentiated_monthly_payment[0] / 0.6

    // === ФОРМИРОВАНИЕ ГРАФИКА С ЗАЩИТОЙ ===
    let obj_payment = {}
    let main_debt = state.loan_amount

    for(let i = 0; i <= state.differentiated_monthly_payment.length - 1; i++){
      if (main_debt <= 0.005) break
      
      const monthly_interest_payment = main_debt * interest_rate
      
      // === КЛЮЧЕВОЕ ИСПРАВЛЕНИЕ: не позволяем погасить больше, чем осталось ===
      let principal_payment = equal_payment_full
      let isLastPayment = false
      
      if (principal_payment > main_debt) {
        principal_payment = main_debt
        isLastPayment = true
      }
      
      main_debt = main_debt - principal_payment
      
      // Пересчитываем платёж для последнего месяца
      const payment_amount = isLastPayment 
        ? principal_payment + monthly_interest_payment 
        : state.differentiated_monthly_payment[i]
      
      obj_payment.payment = parseFloat(payment_amount.toFixed(2))
      obj_payment.principal_payment = parseFloat(principal_payment.toFixed(2))
      obj_payment.monthly_interest_payment = parseFloat(monthly_interest_payment.toFixed(2))
      obj_payment.remainder = parseFloat(main_debt.toFixed(2))
      obj_payment.month = state.dateArray[i]
      
      state.payment_differentiated_list.push(obj_payment)
      obj_payment = {}
    }
    
    state.share_percent_small_chart = ((state.annuity_overpayment * 100) / state.annuity_debt_plus_interest).toFixed(1)
    state.arr_data_small_debt.push(state.annuity_overpayment)
    state.arr_data_small_debt.push(state.appartment_price)
    if(state.share_percent_small_chart == "NaN"){
      state.share_percent_small_chart = 0
    }
  }
},
