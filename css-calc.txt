export default function SearchEasternArcticBuild() {
  const testElements = document.querySelectorAll('search-eastern-arctic-mortgage');
  testElements.forEach((item) => {
    console.log('item');
    console.log(item);
    const hint = item.getAttribute('title-hint');
    if (item.closest('#search-eastern-arctic-mortgage') && item.hasAttribute('v-cloak')) {
      item.removeAttribute('v-cloak')
    }

    // Создаём контейнер с правильными атрибутами для менеджера
    const vueContainer = document.createElement('div');
    vueContainer.setAttribute('data-vue-app', 'search-eastern-arctic-mortgage');
    vueContainer.setAttribute('data-title-hint', hint);
    // Заменяем старый элемент новым
    item.parentNode.replaceChild(vueContainer, item);
  });
}

</div>
