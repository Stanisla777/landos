resetCellHeights() {
  // Проверяем, существует ли this.$refs.calendarColumn
  if (this.$refs.calendarColumn) {
    // Получаем все ячейки
    const cells = this.$refs.calendarColumn.querySelectorAll('.marathon-2025__calendar-day');
    
    // Проверяем, что cells существует и это NodeList с элементами
    if (cells && cells.length > 0) {
      cells.forEach(cell => {
        cell.style.height = ''; // Сбрасываем inline-style
      });
    }
  }
}
