methods: {
  // Безопасный парсинг: возвращает объект или {}
  safeJsonParse(value) {
    if (!value) return {};
    if (typeof value === 'object' && value !== null && !Array.isArray(value)) {
      return value;
    }
    if (typeof value === 'string') {
      try {
        return JSON.parse(value);
      } catch (e) {
        console.warn('Ошибка парсинга JSON:', e);
        return {};
      }
    }
    return {};
  },

  parseDataMethod(){
    // 1. Парсим базовые данные (контейнер)
    const containerData = this.safeJsonParse(this.modalData.htmlTooltipContainer);
    
    // 2. Парсим специфичные данные (элемент)
    const itemData = this.safeJsonParse(this.modalData.htmlTooltip);
    
    // 3. Объединяем: itemData имеет приоритет при совпадении ключей
    this.parseData = { ...containerData, ...itemData };
  },
  
  handleOpen(event) {
    this.modalData = event.detail || {};
    this.parseDataMethod(); // ← Сначала парсим
    this.$nextTick(() => {
      this.setupEventListeners();
    });
    this.showModal = true;
    document.body.classList.add('modal-opened');
  },
  // ... остальные методы
}
