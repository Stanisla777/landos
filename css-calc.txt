"scripts": {
  "dev": "rspack build --mode productionDev && node scripts/compile-pug.js && prettier --print-width=120 --parser html --write dist/*.html",
  "build": "rspack build --mode production && node scripts/compile-pug.js",
  "start": "nodemon --watch src/ejs --ext ejs --exec rspack serve --mode development",
  "lint": "eslint --ext .js, --ignore-path .gitignore ."
}

module.exports = (env, argv) => {
  const isProduction = argv.mode === 'production';
  const isProductionDev = argv.mode === 'productionDev';

  if (isProduction) {
    // ===== PRODUCTION (npm run build) =====
    config.mode = 'production';
    config.devtool = false;
    config.optimization.minimize = true;

    config.output.filename = 'js/[name].bundle.js';
    config.output.chunkFilename = 'js/[name].bundle.js';
    config.output.publicPath = '/dist/';

    // Отключаем генерацию HTML
    config.plugins = config.plugins.filter(plugin =>
      !(plugin instanceof rspack.HtmlRspackPlugin)
    );

    const definePluginIndex = config.plugins.findIndex(p => p instanceof rspack.DefinePlugin);
    if (definePluginIndex !== -1) {
      config.plugins[definePluginIndex] = new rspack.DefinePlugin({
        __VUE_OPTIONS_API__: JSON.stringify(true),
        __VUE_PROD_DEVTOOLS__: JSON.stringify(false),
        __VUE_PROD_HYDRATION_MISMATCH_DETAILS__: JSON.stringify(false),
        'process.env.NODE_ENV': JSON.stringify('production')
      });
    }

  } else if (isProductionDev) {
    // ===== PRODUCTION DEV (npm run dev) =====
    config.mode = 'production'; // ← режим production для минификации
    config.devtool = false;
    config.optimization.minimize = true; // ✅ минификация

    config.output.filename = 'js/[name].bundle.js';
    config.output.chunkFilename = 'js/[name].bundle.js';
    config.output.publicPath = '/dist/';

    // НЕ отключаем HTML — нужны для бэкендера

    const definePluginIndex = config.plugins.findIndex(p => p instanceof rspack.DefinePlugin);
    if (definePluginIndex !== -1) {
      config.plugins[definePluginIndex] = new rspack.DefinePlugin({
        __VUE_OPTIONS_API__: JSON.stringify(true),
        __VUE_PROD_DEVTOOLS__: JSON.stringify(false),
        __VUE_PROD_HYDRATION_MISMATCH_DETAILS__: JSON.stringify(false),
        'process.env.NODE_ENV': JSON.stringify('production') // ← production для минификации
      });
    }

  } else {
    // ===== DEVELOPMENT SERVER (npm run start) =====
    config.mode = 'development';
    config.devtool = 'source-map';
    config.optimization.minimize = false;
    config.output.filename = 'js/[name].bundle.js';
    config.output.publicPath = '/';

    config.devServer = {
      port: 8081,
      open: false,
      hot: true,
      liveReload: false,
      static: {
        directory: path.join(__dirname, 'dist'),
        watch: true,
        publicPath: '/'
      },
      devMiddleware: {
        publicPath: '/',
        writeToDisk: true
      },
      client: {
        overlay: {
          errors: true,
          warnings: false
        },
        progress: true,
        reconnect: true
      }
    };
  }

  return config;
};
