setupMiddlewares: (middlewares, devServer) => {
  const fs = require('fs');
  const path = require('path');
  
  const ejsDir = path.join(__dirname, 'src/ejs');
  
  let isCompiling = false;
  let pendingChanges = false;
  
  const compile = () => {
    if (isCompiling) {
      pendingChanges = true;
      return;
    }
    
    isCompiling = true;
    pendingChanges = false;
    
    devServer.compiler.run((err, stats) => {
      isCompiling = false;
      
      if (err) {
        console.error('Compilation error:', err);
      } else {
        console.log('✅ Recompiled successfully');
        
        if (devServer.webSocketServer) {
          devServer.webSocketServer.clients.forEach((client) => {
            if (client.readyState === 1) {
              client.send(JSON.stringify({ type: 'ok' }));
            }
          });
        }
        
        devServer.middleware.invalidate();
      }
      
      // Если были изменения во время компиляции - запускаем ещё раз
      if (pendingChanges) {
        compile();
      }
    });
  };
  
  fs.watch(ejsDir, { recursive: true }, (eventType, filename) => {
    if (filename && filename.endsWith('.ejs')) {
      console.log(`🔄 EJS changed: ${filename}`);
      compile();
    }
  });

  return middlewares;
}
