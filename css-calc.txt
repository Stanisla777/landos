// Настройки devServer для разработки
config.devServer = {
  port: 8080,
  open: true,
  hot: true,
  liveReload: true,
  watchFiles: {
    paths: ['src/ejs/**/*.ejs', 'src/pug/**/*.pug', 'src/**/*'],
    options: {
      usePolling: true,
      interval: 1000
    }
  },
  static: {
    directory: path.join(__dirname, 'dist'),
    watch: true,
    publicPath: '/'
  },
  client: {
    overlay: {
      errors: true,
      warnings: false
    },
    progress: true,
    reconnect: true
  },
  devMiddleware: {
    publicPath: '/',
    writeToDisk: false
  },
  // 👇 Добавляем middleware с динамическим импортом
  setupMiddlewares: (middlewares, devServer) => {
    // Динамический импорт chokidar
    import('chokidar').then(({ default: chokidar }) => {
      const watcher = chokidar.watch('src/ejs/**/*.ejs', {
        ignored: /node_modules/,
        persistent: true,
        usePolling: true,
        interval: 500
      });

      watcher.on('change', (filePath) => {
        console.log(`🔄 EJS файл изменён: ${filePath}`);
        
        // Отправляем сигнал на перезагрузку
        if (devServer.webSocketServer) {
          devServer.webSocketServer.clients.forEach((client) => {
            if (client.send) {
              client.send(JSON.stringify({
                type: 'static-changed'
              }));
            }
          });
        }
      });
    }).catch(err => {
      console.error('Ошибка загрузки chokidar:', err);
    });

    return middlewares;
  }
};
