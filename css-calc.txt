<template>
  <!-- 
    shouldRender: контролирует рендер при загрузке (v-if)
    isHidden: контролирует видимость после клика (класс)
  -->
  <div 
    v-if="shouldRender" 
    class="cookies-vue" 
    :class="{ 'cookies-vue--hidden': isHidden }"
  >
    <div class="cookies-vue__body">
      <div class="cookies-vue__des">
        <p v-html="description"></p>
      </div>
      <div class="btn_s white_color_black_main medium mobile-100" @click="setCookies">
        {{ button }}
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: "Cookies",
  props: {
    instanceId: { type: String, required: true },
    description: { type: String, default: 'Подтвердить' },
    button: { type: String, default: 'Подтвердить' }
  },
  data() {
    return {
      shouldRender: false, // Не рендерим по умолчанию
      isHidden: false,     // Видим по умолчанию
    };
  },
  methods: {
    getCookie(name) {
      const matches = document.cookie.match(new RegExp(
        `(?:^|; )${name.replace(/([\.$?*|{}\(\)\[\]\\\/\+^])/g, '\\$1')}=([^;]*)`
      ));
      return matches ? decodeURIComponent(matches[1]) : undefined;
    },
    setCookies(){
      document.cookie = 'cookiewindowclosed=yes; path=/; max-age=157248000';
      this.isHidden = true; // Скрываем, но НЕ убираем из DOM
      
      // Отправляем событие для Telegram (если нужно)
      if (window.innerWidth <= 480) {
        setTimeout(() => {
          window.dispatchEvent(new CustomEvent('cookies-closed-mobile'));
        }, 15000);
      }
    }
  },
  mounted() {
    // Проверяем куки ТОЛЬКО при монтировании
    if (!this.getCookie('cookiewindowclosed')) {
      this.shouldRender = true; // Рендерим компонент
    }
    // Если куки есть → shouldRender остаётся false → компонент не в DOM
  }
}
</script>

<style scoped>
.cookies-vue {
  /* Плавное скрытие */
  transition: opacity 0.3s ease, visibility 0.3s ease;
}

.cookies-vue--hidden {
  opacity: 0;
  visibility: hidden; /*элемент скрыт, но занимает место в flexbox */
  pointer-events: none;
  height: auto; /* Сохраняет высоту контента */
}
</style>
