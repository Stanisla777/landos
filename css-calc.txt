mutationCalculateOldCredit(state) {
  // Проверка на наличие всех необходимых данных
  if (!state.old_debt || !state.old_rate || !state.credit_old_time || !state.start_payment || !state.start_payment[0]) {
    return;
  }

  // Очищаем массивы
  state.arr_data_small_debt = [];
  state.annuity_monthly_payment_old_credid = 0;
  state.differentiated_monthly_payment = [];
  state.annuity_debt_plus_interest_old_credit = 0;
  state.annuity_overpayment_old_credit = 0;
  state.payment_annuity_list = [];
  state.payment_differentiated_list = [];
  state.dateArray = [];
  state.dateArrayPsk = [];

  // Используем setTimeout с микро-задачей, чтобы не блокировать UI
  setTimeout(() => {
    try {
      // рассчитываю платежи
      const old_rate = (state.old_rate / 12) / 100;
      const creditTime = state.credit_old_time;
      const oldDebt = state.old_debt;
      const startPaymentDate = state.start_payment[0];
      const startPaymentDay = state.start_payment[1];

      // формирую даты (оптимизировано)
      const month_array = new Array(creditTime + 1);
      const month_year = new Array(creditTime + 1);
      
      for (let i = 0; i <= creditTime; i++) {
        const dateObj = new Date(startPaymentDate);
        dateObj.setMonth(dateObj.getMonth() + i);
        month_array[i] = dateObj.getUTCMonth() + 1;
        month_year[i] = dateObj.getUTCFullYear();
      }

      // Форматируем месяцы (оптимизировано)
      const monthFormatted = new Array(creditTime + 1);
      for (let i = 0; i <= creditTime; i++) {
        let month = month_array[i];
        if (i < creditTime && month === month_array[i + 1]) {
          month--;
        }
        monthFormatted[i] = month < 10 ? `0${month}` : String(month);
      }

      // Заполняем массив дат
      for (let i = 0; i <= creditTime; i++) {
        state.dateArray.push(`${startPaymentDay}.${monthFormatted[i]}.${month_year[i]}`);
      }

      // Рассчитываем аннуитет
      const power = (1 + old_rate) ** creditTime;
      const annuity_ratio = old_rate * power / (power - 1);
      const annuity_monthly_payment = oldDebt * annuity_ratio;
      
      state.annuity_monthly_payment_old_credid = annuity_monthly_payment;
      state.annuity_debt_plus_interest_old_credit = annuity_monthly_payment * creditTime;
      state.annuity_overpayment_old_credit = state.annuity_debt_plus_interest_old_credit - oldDebt;

      // Формирую график платежей (оптимизировано)
      state.payment_annuity_list = new Array(creditTime);
      let main_debt = oldDebt;
      
      for (let i = 1; i <= creditTime; i++) {
        const monthly_interest_payment = main_debt * old_rate;
        const principal_payment = annuity_monthly_payment - monthly_interest_payment;
        main_debt -= principal_payment;
        
        state.payment_annuity_list[i - 1] = {
          payment: parseFloat(annuity_monthly_payment.toFixed(2)),
          principal_payment: parseFloat(principal_payment.toFixed(2)),
          monthly_interest_payment: parseFloat(monthly_interest_payment.toFixed(2)),
          remainder: parseFloat(main_debt.toFixed(2)),
          month: state.dateArray[i]
        };
      }

      // ПСК (вынес в отдельную функцию для оптимизации)
      this._calculatePSK(state, monthFormatted, month_year, startPaymentDay, annuity_monthly_payment, oldDebt, creditTime);

      setTimeout(() => {
        state.final_window = false;
      }, 50);
      
    } catch (error) {
      console.error('Error in mutationCalculateOldCredit:', error);
      state.final_window = false;
    }
  }, 0);
},

// Выносим расчёт ПСК в отдельную функцию
_calculatePSK(state, monthFormatted, month_year, startPaymentDay, annuity_monthly_payment, oldDebt, creditTime) {
  try {
    // Заполняем dateArrayPsk
    state.dateArrayPsk = new Array(creditTime + 1);
    
    for (let i = 0; i <= creditTime; i++) {
      const dateStr = `${month_year[i]}-${monthFormatted[i]}-${startPaymentDay}`;
      if (i === 0) {
        state.dateArrayPsk[i] = {
          data: new Date(dateStr),
          payment: -oldDebt
        };
      } else {
        state.dateArrayPsk[i] = {
          data: new Date(dateStr),
          payment: annuity_monthly_payment
        };
      }
    }

    // Расчёт ПСК
    const m = state.dateArrayPsk.length;
    const bp = 30;
    const cbp = Math.round(365 / bp);
    
    const days = new Array(m);
    for (let k = 0; k < m; k++) {
      days[k] = (state.dateArrayPsk[k].data - state.dateArrayPsk[0].data) / (24 * 60 * 60 * 1000);
    }
    
    const e = new Array(m);
    const q = new Array(m);
    for (let k = 0; k < m; k++) {
      e[k] = (days[k] % bp) / bp;
      q[k] = Math.floor(days[k] / bp);
    }
    
    let i = 0;
    let x = 1;
    let x_m = 0;
    const s = 0.000001;
    let iterations = 0;
    const maxIterations = 1000; // Защита от бесконечного цикла
    
    while (x > 0 && iterations < maxIterations) {
      x_m = x;
      x = 0;
      for (let k = 0; k < m; k++) {
        x += state.dateArrayPsk[k].payment / ((1 + e[k] * i) * Math.pow(1 + i, q[k]));
      }
      i += s;
      iterations++;
    }
    
    if (x > x_m) {
      i -= s;
    }
    
    state.psk = Math.floor(i * cbp * 100 * 1000) / 1000;
    
  } catch (error) {
    console.error('Error calculating PSK:', error);
    state.psk = 0;
  }
}



actions: {
  ActionDebtOldCredit({ commit, state }, param) {
    commit('mutationDebtOldCredit', param);
    // Используем debounce или requestAnimationFrame
    requestAnimationFrame(() => {
      commit('mutationCalculateOldCredit');
    });
  },

  ActionRateOldCredit({ commit, state }, param) {
    commit('mutationRateOldCredit', param);
    requestAnimationFrame(() => {
      commit('mutationCalculateOldCredit');
    });
  },

  ActionTimeOldCredit({ commit, state }, param) {
    commit('mutationTimeOldCredit', param);
    requestAnimationFrame(() => {
      commit('mutationCalculateOldCredit');
    });
  },
  
  // Добавляем debounced версию для частых вызовов
  ActionDebouncedCalculate({ commit }) {
    if (this._calculateTimeout) {
      clearTimeout(this._calculateTimeout);
    }
    this._calculateTimeout = setTimeout(() => {
      commit('mutationCalculateOldCredit');
    }, 100);
  }
}
