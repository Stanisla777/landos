mutationIncome(state, received_perem) {
  console.log('received_perem', received_perem);
  
  if (received_perem !== null && received_perem !== '') {
    // Заменяем запятую на точку, чтобы parseFloat мог распарсить дробь
    const normalized = String(received_perem).replace(',', '.');
    state.family_income = parseFloat(normalized);
  } else {
    state.family_income = null; // или 0, в зависимости от логики
  }
  
  console.log('state.family_income', state.family_income);
}
