приложение:

<template>
  <div class="usefulness-survey__container js--usefulness-survey"
       :data-yes-label="yesLabel"
       :data-yes-required-hint="yesRequiredHint"
       :data-no-label="noLabel"
       :data-no-required-hint="noRequiredHint">
    <!-- остальной шаблон без изменений -->
  </div>
</template>

<script>
import universalForm from '../universal-form/universal-form';

export default {
  name: 'UsefulnessSurvey',
  props: {
    instanceId: {
      type: String,
      required: true
    },
    // Явно объявляем все data-атрибуты, которые могут прийти
    yesLabel: {
      type: String,
      default: ''
    },
    yesRequiredHint: {
      type: String,
      default: ''
    },
    noLabel: {
      type: String,
      default: ''
    },
    noRequiredHint: {
      type: String,
      default: ''
    }
  },
  data() {
    return {
      value_radio: '',
      data_yes_label: '',
      data_yes_required_hint: '',
      data_no_label: '',
      data_no_required_hint: '',
      data_required_hint: 'Поле не заполнено'
    };
  },
  methods: {
    changeUsefulness(event) {
      const element = event.currentTarget;
      if (element.hasAttribute('data-value') && this.$refs.feedBackTextarea) {
        this.value_radio = element.getAttribute('data-value');
        
        if (this.value_radio === 'yes') {
          this.data_required_hint = this.yesRequiredHint;
          if (this.$refs.feedBackTextarea.querySelector('.js--input__error_required')) {
            this.$refs.feedBackTextarea.querySelector('.js--input__error_required').textContent = this.data_required_hint;
          }
        } else if (this.value_radio === 'no') {
          this.data_required_hint = this.noRequiredHint;
          if (this.$refs.feedBackTextarea.querySelector('.js--input__error_required')) {
            this.$refs.feedBackTextarea.querySelector('.js--input__error_required').textContent = this.data_required_hint;
          }
        }
      }
    }
  },
  mounted() {
    // Теперь данные уже в this
    this.data_yes_label = this.yesLabel;
    this.data_yes_required_hint = this.yesRequiredHint;
    this.data_no_label = this.noLabel;
    this.data_no_required_hint = this.noRequiredHint;

    // Проверяем, какой радио-элемент выбран по умолчанию
    if (this.$refs.usefulnessSurvey) {
      const array_radio = this.$refs.usefulnessSurvey.querySelectorAll('input[type="radio"]');
      for (let item of array_radio) {
        if (item.checked) {
          if (item.hasAttribute('data-value')) {
            this.value_radio = item.getAttribute('data-value');
            if (this.value_radio === 'yes') {
              this.data_required_hint = this.yesRequiredHint;
            } else if (this.value_radio === 'no') {
              this.data_required_hint = this.noRequiredHint;
            }
          }
        }
      }
    }

    universalForm();
  }
};
</script>
