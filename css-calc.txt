selectDay(day) {
  if (!day.hasEvents) return;

  // Устанавливаем выбранный день
  this.selectedDate = day.date;

  // Проверяем, относится ли день к другому месяцу
  const selectedDateObj = this.parseDate(day.date);
  const selectedMonth = selectedDateObj.getMonth();
  const selectedYear = selectedDateObj.getFullYear();

  // Если месяц или год не совпадают с текущим — переключаем
  if (selectedMonth !== this.currentMonth || selectedYear !== this.currentYear) {
    this.currentMonth = selectedMonth;
    this.currentYear = selectedYear;
    // selectedDate уже установлен — не трогаем
  }
}
