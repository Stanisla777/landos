forceWrapperHeight() {
    this.$nextTick(() => {
      // Небольшая задержка для гарантии, что контент отрисован
      setTimeout(() => {
        const container = this.$refs.SlideTestWindow;
        if (!container) return;

        // Находим активный слайд внутри контейнера
        const activeSlide = container.querySelector('.swiper-slide-active');
        if (!activeSlide) return;

        // Замеряем реальную высоту контента (включая margin/padding)
        const contentHeight = activeSlide.scrollHeight;
        
        // Находим wrapper и применяем высоту
        const wrapper = container.querySelector('.swiper-wrapper');
        if (wrapper) {
          // Сохраняем старую высоту для плавного перехода (опционально)
          const oldHeight = wrapper.offsetHeight;
          if (oldHeight !== contentHeight) {
            wrapper.style.transition = 'height 0.2s ease';
            wrapper.style.height = `${contentHeight}px`;
            
            // Убираем transition после анимации, чтобы не мешал другим изменениям
            setTimeout(() => {
              wrapper.style.transition = '';
            }, 200);
          }
        }
        
        // Отладка (удалите после проверки)
        console.log('Height recalculated:', {
          slide: activeSlide.classList,
          height: contentHeight,
          wrapper: wrapper?.style?.height
        });
      }, 30); // 30ms — достаточно для рендера, но не заметно для пользователя
    });
  },

mounted() {
  this.$nextTick(() => {
    this.initSlider();
    this.countTrueAnswer();
    this.countQuestion();
    
    if (this.$el.closest('.content-block-new')) {
      this.for_insruction = true;
    }
    
    // Пересчитываем высоту после первой отрисовки
    this.forceWrapperHeight();
  });
},
