<script>
export default {
  name: "component-download",
  data() {
    return {
      isDownloading: false,
      downloadError: null
    };
  },
  methods: {
    async downloadArchive() {
      this.isDownloading = true;
      this.downloadError = null;

      // Путь к файлу. 
      // ВАЖНО: В Vue файлы для прямого доступа по URL должны лежать в папке `public`!
      const fileUrl = 'http://localhost:8081/img/documents/archive.zip'; 
      const fileName = 'documents_archive.zip'; // Имя, под которым файл сохранится у пользователя

      try {
        // 1. Делаем запрос на получение файла
        const response = await fetch(fileUrl);
        
        // 2. Проверяем, успешен ли запрос (код 200-299)
        if (!response.ok) {
          throw new Error(`Ошибка сети: ${response.status}`);
        }

        // 3. Преобразуем ответ в Blob (бинарный объект)
        const blob = await response.blob();

        // 4. Создаем временную ссылку на этот Blob в памяти браузера
        const downloadUrl = window.URL.createObjectURL(blob);

        // 5. Создаем невидимый элемент <a> для инициирования скачивания
        const link = document.createElement('a');
        link.href = downloadUrl;
        link.download = fileName; // Атрибут download заставляет браузер скачать файл, а не открывать его
        
        // 6. Добавляем ссылку в DOM, кликаем по ней и сразу удаляем
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);

        // 7. Очищаем память, удаляя временную ссылку на Blob
        window.URL.revokeObjectURL(downloadUrl);

      } catch (error) {
        console.error('Ошибка при скачивании:', error);
        this.downloadError = 'Не удалось скачать файл. Попробуйте позже.';
      } finally {
        // В любом случае (успех или ошибка) убираем состояние загрузки
        this.isDownloading = false;
      }
    }
  }
}
</script>
