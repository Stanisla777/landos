Все приложения добавляются в объект в массив vueComponents с тремя полями:
selector — CSS-селектор (обычно совпадает с именем тега)
component — сам импортированный компонент
getProps — функция, которая собирает атрибуты с DOM-элемента


// Новое приложение 1
  {
    selector: 'search-family-mortgage',
    component: SearchFamilyMortgage, // импортированный компонент
    getProps: (el) => ({
      param1: el.getAttribute('param1') || '',
      param2: el.getAttribute('param2') || '',
      // любые другие атрибуты, которые передаёт бэкенд
    })
  },
  
  // Новое приложение 2
  {
    selector: 'v-mortgage-calculator',
    component: MortgageCalculator,
    getProps: (el) => ({
      minSum: el.getAttribute('min-sum') || '1000000',
      maxSum: el.getAttribute('max-sum') || '10000000',
      rate: el.getAttribute('rate') || '7.5'
    })
  },
