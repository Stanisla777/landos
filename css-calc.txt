            if(massive_2.length>0){

            }
          }
        }
        else if(item.FIELD_TYPE=="1"){
          const keys_2 = Object.values(item.ANSWERS)
          for(let i=0;i<keys_2.length;i+=1){
            if(keys_2[i].FIELD_WIDTH=="1"){

            }
          }
        }
