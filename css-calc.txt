initAgeMask() {
  this.ageMaskRef = IMask(this.$refs.dataFieldCount, {
    mask: Number,
    min: 0,
    max: 999999, // ← временно ставим большое число, чтобы не блокировать ввод
    scale: 0,
    signed: false,
    thousandsSeparator: '',
    radix: '.',
    mapToRadix: [],
    normalizeZeros: true,
    autofix: false, // отключаем, если не поддерживается
  });

  this.ageMaskRef.on('accept', () => {
    const unmasked = this.ageMaskRef.unmaskedValue;
    if (!unmasked) {
      this.dataFieldCount = '';
      return;
    }

    let num = parseInt(unmasked, 10);
    if (isNaN(num)) num = 0;

    // Ограничиваем до 30
    if (num > 30) {
      num = 30;
      // Принудительно обновляем поле
      this.ageMaskRef.unmaskedValue = '30';
    }

    this.dataFieldCount = this.ageMaskRef.value;
  });
}
