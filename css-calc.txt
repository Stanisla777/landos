initAgeMask() {
  this.ageMaskRef = IMask(this.$refs.dataFieldCount, {
    mask: Number,
    min: 0,
    max: 999999, // ← временно ставим большое число, чтобы не блокировать ввод
    scale: 0,
    signed: false,
    thousandsSeparator: '',
    radix: '.',
    mapToRadix: [],
    normalizeZeros: true,
    autofix: false, // отключаем, если не поддерживается
  });

  this.ageMaskRef.on('accept', () => {
    const unmasked = this.ageMaskRef.unmaskedValue;
    if (!unmasked) {
      this.dataFieldCount = '';
      return;
    }

    let num = parseInt(unmasked, 10);
    if (isNaN(num)) num = 0;

    // Ограничиваем до 30
    if (num > 30) {
      num = 30;
      // Принудительно обновляем поле
      this.ageMaskRef.unmaskedValue = '30';
    }

    this.dataFieldCount = this.ageMaskRef.value;
  });
}


mounted() {
  this.initAgeMask();
  this.$nextTick(() => {
    if (this.answers && !this.region_not) {
      const value = this.answers.numberFamily;
      if (value != null) {
        // Убедимся, что значение в диапазоне
        const safeValue = Math.min(30, Math.max(0, parseInt(value, 10) || 0));
        this.ageMaskRef.unmaskedValue = String(safeValue);
        this.dataFieldCount = this.ageMaskRef.value;
      }
    }
  });
}
