<div class="calc-tax-deduc-new__col-input js--tex-deduc-input">
        <input type="text" placeholder="XXXXX.XX.XXXX"
               ref="dataFieldCount"
               @blur="moveAnotherElement"
               @focus="inpFocus"/>
        <div class="calc-tax-deduc-new__input-clear js--clear-calc-tax"
             :class="dataFieldCount!=='' ?
         'active' : ''"
             @click="cleaerField">
        </div>
      </div>

<script>
import IMask from "imask";
data(){
    return {
      dataFieldCount: '',
      ageMaskRef: null
    }
  },
methods:{
    initAgeMask() {
      this.ageMaskRef = IMask(this.$refs.dataFieldCount, {
        mask: Number,
        min: 0,
        max: 999999, // ← временно ставим большое число, чтобы не блокировать ввод
        scale: 0,
        signed: false,
        thousandsSeparator: '',
        radix: '.',
        mapToRadix: [],
        normalizeZeros: true,
        autofix: false, // отключаем, если не поддерживается
      });

      this.ageMaskRef.on('accept', () => {
        const unmasked = this.ageMaskRef.unmaskedValue;
        if (!unmasked) {
          this.dataFieldCount = '';
          return;
        }
        let num = parseInt(unmasked, 10);
        if (isNaN(num)) num = 0;
        // Ограничиваем до 30
        if (num > 30) {
          num = 30;
          // Принудительно обновляем поле
          this.ageMaskRef.unmaskedValue = '30';
        }
        this.dataFieldCount = this.ageMaskRef.value;
      });
    },

    cleaerField(){
      if (this.ageMaskRef) {
        this.ageMaskRef.value = '';
      }
    }
  },
