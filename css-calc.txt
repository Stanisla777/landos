document.querySelectorAll('.enabled').forEach((element) => {
  const prevElement = element.previousElementSibling;
  const nextElement = element.nextElementSibling;
  
  if (prevElement && nextElement && 
      prevElement.classList.contains('disabled') && 
      nextElement.classList.contains('disabled')) {
    element.classList.add('none');
  }
});

--------------------------------

// Этот селектор найдет все .enabled, между двумя .disabled
document.querySelectorAll('.disabled + .enabled + .disabled').forEach((disabledAfter) => {
  const enabledElement = disabledAfter.previousElementSibling;
  if (enabledElement.classList.contains('enabled')) {
    enabledElement.classList.add('none');
  }
});
