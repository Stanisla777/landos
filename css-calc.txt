mounted() {
  this.$nextTick(() => {
    const tooltips = {};
    const attributes = this.$el.attributes;

    for (let i = 0; i < attributes.length; i++) {
      const attr = attributes[i];

      // ✅ Изменили префикс на 'data-tooltip-'
      if (attr.name.startsWith('data-tooltip-')) {
        try {
          const value = JSON.parse(attr.value);
          
          // ✅ Изменили удаление префикса
          const name = attr.name.replace('data-tooltip-', '');
          
          if (!value.name) {
            value.name = name;
          }
          
          tooltips[name] = value;
        } catch (error) {
          console.error(`Ошибка парсинга атрибута ${attr.name}:`, error);
        }
      }
    }

    // ✅ Отправляем в store (проверьте, что используете this.$store)
    if (Object.keys(tooltips).length > 0) {
      this.$store.dispatch('ActionTransferTooltip', tooltips);
    }
  });
}
