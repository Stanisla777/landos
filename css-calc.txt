// Индикаторы шага в заголовке
const stepIndicator1 = modal.querySelector('.js--step-indicator-1');
const stepIndicator2 = modal.querySelector('.js--step-indicator-2');


function updateStepVisibility() {
  // Переключение контента шагов
  if (step1 && step2) {
    step1.style.display = state.step_1 ? 'block' : 'none';
    step2.style.display = state.step_1 ? 'none' : 'block';
  }

  // Переключение индикаторов "Шаг X из 2"
  if (stepIndicator1) stepIndicator1.style.display = state.step_1 ? 'block' : 'none';
  if (stepIndicator2) stepIndicator2.style.display = state.step_1 ? 'none' : 'block';
}
