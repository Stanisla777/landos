import '../scss/style.scss';
import { createApp } from 'vue';
import TelegramBlock from './modules/TelegramBlock.vue';

// Просто монтируем, если элемент существует
document.addEventListener('DOMContentLoaded', () => {
  const elements = document.querySelectorAll('telegram-block');
  
  elements.forEach((element, index) => {
    // Создаём уникальный ID для каждого экземпляра
    const id = `telegram-block-${index}`;
    element.id = id;
    
    // Собираем пропсы из атрибутов
    const props = {
      qr: element.getAttribute('qr') || '',
      mobilicon: element.getAttribute('mobilicon') || '',
      link: element.getAttribute('link') || '',
      description: element.getAttribute('description') || ''
    };
    
    // Создаём и монтируем приложение
    const app = createApp(TelegramBlock, props);
    app.mount(`#${id}`);
  });
});
