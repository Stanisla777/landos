receivedClearHolidayReceive(index){
  //передаю в календарь периодов команду и флаг, что в основном календаре блока досрочного погашения стёрли дату
  this.min_date_holidays_begin = this.min_date_holidays
  this.min_date_holidays_end = this.min_date_holidays
  eventBus.emit('sendToBeginCalendarClear',index)
  this.$emit('sendClearCompareWhithHoliday',index)
  eventBus.emit('sendClearToEnd', index)
  
  //обнуляю календарь конец
  // замена this.$children.find на this.$refs
  const refNameMain = `js-calendar-early-repayment-${index}`;
  const targetComponentMain = this.$refs[refNameMain]?.[0];
  if (targetComponentMain && typeof targetComponentMain.updateCalendar === 'function') {
    targetComponentMain.updateCalendar()
  }
  
  //обнуляю календарь начала
  const refNameBegin = `js-calendar-early-period-begin-${index}`;
  const targetComponentBegin = this.$refs[refNameBegin]?.[0];
  if (targetComponentBegin && targetComponentBegin.$el) {
    const input = targetComponentBegin.$el.querySelector('input.js--calendar-field-begin');
    if (input) {
      input.value = '';
      targetComponentBegin.$el.querySelector('.js-early-period-required')?.classList.remove('js-filled');
      targetComponentBegin.$el.querySelector('input')?.classList.remove('active');
      targetComponentBegin.$el.querySelector('.js--clear-calc-tax')?.classList.remove('active');
    }
    if (typeof targetComponentBegin.updateCalendar === 'function') {
      targetComponentBegin.updateCalendar()
    }
  }
  
  const existingItemsFieldData = this.array_block_early_repayment.find(item => item.id === index);
  if (existingItemsFieldData) {
    existingItemsFieldData.field_data=''
  }
  const existingItemsFieldPeriod = this.array_block_early_repayment_period.find(item => item.id === index);
  if (existingItemsFieldPeriod) {
    existingItemsFieldPeriod.selected_value=null
  }
  const indexArray = this.block_requared.findIndex(obj => obj.id === index);
  if (indexArray !== -1) {
    this.block_requared[indexArray].date_full = false;
    this.block_requared[indexArray].date_begin = false;
  }
  this.$emit('sendButtonResult')
},

receivedEearlyRepayment(array_data) {
  this.array_early_repayment_date = this.addOrReplace(this.array_early_repayment_date,array_data)
  this.array_selected_dates.push(array_data[0])
  this.$emit('sendClearCompareWhithHoliday',array_data[1])
  
  // ✅ ИСПРАВЛЕНО: замена this.$children.find на this.$refs
  const refNameEnd = `js-calendar-early-period-end-${array_data[1]}`;
  const targetComponentEnd = this.$refs[refNameEnd]?.[0];
  if (targetComponentEnd && typeof targetComponentEnd.newMethods === 'function') {
    targetComponentEnd.newMethods([array_data[1],array_data[0]]);
  }
  
  const refNameBegin = `js-calendar-early-period-begin-${array_data[1]}`;
  const targetComponentMain = this.$refs[refNameBegin]?.[0];
  if (targetComponentMain && targetComponentMain.$el) {
    const parent = targetComponentMain.$el.closest('.js--credit-calendar-input');
    if(parent){
      const element_eray = parent.querySelector('input.js--calendar-field-begin');
      if (element_eray){
        element_eray.value = `${this.month[parseInt(array_data[0][0])]}, ${array_data[0][1]}`;
        if(element_eray.classList.contains('js-early-required')) {
          element_eray.classList.add('js-filled')
        }
        if (element_eray.closest('.js--container-block-required')
          && element_eray.closest('.js--container-block-required').querySelector('.js--required-error')) {
          element_eray.closest('.js--container-block-required').querySelector('.js--required-error').classList.add('display-none')
        }
        if (element_eray.closest('.js--for-clear-field')) {
          element_eray.closest('.js--for-clear-field').classList.remove('input_error')
        }
        if (element_eray.closest('.js--for-clear-field') && element_eray.closest('.js--for-clear-field').querySelector('.js--clear-calc-tax')) {
          element_eray.closest('.js--for-clear-field').querySelector('.js--clear-calc-tax').classList.add('active')
        }
        if (element_eray.classList.contains('js-early-period-required')) {
          element_eray.classList.add('js-filled')
        }
      }
    }
  }

  // ... остальной код без изменений ...
},
