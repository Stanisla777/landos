setDownloadUrl(state, url) {
      state.down_load_url = url;
    },

async mutationDownloadDocument({ commit, state }) {
      try {
        const response = await axios({
          method: 'post',
          url: '/api/local/program450/statusrequest/printform/',
          headers: {
            'Content-Type': 'application/json; charset=UTF-8',
            'X-Bitrix-Csrf-Token': window.BX?.bitrix_sessid?.(),
          },
          data: {},
        });

        const data = response.data.result;

        if (data?.url) {
          commit('setDownloadUrl', data.url);
          commit('clearPolling'); // Останавливаем поллинг
          console.log('Ссылка получена:', data.url);
        }
      } catch (error) {
        const errorCode = error.response?.data?.error?.code;
        handleApiError(state, errorCode);
        commit('clearPolling'); // Останавливаем при ошибке
        console.error('Ошибка:', error);
      }
    }
