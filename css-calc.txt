// Вставьте в testBuild() или в консоль
document.querySelectorAll('.wrap-test').forEach(wrap => {
  const observer = new MutationObserver((mutations) => {
    mutations.forEach((m) => {
      m.removedNodes.forEach((node) => {
        // Проверяем сам узел и его потомков
        const isTarget = node.classList?.contains('block-add-favorites') || 
                        node.querySelector?.('.block-add-favorites');
        
        if (isTarget) {
          const stack = new Error().stack;
          console.error('🚨 .block-add-favorites удалён!', {
            timestamp: Date.now(),
            removedBy: m.target.className || m.target.id,
            // Пытаемся найти имя функции в стеке (работает даже с минификацией)
            functionInStack: stack?.match(/at\s+([^\s(]+)/)?.[1] || 'unknown',
            // Показываем 3 уровня стека
            stackPreview: stack?.split('\n').slice(1, 4).map(s => s.trim()),
            // Какой именно узел был удалён
            removedNode: node.nodeName + '.' + (node.className || ''),
            // HTML удалённого узла (первые 100 символов)
            removedHTML: node.outerHTML?.slice(0, 100)
          });
          
          // 🎯 КЛЮЧЕВОЙ МОМЕНТ: ставим точку останова программно
          // После этого логирования выполнение остановится, и вы сможете
          // посмотреть Call Stack в DevTools → вкладка Sources
          debugger; 
        }
      });
    });
  });
  observer.observe(wrap, { childList: true, subtree: true });
  console.log('👁 Observer attached to', wrap);
});
